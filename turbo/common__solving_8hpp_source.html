42#ifdef _WINDOWS
43//
44// The Microsoft Windows API does not support sigprocmask() or sigpending(),
45// so we have to fall back to traditional signal handling.
46//
47static std::atomic<bool> got_signal;
48static void (*prev_sigint)(int);
49static void (*prev_sigterm)(int);
50
51void signal_handler(int signum)
52{
53 signal(SIGINT, signal_handler); // re-arm
54 signal(SIGTERM, signal_handler); // re-arm
55
56 got_signal = true; // volatile
57
58 if (signum == SIGINT && prev_sigint != SIG_DFL && prev_sigint != SIG_IGN) {
59 (*prev_sigint)(signum);
60 }
61 if (signum == SIGTERM && prev_sigterm != SIG_DFL && prev_sigterm != SIG_IGN) {
62 (*prev_sigterm)(signum);
63 }
64}
65
66void block_signal_ctrlc() {
67 prev_sigint = signal(SIGINT, signal_handler);
68 prev_sigterm = signal(SIGTERM, signal_handler);
69}
70
71bool must_quit() {
72 return static_cast<bool>(got_signal);
73}
74
75#else
76
77void block_signal_ctrlc() {
78 sigset_t ctrlc;
79 sigemptyset(&ctrlc);
80 sigaddset(&ctrlc, SIGINT);
81 sigaddset(&ctrlc, SIGTERM);
82 if(sigprocmask(SIG_BLOCK, &ctrlc, NULL) != 0) {
83 printf("%% ERROR: Unable to deal with CTRL-C. Therefore, we will not be able to print the latest solution before quitting.\n");
84 perror(NULL);
85 return;
86 }
87}

89bool must_quit() {
90 sigset_t pending_sigs;
91 sigemptyset(&pending_sigs);
92 sigpending(&pending_sigs);
93 return sigismember(&pending_sigs, SIGINT) == 1 || sigismember(&pending_sigs, SIGTERM) == 1;
94}

97/** Check if the timeout of the current execution is exceeded and returns `false` otherwise.
98 * It also update the statistics relevant to the solving duration and the exhaustive flag if we reach the timeout.
99 */
100template <class A, class Timepoint>
101bool check_timeout(A& a, const Timepoint& start) {
102 a.stats.update_timer(Timer::OVERALL, start);
103 if(a.config.timeout_ms == 0) {
104 return true;
105 }
106 if(a.stats.time_ms_of(Timer::OVERALL) >= static_cast<int64_t>(a.config.timeout_ms)) {
107 if(a.config.verbose_solving) {
108 printf("%% CPU: Timeout reached.\n");
109 }
110 a.stats.exhaustive = false;
111 return false;
112 }
113 return true;
114}

144/** This class is parametrized by a universe of discourse, which is the domain of the variables in the store and various allocators:
145 * - BasicAllocator: default allocator, used to allocate abstract domains, the environment, storing intermediate results, etc.
146 * - PropAllocator: allocator used for the PC abstract domain, to allocate the propagators.
147 * - StoreAllocator: allocator used for the store, to allocate the variables.
148 *
149 * Normally, you should use the fastest memory for the store, then for the propagators and then for the rest.
150 */
151template <class Universe,
152 class BasicAllocator,
153 class PropAllocator,
154 class StoreAllocator>
155struct AbstractDomains {
156 using universe_type = typename Universe::local_type;

159 using LIStore = VStore<universe_type, BasicAllocator>;

161 using IStore = VStore<Universe, StoreAllocator>;
162 using IPC = PIR<IStore, PropAllocator>; // Interval Propagators Completion
163 using ISimplifier = Simplifier<IPC, BasicAllocator>;
164 using Split = SplitStrategy<IPC, BasicAllocator>;
165 using IST = SearchTree<IPC, Split, BasicAllocator>;
166 using IBAB = BAB<IST, LIStore>;
167
168 using basic_allocator_type = BasicAllocator;
169 using prop_allocator_type = PropAllocator;
170 using store_allocator_type = StoreAllocator;

178 /** We copy `other` in a new element, and ignore every variable not used in a GPU block.
179 * This is because copying everything in each block is very slow.
180 *
181 * NOTE: It is not the allocation itself that is slow, I think it calling many copy constructors for atomic variables (note that in simplifier we have an atomic memory if the underlying domain has one).

285 CUDA void allocate(int num_vars) {
286 env = VarEnv<basic_allocator_type>{basic_allocator};
287 store = battery::allocate_shared<IStore, StoreAllocator>(store_allocator, env.extends_abstract_dom(), num_vars, store_allocator);
288 ipc = battery::allocate_shared<IPC, PropAllocator>(prop_allocator, env.extends_abstract_dom(), store, prop_allocator);
289 // If the simplifier is already allocated, it means we are currently reallocating the abstract domains after preprocessing.
290 if(!simplifier) {
291 simplifier = battery::allocate_shared<ISimplifier, BasicAllocator>(basic_allocator, env.extends_abstract_dom(), ipc, basic_allocator);
292 }
293 split = battery::allocate_shared<Split, BasicAllocator>(basic_allocator, env.extends_abstract_dom(), ipc, basic_allocator);
294 eps_split = battery::allocate_shared<Split, BasicAllocator>(basic_allocator, env.extends_abstract_dom(), ipc, basic_allocator);
295 search_tree = battery::allocate_shared<IST, BasicAllocator>(basic_allocator, env.extends_abstract_dom(), ipc, split, basic_allocator);
296 // Note that `best` must have the same abstract type then store (otherwise projection of the variables will fail).
297 best = battery::allocate_shared<LIStore, BasicAllocator>(basic_allocator, store->aty(), num_vars, basic_allocator);
298 bab = battery::allocate_shared<IBAB, BasicAllocator>(basic_allocator, env.extends_abstract_dom(), search_tree, best);
299 if(config.verbose_solving) {
300 printf("%% Abstract domain allocated.\n");
301 }
302 }

316 // Mainly to interpret the IN constraint in IPC instead of only over-approximating in intervals.
317 template <class F>
318 CUDA void typing(F& f) const {
319 switch(f.index()) {
320 case F::Seq:
321 if(f.sig() == ::lala::IN && f.seq(1).is(F::S) && f.seq(1).s().size() > 1) {
322 f.type_as(ipc->aty());
323 return;
324 }
325 for(int i = 0; i < f.seq().size(); ++i) {
326 typing(f.seq(i));
327 }
328 break;
329 case F::ESeq:
330 for(int i = 0; i < f.eseq().size(); ++i) {
331 typing(f.eseq(i));
332 }
333 break;
334 }
335 }

338 // We first try to interpret, and if it does not work, we interpret again with the diagnostics mode turned on.
339 template <class F, class Env, class A>
340 CUDA bool interpret_and_diagnose_and_tell(const F& f, Env& env, A& a) {
341 IDiagnostics diagnostics;
342 if(!interpret_and_tell(f, env, a, diagnostics)) {
343 IDiagnostics diagnostics2;
344 interpret_and_tell<true>(f, env, a, diagnostics2);
345 diagnostics2.print();
346 return false;
347 }
348 return true;
349 }

352 template <class F>
353 CUDA bool interpret(const F& f) {
354 if(config.verbose_solving) {
355 printf("%% Interpreting the formula...\n");
356 }
357 if(!interpret_and_diagnose_and_tell(f, env, *bab)) {
358 return false;
359 }
360 /** If some variables were added during the interpretation, we must resize `best` as well.
361 * If we don't do it now, it will be done during the solving (when calling bab.extract) which will lead to a resize of the underlying store.
362 * The problem is that the resize will be done on the device! If it was allocated in managed memory, it will be now reallocated in device memory, leading to a segfault later on.
363 */
364 if(store->vars() != best->vars()) {
365 store->extract(*best);
366 best->join_top();
367 }
368 stats.variables = store->vars();
369 stats.constraints = ipc->num_deductions();
370 bool can_interpret = true;
371 /** We add a search strategy by default for the variables that potentially do not occur in the previous strategies. */
372 can_interpret &= interpret_default_strategy<F>();
373 can_interpret &= interpret_default_eps_strategy<F>();
374 return can_interpret;
375 }

377 template <class F>
378 CUDA bool prepare_simplifier(F& f) {
379 if(config.verbose_solving) {
380 printf("%% Simplifying the formula...\n");
381 }
382 IDiagnostics diagnostics;
383 typename ISimplifier::template tell_type<basic_allocator_type> tell{basic_allocator};
384 if(top_level_ginterpret_in<IKind::TELL>(*simplifier, f, env, tell, diagnostics)) {
385 simplifier->deduce(std::move(tell));
386 return true;
387 }
388 else if(config.verbose_solving) {
389 printf("WARNING: Could not simplify the formula because:\n");
390 IDiagnostics diagnostics2;
391 top_level_ginterpret_in<IKind::TELL, true>(*simplifier, f, env, tell, diagnostics2);
392 diagnostics2.print();
393 }
394 return false;
395 }

423 FormulaPtr prepare_solver() {
424 // I. Parse the FlatZinc model.
425 FormulaPtr f;
426 if(config.input_format() == InputFormat::FLATZINC) {
427 f = parse_flatzinc(config.problem_path.data(), solver_output);
428 }
429#ifdef WITH_XCSP3PARSER
430 else if(config.input_format() == InputFormat::XCSP3) {
431 f = parse_xcsp3(config.problem_path.data(), solver_output);
432 }
433#endif
434 if(!f) {
435 std::cerr << "Could not parse input file." << std::endl;
436 exit(EXIT_FAILURE);
437 }
438
439 if(config.verbose_solving) {
440 printf("%% Input file parsed\n");
441 }
442
443 if(config.print_ast) {
444 printf("%% Parsed AST:\n");
445 f->print(true);
446 printf("\n");
447 }
448 *f = eval(*f);
449 if(config.verbose_solving) {
450 printf("%% Formula syntactically simplified.\n");
451 }
452 stats.print_stat("parsed_variables", num_quantified_vars(*f));
453 stats.print_stat("parsed_constraints", num_constraints(*f));
454 *f = normalize(ternarize(*f));
455 size_t num_vars = num_quantified_vars(*f);
456 /** TNF = ternary normal form (apply normalize and ternarize functions). */
457 stats.print_stat("tnf_variables", num_vars);
458 stats.print_stat("tnf_constraints", num_constraints(*f));
459 allocate(num_vars);
460 type_and_interpret(*f);
461 return f;
462 }

464 void preprocess() {
465 auto start = stats.start_timer_host();
466 auto raw_formula = prepare_solver();
467 if(prepare_simplifier(*raw_formula)) {
468 GaussSeidelIteration fp_engine;
469 fp_engine.fixpoint(ipc->num_deductions(), [&](size_t i) { return ipc->deduce(i); });
470 if(config.simplify) {
471 fp_engine.fixpoint(simplifier->num_deductions(), [&](size_t i) { return simplifier->deduce(i); });
472 auto f = simplifier->deinterpret();
473 stats.eliminated_variables = simplifier->num_eliminated_variables();
474 stats.eliminated_formulas = simplifier->num_eliminated_formulas();
475 if(config.verbose_solving) {
476 printf("%% Formula simplified.\n");
477 printf("%% Ternarizing the formula...\n");
478 }
479 using F = TFormula<basic_allocator_type>;
480 f = ternarize(f);
481 if(config.verbose_solving) {
482 printf("%% Formula ternarized.\n");
483 if(config.print_ast) {
484 f.print(false);
485 }
486 printf("%% Normalizing the formula...\n");
487 }
488 f = normalize(f);
489 if(config.verbose_solving) {
490 printf("%% Formula normalized.\n");
491 if(config.print_ast) {
492 f.print(false);
493 }
494 }
495 size_t num_vars = num_quantified_vars(f);
496 stats.print_stat("simplified_tnf_vars", num_vars);
497 stats.print_stat("simplified_tnf_constraints", num_constraints(f));
498 allocate(num_vars);
499 type_and_interpret(f);
500 }
501 }
502 stats.stop_timer(Timer::PREPROCESSING, start);
503 stats.print_timing_stat("preprocessing_time", Timer::PREPROCESSING);
504 }

508 CUDA bool interpret_default_strategy() {
509 config.free_search = true;
510 typename F::Sequence seq;
511 seq.push_back(F::make_nary("first_fail", {}));
512 seq.push_back(F::make_nary("indomain_min", {}));
513 for(int i = 0; i < env.num_vars(); ++i) {
514 seq.push_back(F::make_avar(env[i].avars[0]));
515 }
516 F search_strat = F::make_nary("search", std::move(seq));
517 if(!interpret_and_diagnose_and_tell(search_strat, env, *bab)) {
518 return false;
519 }
520 return true;
521 }
