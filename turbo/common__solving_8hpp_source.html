59 void signal_handler(int signum)
60 {
61  std::signal(SIGINT, signal_handler); // re-arm
62  std::signal(SIGTERM, signal_handler); // re-arm
63  got_signal = true; // volatile
64  if (signum == SIGINT && prev_sigint != SIG_DFL && prev_sigint != SIG_IGN) {
65  (*prev_sigint)(signum);
66  }
67  if (signum == SIGTERM && prev_sigterm != SIG_DFL && prev_sigterm != SIG_IGN) {
68  (*prev_sigterm)(signum);
69  }
70 }

311  // Mainly to interpret the IN constraint in IProp instead of only over-approximating in intervals.
312  template <class F>
313  CUDA void typing(F& f, bool toplevel = true) const {
314  if(toplevel && config.verbose_solving) {
315  printf("%% Typing the formula...\n");
316  }
317  switch(f.index()) {
318  case F::Seq:
319  if(f.sig() == ::lala::IN && f.seq(1).is(F::S) && f.seq(1).s().size() > 1) {
320  f.type_as(iprop->aty());
321  return;
322  }
323  for(int i = 0; i < f.seq().size(); ++i) {
324  typing(f.seq(i), false);
325  }
326  break;
327  case F::ESeq:
328  for(int i = 0; i < f.eseq().size(); ++i) {
329  typing(f.eseq(i), false);
330  }
331  break;
332  }
333  if(toplevel && config.print_ast) {
334  printf("%% Typed AST:\n");
335  f.print(true);
336  printf("\n");
337  }
338  }

341  template <class F, class Env, class A>
342  CUDA bool interpret_and_diagnose_and_tell(const F& f, Env& env, A& a) {
343  IDiagnostics diagnostics;
344  if(!interpret_and_tell(f, env, a, diagnostics)) {

354  template <class F>
355  CUDA bool interpret(const F& f, bool with_default_strats) {
356  if(config.verbose_solving) {
357  printf("%% Interpreting the formula...\n");
358  }
359  if(!interpret_and_diagnose_and_tell(f, env, *bab)) {
360  return false;
361  }
362  if(config.print_ast) {
363  printf("%% Interpreted AST:\n");
364  iprop->deinterpret(env).print();
365  printf("\n");
366  }
367  if(config.verbose_solving) {
368  printf("%% Formula has been interpreted.\n");
369  }
370  /** If some variables were added during the interpretation, we must resize `best` as well.
371  * If we don't do it now, it will be done during the solving (when calling bab.extract) which will lead to a resize of the underlying store.
372  * The problem is that the resize will be done on the device! If it was allocated in managed memory, it will be now reallocated in device memory, leading to a segfault later on.
373  */
374  if(store->vars() != best->vars()) {
375  store->extract(*best);
376  best->join_top();
377  }
378  stats.variables = store->vars();
379  stats.constraints = iprop->num_deductions();
380  bool can_interpret = true;
381  if(with_default_strats) {
382  /** We add a search strategy by default for the variables that potentially do not occur in the previous strategies. */
383  can_interpret &= interpret_default_strategy<F>();
384  can_interpret &= interpret_default_eps_strategy<F>();
385  }
386  return can_interpret;
387  }

440  void preprocess_ipc(F& f) {
441  size_t num_vars = num_quantified_vars(f);
442  allocate(num_vars, true);
443  typing(f);
444  /** If we don't simplify, it is our last chance to interpret the default search strategies. */
445  if(!interpret(f, config.disable_simplify)) {
446  exit(EXIT_FAILURE);
447  }
448  GaussSeidelIteration fp_engine;
449  fp_engine.fixpoint(iprop->num_deductions(), [&](size_t i) { return iprop->deduce(i); });
450  /* We need to initialize the simplifier even if we don't simplify.
451  This is because the simplifier equivalence classes is used in SolverOutput. */
452  initialize_simplifier(f);
453  if(config.disable_simplify) {
454  return;
455  }
456  if(config.verbose_solving) {
457  printf("%% Simplifying the formula...\n");
458  }
459  fp_engine.fixpoint(simplifier->num_deductions(), [&](size_t i) { return simplifier->deduce(i); });
460  f = simplifier->deinterpret();
461  if(config.verbose_solving) {
462  printf("%% Formula simplified.\n");
463  }
464  f = normalize(f);
465  num_vars = num_quantified_vars(f);
466  stats.print_stat("variables_after_simplification", num_vars);
467  stats.print_stat("constraints_after_simplification", num_constraints(f));
468  allocate(num_vars, false);
469  typing(f);
470  if(!interpret(f, true)) {
471  exit(EXIT_FAILURE);
472  }
473  }

475  void preprocess_tcn(F& f) {
476  f = ternarize(f);
477  battery::vector<F> extra;
478  f = normalize(f, extra);
479  size_t num_vars = num_quantified_vars(f);
480  stats.print_stat("tnf_variables", num_vars);
481  stats.print_stat("tnf_constraints", num_tnf_constraints(f));
482  allocate(num_vars, true);
483  if(!interpret(f, config.disable_simplify)) {
484  exit(EXIT_FAILURE);
485  }
486  simplifier->init_env(env);
487  if(config.disable_simplify) {
488  /** Even when we don't simplify, we still need to initialize the equivalence classes.
489  * This is necessary to call `print_variable` on `simplifier` when finding a solution. */
490  simplifier->initialize(num_vars, 0);
491  return;
492  }
493  auto& tnf = f.seq();
494  simplifier->initialize_tnf(num_vars, tnf);
495  size_t preprocessing_fixpoint_iterations = 0;
496  SimplifierStats preprocessing_stats;
497  size_t eliminated_variables = 0;
498  local::B has_changed = true;
499  GaussSeidelIteration fp_engine;
500  /** We apply several preprocessing steps until we reach a fixpoint. */
501  while(!iprop->is_bot() && has_changed) {
502  has_changed = false;
503  preprocessing_fixpoint_iterations++;
504  SimplifierStats local_preprocessing_stats;
505  fp_engine.fixpoint(iprop->num_deductions(), [&](size_t i) { return iprop->deduce(i); }, has_changed);
506  if(has_changed) {
507  simplifier->meet_equivalence_classes();
508  }
509  has_changed |= simplifier->algebraic_simplify(tnf, local_preprocessing_stats);
510  simplifier->eliminate_entailed_constraints(*iprop, tnf, local_preprocessing_stats);
511  has_changed |= simplifier->i_cse(tnf, local_preprocessing_stats);
512  if(has_changed) {
513  simplifier->meet_equivalence_classes();
514  local_preprocessing_stats.print(stats, preprocessing_fixpoint_iterations);
515  }
516  preprocessing_stats.merge(local_preprocessing_stats);
517  }
518  // simplifier->eliminate_entailed_constraints(*iprop, tnf, preprocessing_stats);
519  simplifier->eliminate_useless_variables(tnf, eliminated_variables);
520  f = simplifier->deinterpret(tnf, true);
521  F extra_f = F::make_nary(AND, std::move(extra));
522  simplifier->substitute(extra_f);
523  if(config.verbose_solving) {
524  printf("%% Formula simplified.\n");
525  }
526  F f2 = F::make_binary(std::move(f), AND, std::move(extra_f));
527  num_vars = num_quantified_vars(f2);
528  preprocessing_stats.print(stats);
529  stats.print_stat("eliminated_variables", eliminated_variables);
530  stats.print_stat("preprocessing_fixpoint_iterations", preprocessing_fixpoint_iterations);
531  stats.print_stat("variables_after_simplification", num_vars);
532  stats.print_stat("constraints_after_simplification", num_tnf_constraints(f2));
533  if(iprop->is_bot()) {
534  return;
535  }
536  allocate(num_vars, false);
537  if(!interpret(f2, true)) {
538  exit(EXIT_FAILURE);
539  }
540  }

619  /** Only for PIR abstract domain. */
620  void analyze_pir() const {
621  if(config.verbose_solving) {
622  printf("%% Analyzing the constraint network...\n");
623  }
624  if(iprop->is_bot()) {
625  printf("%% Constraint network UNSAT at root, analysis cancelled...\n");
626  return;
627  }
628  std::vector<vstat> vstats(store->vars(), vstat{});
629  for(int i = 0; i < store->vars(); ++i) {
630  auto width = (*store)[i].width().lb();
631  vstats[i].infinite_domain = width.is_top();
632  if(!vstats[i].infinite_domain) {
633  vstats[i].domain_size = width.value() + 1;
634  }
635  }
636 
637  std::map<Sig, size_t> op_stats{{{Sig::EQ,0}, {Sig::LEQ,0}, {Sig::NEQ,0}, {Sig::GT,0}, {ADD,0}, {MUL,0}, {EMOD,0}, {EDIV,0}, {MIN,0}, {MAX,0}}};
638  std::map<Sig, size_t> reified_op_stats{{{Sig::EQ,0}, {Sig::LEQ,0}}};
639 
640  for(int i = 0; i < iprop->num_deductions(); ++i) {
641  bytecode_type bytecode = iprop->load_deduce(i);
642  vstats[bytecode.x.vid()].num_occurrences++;
643  vstats[bytecode.y.vid()].num_occurrences++;
644  vstats[bytecode.z.vid()].num_occurrences++;
645  if(!op_stats.contains(bytecode.op)) {
646  printf("%% WARNING: operator not explicitly managed in PIR: %d\n", bytecode.op);
647  op_stats[bytecode.op] = 0;
648  }
649  auto dom = iprop->project(bytecode.x);
650  if((is_arithmetic_comparison(bytecode.op)) &&
651  (dom.lb().value() != dom.ub().value() || dom.lb().value() == 0))
652  {
653  if(dom.lb().value() != dom.ub().value()) {
654  reified_op_stats[bytecode.op]++;
655  }
656  else {
657  op_stats[negate_arithmetic_comparison(bytecode.op)]++;
658  }
659  }
660  else {
661  op_stats[bytecode.op]++;
662  }
663  }
664 
665  size_t num_constants = 0;
666  size_t num_2bits_vars = 0;
667  size_t num_64bits_vars = 0;
668  size_t num_128bits_vars = 0;
669  size_t num_256bits_vars = 0;
670  size_t num_512bits_vars = 0;
671  size_t num_65536bits_vars = 0;
672  size_t num_infinites = 0;
673  double average_occ_vars = 0;
674  size_t sum_props_of_vars = 0;
675  size_t max_occ_vars = 0;
676  size_t num_vars_in_2_constraints = 0;
677  size_t num_vars_in_3_constraints = 0;
678  size_t num_vars_in_4_to_10_constraints = 0;
679  size_t num_vars_in_more_than_10_constraints = 0;
680  size_t largest_dom = 0;
681  size_t sum_domain_size = 0;
682  for(int i = 0; i < vstats.size(); ++i) {
683  if(vstats[i].infinite_domain) {
684  ++num_infinites;
685  }
686  else {
687  largest_dom = battery::max(largest_dom, vstats[i].domain_size);
688  sum_domain_size += vstats[i].domain_size;
689  }
690  if(vstats[i].domain_size == 1) {
691  ++num_constants;
692  }
693  else {
694  num_2bits_vars += vstats[i].domain_size == 2;
695  num_64bits_vars += vstats[i].domain_size <= 64;
696  num_128bits_vars += vstats[i].domain_size <= 128;
697  num_256bits_vars += vstats[i].domain_size <= 256;
698  num_512bits_vars += vstats[i].domain_size <= 512;
699  num_65536bits_vars += vstats[i].domain_size <= 65536;
700  sum_props_of_vars += vstats[i].num_occurrences;
701  max_occ_vars = battery::max(max_occ_vars, vstats[i].num_occurrences);
702  }
703  num_vars_in_2_constraints += vstats[i].num_occurrences == 2;
704  num_vars_in_3_constraints += vstats[i].num_occurrences == 3;
705  num_vars_in_4_to_10_constraints += vstats[i].num_occurrences > 3 && vstats[i].num_occurrences <= 10;
706  num_vars_in_more_than_10_constraints += vstats[i].num_occurrences > 10;
707  }
708  average_occ_vars = static_cast<double>(sum_props_of_vars) / static_cast<double>(vstats.size() - num_constants);
709 
710  stats.print_stat("num_constants", num_constants);
711  stats.print_stat("num_infinite_domains", num_infinites);
712  // Print the number of average occurrence of variables in the constraints.
713  stats.print_stat("sum_props_of_vars", sum_props_of_vars);
714  stats.print_memory_statistics(config.verbose_solving, "sum_props_of_vars", sum_props_of_vars*4); // 1 integer per propagator indexes.
715  stats.print_stat("avg_constraints_per_unassigned_var", average_occ_vars);
716  // Print the maximum number of occurrence of a single variable in the constraints.
717  stats.print_stat("max_constraints_per_unassigned_var", max_occ_vars);
718  // Print the number of variables occurring only twice in the constraints.
719  stats.print_stat("num_vars_in_2_constraints", num_vars_in_2_constraints);
720  stats.print_stat("num_vars_in_3_constraints", num_vars_in_3_constraints);
721  stats.print_stat("num_vars_in_4_to_10_constraints", num_vars_in_4_to_10_constraints);
722  stats.print_stat("num_vars_in_more_than_10_constraints", num_vars_in_more_than_10_constraints);
723  // Print the number of bits required to represent the bounded variables in a bitset.
724  stats.print_stat("sum_domain_size", sum_domain_size);
725  stats.print_memory_statistics(config.verbose_solving, "sum_domain_size", sum_domain_size/8);
726  // Print the largest variable in the constraint network.
727  stats.print_stat("largest_domain", largest_dom);
728  // Print the number of variables with a width of 2, 64, 128, ... bits or less in the constraint network.
729  stats.print_stat("num_2bits_vars", num_2bits_vars);
730  stats.print_stat("num_64bits_vars", num_64bits_vars);
731  stats.print_stat("num_128bits_vars", num_128bits_vars);
732  stats.print_stat("num_256bits_vars", num_256bits_vars);
733  stats.print_stat("num_512bits_vars", num_512bits_vars);
734  stats.print_stat("num_65536bits_vars", num_65536bits_vars);
735  // For each operator, print how many times they each occur in the constraint network.
736  for(auto& [sig,count] : op_stats) {
737  std::string stat_name("num_op_");
738  stat_name += string_of_sig_txt(sig);
