36void block_signal_ctrlc() {
37 sigset_t ctrlc;
38 sigemptyset(&ctrlc);
39 sigaddset(&ctrlc, SIGINT);
40 sigaddset(&ctrlc, SIGTERM);
41 if(sigprocmask(SIG_BLOCK, &ctrlc, NULL) != 0) {
42 printf("%% ERROR: Unable to deal with CTRL-C. Therefore, we will not be able to print the latest solution before quitting.\n");
43 perror(NULL);
44 return;
45 }
46}

48bool must_quit() {
49 sigset_t pending_sigs;
50 sigemptyset(&pending_sigs);
51 sigpending(&pending_sigs);
52 return sigismember(&pending_sigs, SIGINT) == 1 || sigismember(&pending_sigs, SIGTERM) == 1;
53}

55/** Check if the timeout of the current execution is exceeded and returns `false` otherwise.
56 * It also update the statistics relevant to the solving duration and the exhaustive flag if we reach the timeout.
57 */
58template <class A, class Timepoint>
59bool check_timeout(A& a, const Timepoint& start) {
60 auto now = std::chrono::high_resolution_clock::now();
61 a.stats.duration = std::chrono::duration_cast<std::chrono::milliseconds>(now - start).count();
62 if(a.config.timeout_ms == 0) {
63 return true;
64 }
65 if(a.stats.duration >= a.config.timeout_ms) {
66 if(a.config.verbose_solving) {
67 printf("%% CPU: Timeout reached.\n");
68 }
69 a.stats.exhaustive = false;
70 return false;
71 }
72 return true;
73}

102/** This class is parametrized by a universe of discourse, which is the domain of the variables in the store and various allocators:
103 * - BasicAllocator: default allocator, used to allocate abstract domains, the environment, storing intermediate results, etc.
104 * - PropAllocator: allocator used for the PC abstract domain, to allocate the propagators.
105 * - StoreAllocator: allocator used for the store, to allocate the variables.
106 *
107 * Normally, you should use the fastest memory for the store, then for the propagators and then for the rest.
108 */
109template <class Universe,
110 class BasicAllocator,
111 class PropAllocator,
112 class StoreAllocator>
113struct AbstractDomains {
114 using universe_type = typename Universe::local_type;
115
116 /** Version of the abstract domains with a simple allocator, to represent the best solutions. */
117 using LIStore = VStore<universe_type, BasicAllocator>;
118
119 using IStore = VStore<Universe, StoreAllocator>;
120 using IPC = PC<IStore, PropAllocator>; // Interval Propagators Completion
121 using ISimplifier = Simplifier<IPC, BasicAllocator>;
122 using Split = SplitStrategy<IPC, BasicAllocator>;
123 using IST = SearchTree<IPC, Split, BasicAllocator>;
124 using IBAB = BAB<IST, LIStore>;
125
126 using basic_allocator_type = BasicAllocator;
127 using prop_allocator_type = PropAllocator;
128 using store_allocator_type = StoreAllocator;
129
130 using this_type = AbstractDomains<Universe, BasicAllocator, PropAllocator, StoreAllocator>;

136 /** We copy `other` in a new element, and ignore every variable not used in a GPU block.
137 * This is because copying everything in each block is very slow.
138 *
139 * NOTE: It is not the allocation itself that is slow, I think it calling many copy constructors for atomic variables (note that in simplifier we have an atomic memory if the underlying domain has one).
140 */
141 template <class U2, class BasicAlloc2, class PropAllocator2, class StoreAllocator2>
142 CUDA AbstractDomains(const tag_gpu_block_copy&,
143 const AbstractDomains<U2, BasicAlloc2, PropAllocator2, StoreAllocator2>& other,
144 const BasicAllocator& basic_allocator = BasicAllocator(),
145 const PropAllocator& prop_allocator = PropAllocator(),
146 const StoreAllocator& store_allocator = StoreAllocator())

148 , prop_allocator(prop_allocator)
149 , store_allocator(store_allocator)
150 , fzn_output(basic_allocator)
151 , config(other.config, basic_allocator)
152 , stats(other.stats)
153 , env(basic_allocator)
154 , store(store_allocator)
155 , ipc(prop_allocator)
156 , simplifier(basic_allocator)
157 , split(basic_allocator)

171 }

238 CUDA void allocate(int num_vars) {
239 env = VarEnv<basic_allocator_type>{basic_allocator};
240 store = battery::allocate_shared<IStore, StoreAllocator>(store_allocator, env.extends_abstract_dom(), num_vars, store_allocator);
241 ipc = battery::allocate_shared<IPC, PropAllocator>(prop_allocator, env.extends_abstract_dom(), store, prop_allocator);
242 // If the simplifier is already allocated, it means we are currently reallocating the abstract domains after preprocessing.
243 if(!simplifier) {
244 simplifier = battery::allocate_shared<ISimplifier, BasicAllocator>(basic_allocator, env.extends_abstract_dom(), ipc, basic_allocator);
245 }
246 split = battery::allocate_shared<Split, BasicAllocator>(basic_allocator, env.extends_abstract_dom(), ipc, basic_allocator);
247 eps_split = battery::allocate_shared<Split, BasicAllocator>(basic_allocator, env.extends_abstract_dom(), ipc, basic_allocator);
248 search_tree = battery::allocate_shared<IST, BasicAllocator>(basic_allocator, env.extends_abstract_dom(), ipc, split, basic_allocator);
249 // Note that `best` must have the same abstract type then store (otherwise projection of the variables will fail).
250 best = battery::allocate_shared<LIStore, BasicAllocator>(basic_allocator, store->aty(), num_vars, basic_allocator);
251 bab = battery::allocate_shared<IBAB, BasicAllocator>(basic_allocator, env.extends_abstract_dom(), search_tree, best);
252 if(config.verbose_solving) {
253 printf("%% Abstract domain allocated.\n");
254 }
255 }

269 // Mainly to interpret the IN constraint in IPC instead of only over-approximating in intervals.
270 template <class F>
271 CUDA void typing(F& f) const {
272 switch(f.index()) {
273 case F::Seq:
274 if(f.sig() == ::lala::IN && f.seq(1).is(F::S) && f.seq(1).s().size() > 1) {
275 f.type_as(ipc->aty());
276 return;
277 }
278 for(int i = 0; i < f.seq().size(); ++i) {
279 typing(f.seq(i));
280 }
281 break;
282 case F::ESeq:
283 for(int i = 0; i < f.eseq().size(); ++i) {
284 typing(f.eseq(i));
285 }
286 break;
287 }
288 }

292 // We first try to interpret, and if it does not work, we interpret again with the diagnostics mode turned on.
293 template <class F, class Env, class A>
294 CUDA bool interpret_and_diagnose_and_tell(const F& f, Env& env, A& a) {
295 IDiagnostics diagnostics;
296 if(!interpret_and_tell(f, env, a, diagnostics)) {
297 IDiagnostics diagnostics2;
298 interpret_and_tell<true>(f, env, a, diagnostics2);
299 diagnostics2.print();
300 return false;
301 }
302 return true;
303 }

306 template <class F>
307 CUDA bool interpret(const F& f) {
308 if(config.verbose_solving) {
309 printf("%% Interpreting the formula...\n");
310 }
311 if(!interpret_and_diagnose_and_tell(f, env, *bab)) {
312 return false;
313 }
314 stats.variables = store->vars();
315 stats.constraints = ipc->num_refinements();
316 bool can_interpret = true;
317 if(split->num_strategies() == 0) {
318 can_interpret &= interpret_default_strategy<F>();
319 }
320 if(eps_split->num_strategies() == 0) {
321 can_interpret &= interpret_default_eps_strategy<F>();
322 }
323 return can_interpret;
324 }

326 template <class F>
327 CUDA bool prepare_simplifier(F& f) {
328 if(config.verbose_solving) {
329 printf("%% Simplifying the formula...\n");
330 }
331 IDiagnostics diagnostics;
332 typename ISimplifier::template tell_type<basic_allocator_type> tell{basic_allocator};
333 if(top_level_ginterpret_in<IKind::TELL>(*simplifier, f, env, tell, diagnostics)) {
334 simplifier->tell(std::move(tell));
335 return true;
336 }
337 else if(config.verbose_solving) {
338 printf("WARNING: Could not simplify the formula because:\n");
339 IDiagnostics diagnostics2;
340 top_level_ginterpret_in<IKind::TELL, true>(*simplifier, f, env, tell, diagnostics2);
341 diagnostics2.print();
342 }
343 return false;
344 }

373 FormulaPtr prepare_solver() {
374 auto start = std::chrono::high_resolution_clock::now();
375
376 // I. Parse the FlatZinc model.
377 FormulaPtr f;
378 if(config.input_format() == InputFormat::FLATZINC) {
379 f = parse_flatzinc(config.problem_path.data(), fzn_output);
380 }
381 else if(config.input_format() == InputFormat::XCSP3) {
382 f = parse_xcsp3(config.problem_path.data(), fzn_output);
383 }
384 if(!f) {
385 std::cerr << "Could not parse input file." << std::endl;
386 exit(EXIT_FAILURE);
387 }
388
389 if(config.verbose_solving) {
390 printf("%% Input file parsed\n");
391 }
392
393 if(config.print_ast) {
394 printf("%% Parsed AST:\n");
395 f->print(true);
396 printf("\n");
397 }
398
399 allocate(num_quantified_vars(*f));
400 type_and_interpret(*f);
401
402 auto interpretation_time = std::chrono::high_resolution_clock::now();
403 stats.interpretation_duration += std::chrono::duration_cast<std::chrono::milliseconds>(interpretation_time - start).count();
404 return f;
405 }

407 void preprocess() {
408 auto raw_formula = prepare_solver();
409 auto start = std::chrono::high_resolution_clock::now();
410 if(prepare_simplifier(*raw_formula)) {
411 GaussSeidelIteration fp_engine;
412 fp_engine.fixpoint(*ipc);
413 fp_engine.fixpoint(*simplifier);
414 auto f = simplifier->deinterpret();
415 stats.eliminated_variables = simplifier->num_eliminated_variables();
416 stats.eliminated_formulas = simplifier->num_eliminated_formulas();
417 allocate(num_quantified_vars(f));
418 type_and_interpret(f);
419 }
420 auto interpretation_time = std::chrono::high_resolution_clock::now();
421 stats.interpretation_duration += std::chrono::duration_cast<std::chrono::milliseconds>(interpretation_time - start).count();
422 }

425 template <class F>
426 CUDA bool interpret_default_strategy() {
427 if(config.verbose_solving) {
428 printf("%% No split strategy provided, using the default one (first_fail, indomain_min).\n");
429 }
430 config.free_search = true;
431 typename F::Sequence seq;
432 seq.push_back(F::make_nary("first_fail", {}));
433 seq.push_back(F::make_nary("indomain_min", {}));
434 for(int i = 0; i < env.num_vars(); ++i) {
435 seq.push_back(F::make_avar(env[i].avars[0]));
436 }
437 F search_strat = F::make_nary("search", std::move(seq));
438 if(!interpret_and_diagnose_and_tell(search_strat, env, *bab)) {
439 return false;
440 }
441 return true;
442 }
