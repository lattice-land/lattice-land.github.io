34 void block_signal_ctrlc() {
35  sigset_t ctrlc;
36  sigemptyset(&ctrlc);
37  sigaddset(&ctrlc, SIGINT);
38  sigaddset(&ctrlc, SIGTERM);
39  if(sigprocmask(SIG_BLOCK, &ctrlc, NULL) != 0) {
40  printf("%% ERROR: Unable to deal with CTRL-C. Therefore, we will not be able to print the latest solution before quitting.\n");
41  perror(NULL);
42  return;
43  }

53 /** Check if the timeout of the current execution is exceeded and returns `false` otherwise.
54  * It also update the statistics relevant to the solving duration and the exhaustive flag if we reach the timeout.
55  */
56 template <class A, class Timepoint>
57 bool check_timeout(A& a, const Timepoint& start) {
58  auto now = std::chrono::high_resolution_clock::now();
59  a.stats.duration = std::chrono::duration_cast<std::chrono::milliseconds>(now - start).count();
60  if(a.config.timeout_ms == 0) {
61  return true;
62  }
63  if(a.stats.duration >= a.config.timeout_ms) {
64  a.stats.exhaustive = false;
65  return false;
66  }
67  return true;
68 }

97  * - BasicAllocator: default allocator, used to allocate abstract domains, the environment, storing intermediate results, etc.
98  * - PropAllocator: allocator used for the PC abstract domain, to allocate the propagators.

232  // Mainly to interpret the IN constraint in IPC instead of only over-approximating in intervals.
233  template <class F>
234  CUDA void typing(F& f) const {
235  switch(f.index()) {
236  case F::Seq:
237  if(f.sig() == ::lala::IN && f.seq(1).is(F::S) && f.seq(1).s().size() > 1) {
238  f.type_as(ipc->aty());
239  return;
240  }
241  for(int i = 0; i < f.seq().size(); ++i) {
242  typing(f.seq(i));
243  }
244  break;
245  case F::ESeq:
246  for(int i = 0; i < f.eseq().size(); ++i) {
247  typing(f.eseq(i));
248  }
249  break;
250  }
251  }

253  template <class F>
254  CUDA bool interpret(const F& f) {
255  if(config.verbose_solving) {
256  printf("%% Interpreting the formula...\n");
257  }
258  auto r = bab->interpret_tell_in(f, env);
259  if(!r.has_value()) {
260  r.print_diagnostics();
261  return false;
262  }
263  local::BInc has_changed;
264  bab->tell(std::move(r.value()), has_changed);
265  stats.variables = store->vars();
266  stats.constraints = ipc->num_refinements();
267  bool can_interpret = true;
268  if(split->num_strategies() == 0) {
269  can_interpret &= interpret_default_strategy<F>();
270  }
271  if(eps_split->num_strategies() == 0) {
272  can_interpret &= interpret_default_eps_strategy<F>();
273  }
274  return can_interpret;
275  }

277  void prepare_solver() {
278  auto start = std::chrono::high_resolution_clock::now();
279 
280  // I. Parse the FlatZinc model.
281  using FormulaPtr = battery::shared_ptr<TFormula<basic_allocator_type>, basic_allocator_type>;
282  FormulaPtr f;
283  if(config.input_format() == InputFormat::FLATZINC) {
284  f = parse_flatzinc(config.problem_path.data(), fzn_output);
285  }
286  else if(config.input_format() == InputFormat::XCSP3) {
287  f = parse_xcsp3(config.problem_path.data(), fzn_output);
288  }
289  if(!f) {
290  std::cerr << "Could not parse input file." << std::endl;
291  exit(EXIT_FAILURE);
292  }
293 
294  if(config.verbose_solving) {
295  printf("%% Input file parsed\n");
296  }
297 
298  if(config.print_ast) {
299  printf("%% Parsed AST:\n");
300  f->print(true);
301  printf("\n");
302  }
303 
304  // II. Create the abstract domain.
305  allocate(num_quantified_vars(*f));
306 
307  // III. Interpret the formula in the abstract domain.
308  typing(*f);
309  if(config.print_ast) {
310  printf("%% Typed AST:\n");
311  f->print(true);
312  printf("\n");
313  }
314  if(!interpret(*f)) {
315  exit(EXIT_FAILURE);
316  }
317 
318  if(config.print_ast) {
319  printf("%% Interpreted AST:\n");
320  ipc->deinterpret(env).print(true);
321  printf("\n");
322  }
323 
324  auto interpretation_time = std::chrono::high_resolution_clock::now();
325  stats.interpretation_duration = std::chrono::duration_cast<std::chrono::milliseconds>(interpretation_time - start).count();
326 
327  if(config.verbose_solving) {
328  printf("%% Formula has been loaded, configuration and solving begin...\n");
329  }
330  }

333  template <class F>
334  CUDA bool interpret_default_strategy() {
335  if(config.verbose_solving) {
336  printf("%% No split strategy provided, using the default one (first_fail, indomain_split).\n");
337  }
338  config.free_search = true;
339  typename F::Sequence seq;
340  seq.push_back(F::make_nary("first_fail", {}));
341  seq.push_back(F::make_nary("indomain_split", {}));
342  for(int i = 0; i < env.num_vars(); ++i) {
343  seq.push_back(F::make_avar(env[i].avars[0]));
344  }
345  F search_strat = F::make_nary("search", std::move(seq));
346  auto r = bab->interpret_tell_in(search_strat, env);
347  if(!r.has_value()) {
348  r.print_diagnostics();
349  return false;
350  }
351  local::BInc has_changed;
352  bab->tell(std::move(r.value()), has_changed);
353  return true;
354  }
