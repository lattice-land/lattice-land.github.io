167 template <class Alloc>
168 CUDA void meet(const Statistics<Alloc>& other) {
169 nodes += other.nodes;
170 fails += other.fails;
171 solutions += other.solutions;
172 depth_max = battery::max(depth_max, other.depth_max);
173 exhaustive = exhaustive && other.exhaustive;
174 eps_solved_subproblems += other.eps_solved_subproblems;
175 eps_skipped_subproblems += other.eps_skipped_subproblems;
176 num_blocks_done += other.num_blocks_done;
177 fixpoint_iterations += other.fixpoint_iterations;
178 timers.meet(other.timers);
179 }
