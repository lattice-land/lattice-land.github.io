157 template <class Alloc>
158 CUDA void meet(const Statistics<Alloc>& other) {
159 nodes += other.nodes;
160 fails += other.fails;
161 solutions += other.solutions;
162 depth_max = battery::max(depth_max, other.depth_max);
163 exhaustive = exhaustive && other.exhaustive;
164 eps_solved_subproblems += other.eps_solved_subproblems;
165 eps_skipped_subproblems += other.eps_skipped_subproblems;
166 num_blocks_done += other.num_blocks_done;
167 fixpoint_iterations += other.fixpoint_iterations;
168 num_deductions += other.num_deductions;
169 timers.meet(other.timers);
170 }
