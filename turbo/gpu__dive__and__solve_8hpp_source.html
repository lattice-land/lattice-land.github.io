56 /** Depending on the problem, we can store the abstract elements in different memories.
57  * The "worst" is everything in global memory (GLOBAL) when the problem is too large for the shared memory.
58  * The "best" is when both the store of variables and the propagators (STORE_PC_SHARED) can be stored in shared memory.
59  * A third possibility is to store only the variables' domains in the shared memory (STORE_SHARED).
60 */
61 enum class MemoryKind {
62  GLOBAL,
63  STORE_SHARED,
64  STORE_PC_SHARED
65 };

116 /** `GridData` represents the problem to be solved (`root`), the data of each block when running EPS (`blocks`), the index of the subproblem that needs to be solved next (`next_subproblem`), an estimation of the best bound found across subproblems in case of optimisation problem and other global information.
117  */
118 template <class S>
119 struct GridData {
120  using GridCP = typename S::GridCP;
121  using BlockCP = typename S::BlockCP;
122  using U2 = typename S::U2;
123 
124  GridCP root;
125  // `blocks_root` is a copy of `root` but with the same allocators than the ones used in `blocks`.
126  // This is helpful to share immutable data among blocks (for instance the propagators).
127  bt::shared_ptr<BlockCP, bt::global_allocator> blocks_root;
128  // Stop from the CPU, for instance because of a timeout.
129  volatile bool cpu_stop;
130  // Boolean indicating that the blocks have been reduced, and the CPU can now print the statistics.
131  volatile bool blocks_reduced;
132  MemoryConfig mem_config;
133  bt::vector<BlockData<S>, bt::global_allocator> blocks;
134  // Stop from a block on the GPU, for instance because we found a solution.
135  bt::shared_ptr<B<bt::atomic_memory_grid>, bt::global_allocator> gpu_stop;
136  bt::shared_ptr<ZLB<size_t, bt::atomic_memory_grid>, bt::global_allocator> next_subproblem;
137  bt::shared_ptr<U2, bt::global_allocator> best_bound;
138 
139  // All of what follows is only to support printing while the kernel is running.
140  // In particular, we transfer the solution to the CPU where it is printed, because printing on the GPU can be very slow when the problem is large.
141  bt::shared_ptr<cuda::binary_semaphore<cuda::thread_scope_device>, bt::global_allocator> print_lock;
142  cuda::std::atomic_flag ready_to_produce;
143  cuda::std::atomic_flag ready_to_consume;
144 
145  GridData(const GridCP& root, const MemoryConfig& mem_config)
146  : root(root)
147  , mem_config(mem_config)
148  , cpu_stop(false)
149  , blocks_reduced(false)
150  {
151  ready_to_consume.clear();
152  ready_to_produce.clear();
153  cuda::atomic_thread_fence(cuda::memory_order_seq_cst, cuda::thread_scope_system);
154  }
155 
156  template <class BlockBAB>
157  __device__ void produce_solution(const BlockBAB& bab) {
158  print_lock->acquire();
159  if(!cpu_stop) {
160  bab.extract(*(root.bab));
161  cuda::atomic_thread_fence(cuda::memory_order_seq_cst, cuda::thread_scope_system);
162  ready_to_consume.test_and_set(cuda::std::memory_order_seq_cst);
163  ready_to_consume.notify_one();
164  // Wait the CPU has consumed the solution before continuing.
165  // This avoids a problem with "blocks_reduced" being `true` too fast.
166  ready_to_produce.wait(false, cuda::std::memory_order_seq_cst);
167  ready_to_produce.clear();
168  }
169  print_lock->release();
170  }
171 
172  __host__ bool consume_solution() {
173  ready_to_consume.wait(false, cuda::std::memory_order_seq_cst);
174  ready_to_consume.clear();
175  if(blocks_reduced) {
176  root.print_final_solution();
177  if(root.config.print_statistics) {
178  root.print_mzn_statistics();
179  }
180  return true;
181  }
182  else {
183  root.print_solution();
184  }
185  ready_to_produce.test_and_set(cuda::std::memory_order_seq_cst);
186  ready_to_produce.notify_one();
187  return false;
188  }

190  __device__ void allocate() {
191  assert(threadIdx.x == 0 && blockIdx.x == 0);
192  auto root_mem_config(mem_config);
193  root_mem_config.mem_kind = MemoryKind::GLOBAL;
194  blocks_root = bt::make_shared<BlockCP, bt::global_allocator>(
195  typename BlockCP::tag_gpu_block_copy{},
196  false, // Due to different allocators between BlockCP and GridCP, it won't be able to share data anyways.
197  root,
198  bt::global_allocator{},
199  root_mem_config.make_pc_pool(bt::pool_allocator(nullptr,0)),
200  root_mem_config.make_store_pool(bt::pool_allocator(nullptr,0)));
201  blocks = bt::vector<BlockData<S>, bt::global_allocator>(root.config.or_nodes);
202  gpu_stop = bt::make_shared<B<bt::atomic_memory_grid>, bt::global_allocator>(false);
203  print_lock = bt::make_shared<cuda::binary_semaphore<cuda::thread_scope_device>, bt::global_allocator>(1);
204  next_subproblem = bt::make_shared<ZLB<size_t, bt::atomic_memory_grid>, bt::global_allocator>(0);
205  best_bound = bt::make_shared<U2, bt::global_allocator>();
206  }
207 
208  __device__ void deallocate() {
209  assert(threadIdx.x == 0 && blockIdx.x == 0);
210  blocks = bt::vector<BlockData<S>, bt::global_allocator>();
211  blocks_root->deallocate();
212  blocks_root.reset();
213  gpu_stop.reset();
214  print_lock.reset();
215  next_subproblem.reset();
216  best_bound.reset();
217  }

220 /** `BlockData` contains all the structures required to solve a subproblem including the problem itself (`root`) and the fixpoint engine (`fp_engine`). */
221 template <class S>
222 struct BlockData {
223  using GridCP = typename S::GridCP;
224  using BlockCP = typename S::BlockCP;
225 
226  using snapshot_type = typename BlockCP::IST::snapshot_type<bt::global_allocator>;
227  size_t subproblem_idx;
228  bt::shared_ptr<FPEngine, bt::pool_allocator> fp_engine;
229  bt::shared_ptr<AtomicBool, bt::pool_allocator> has_changed;
230  bt::shared_ptr<AtomicBool, bt::pool_allocator> stop;
231  bt::shared_ptr<BlockCP, bt::global_allocator> root;
232  bt::shared_ptr<snapshot_type, bt::global_allocator> snapshot_root;
233 
234  __device__ BlockData():
235  has_changed(nullptr, bt::pool_allocator(nullptr, 0)),
236  stop(nullptr, bt::pool_allocator(nullptr, 0))
237  {}
238 
239 public:
240  /** Initialize the block data.
241  * Allocate the abstract domains in the best memory available depending on how large are the abstract domains. */
242  __device__ void allocate(GridData<S>& grid_data, unsigned char* shared_mem) {
243  auto block = cooperative_groups::this_thread_block();
244  if(threadIdx.x == 0) {
245  subproblem_idx = blockIdx.x;
246  MemoryConfig& mem_config = grid_data.mem_config;
247  bt::pool_allocator shared_mem_pool(mem_config.make_shared_pool(shared_mem));
248  fp_engine = bt::allocate_shared<FPEngine, bt::pool_allocator>(shared_mem_pool);
249  has_changed = bt::allocate_shared<AtomicBool, bt::pool_allocator>(shared_mem_pool, true);
250  stop = bt::allocate_shared<AtomicBool, bt::pool_allocator>(shared_mem_pool, false);
251  root = bt::make_shared<BlockCP, bt::global_allocator>(typename BlockCP::tag_gpu_block_copy{},
252  (mem_config.mem_kind != MemoryKind::STORE_PC_SHARED),
253  *(grid_data.blocks_root),
254  bt::global_allocator{},
255  mem_config.make_pc_pool(shared_mem_pool),
256  mem_config.make_store_pool(shared_mem_pool));
257  snapshot_root = bt::make_shared<snapshot_type, bt::global_allocator>(root->search_tree->template snapshot<bt::global_allocator>());
258  }
259  block.sync();
260  fp_engine->init(root->iprop->num_deductions());
261  block.sync();
262  }
263 
264  __device__ void deallocate_shared() {
265  if(threadIdx.x == 0) {
266  fp_engine.reset();
267  has_changed.reset();
268  stop.reset();
269  root->deallocate();
270  snapshot_root.reset();
271  }
272  __syncthreads();
273  }
274 
275  __device__ void restore() {
276  if(threadIdx.x == 0) {
277  root->search_tree->restore(*snapshot_root);
278  root->eps_split->reset();
279  }
280  __syncthreads();
281  }

298 /** We update the bound found by the current block so it is visible to all other blocks.
299  * Note that this operation might not always succeed, which is okay, the best bound is still preserved in `block_data` and then reduced at the end (in `reduce_blocks`).
300  * The worst that can happen is that a best bound is found twice, which does not prevent the correctness of the algorithm.
301  */
302 template <class S>
303 __device__ bool update_grid_best_bound(BlockData<S>& block_data, GridData<S>& grid_data) {
304  using U0 = typename S::U0;
305  assert(threadIdx.x == 0);
306  if(block_data.root->bab->is_optimization()) {
307  const auto& bab = block_data.root->bab;
308  auto local_best = bab->optimum().project(bab->objective_var());
309  // printf("[new bound] %d: [%d..%d] (current best: [%d..%d])\n", blockIdx.x, local_best.lb().value(), local_best.ub().value(), grid_data.best_bound->lb().value(), grid_data.best_bound->ub().value());
310  if(bab->is_maximization()) {
311  return grid_data.best_bound->meet_lb(dual_bound<typename U0::LB>(local_best.ub()));
312  }
313  else {
314  return grid_data.best_bound->meet_ub(dual_bound<typename U0::UB>(local_best.lb()));
315  }
316  }
317  return false;
318 }

320 /** This function updates the best bound of the current block according to the best bound found so far across blocks.
321  * We directly update the store with the best bound.
322  * This function should be called in each node, since the best bound is erased on backtracking (it is not included in the snapshot).
323  */
324 template <class S>
325 __device__ void update_block_best_bound(BlockData<S>& block_data, GridData<S>& grid_data) {
326  using U0 = typename S::U0;
327  if(threadIdx.x == 0 && block_data.root->bab->is_optimization()) {
328  const auto& bab = block_data.root->bab;
329  VarEnv<bt::global_allocator> empty_env{};
330  auto best_formula = bab->template deinterpret_best_bound<bt::global_allocator>(
331  bab->is_maximization()
332  ? U0(dual_bound<typename U0::UB>(grid_data.best_bound->lb()))
333  : U0(dual_bound<typename U0::LB>(grid_data.best_bound->ub())));
334  // printf("global best: "); grid_data.best_bound->ub().print(); printf("\n");
335  // best_formula.print(); printf("\n");
336  IDiagnostics diagnostics;
337  interpret_and_tell(best_formula, empty_env, *block_data.root->store, diagnostics);
338  }
339 }

341 /** Propagate a node of the search tree and process the leaf nodes (failed or solution).
342  * Branching on unknown nodes is a task left to the caller.
343  */
344 template <class S>
345 __device__ bool propagate(BlockData<S>& block_data, GridData<S>& grid_data) {
346  using BlockCP = typename S::BlockCP;
347  bool is_leaf_node = false;
348  BlockCP& cp = *block_data.root;
349  auto group = cooperative_groups::this_thread_block();
350  auto& fp_engine = *block_data.fp_engine;
351  auto& iprop = *cp.iprop;
352  auto start = cp.stats.start_timer_device();
353  int fp_iterations;
354  switch(cp.config.fixpoint) {
355  case FixpointKind::AC1:
356  fp_iterations = fp_engine.fixpoint(
357  [&](int i){ return iprop.deduce(i); },
358  [&](){ return iprop.is_bot(); });
359  break;
360  case FixpointKind::WAC1:
361  if(fp_engine.num_active() <= cp.config.wac1_threshold) {
362  fp_iterations = fp_engine.fixpoint(
363  [&](int i){ return iprop.deduce(i); },
364  [&](){ return iprop.is_bot(); });
365  }
366  else {
367  fp_iterations = fp_engine.fixpoint(
368  [&](int i){ return warp_fixpoint<CUDA_THREADS_PER_BLOCK>(iprop, i); },
369  [&](){ return iprop.is_bot(); });
370  }
371  break;
372  }
373  start = cp.stats.stop_timer(Timer::FIXPOINT, start);
374  if(!iprop.is_bot()) {
375  fp_engine.select([&](int i) { return !iprop.ask(i); });
376  start = cp.stats.stop_timer(Timer::SELECT_FP_FUNCTIONS, start);
377  if(fp_engine.num_active() == 0) {
378  is_leaf_node = cp.store->template is_extractable<AtomicExtraction>(group);
379  }
380  }
381  else {
382  is_leaf_node = true;
383  }
384  if(threadIdx.x == 0) {
385  cp.stats.fixpoint_iterations += fp_iterations;
386  bool is_pruned = cp.on_node();
387  if(iprop.is_bot()) {
388  cp.on_failed_node();
389  }
390  else if(is_leaf_node) { // is_leaf_node is set to true above.
391  if(cp.bab->is_satisfaction() || cp.bab->compare_bound(*cp.store, cp.bab->optimum())) {
392  cp.bab->deduce();
393  bool best_has_changed = update_grid_best_bound(block_data, grid_data);
394  if(cp.bab->is_satisfaction() || (best_has_changed && cp.is_printing_intermediate_sol())) {
395  grid_data.produce_solution(*cp.bab);
396  }
397  is_pruned |= cp.update_solution_stats();
398  }
399  }
400  if(is_pruned) {
401  grid_data.gpu_stop->join(true);
402  }
403  cp.stats.stop_timer(Timer::SEARCH, start);
404  }
405  if(is_leaf_node) {
406  fp_engine.reset(cp.iprop->num_deductions());
407  }
408  return is_leaf_node;
409 }

411 /** The initial problem tackled during the dive must always be the same.
412  * Hence, don't be tempted to add the objective during diving because it might lead to ignoring some subproblems since the splitting decisions will differ.
413  */
414 template <class S>
415 __device__ size_t dive(BlockData<S>& block_data, GridData<S>& grid_data) {
416  using BlockCP = typename S::BlockCP;
417  BlockCP& cp = *block_data.root;
418  auto& fp_engine = *block_data.fp_engine;
419  auto& stop = *block_data.stop;
420  // Note that we use `block_has_changed` to stop the "diving", not really to indicate something has changed or not (since we do not need this information for this algorithm).
421  auto& stop_diving = *block_data.has_changed;
422  stop.meet(false);
423  stop_diving.meet(false);
424  fp_engine.barrier();
425  size_t remaining_depth = grid_data.root.config.subproblems_power;
426  while(remaining_depth > 0 && !stop_diving && !stop) {
427  bool is_leaf_node = propagate(block_data, grid_data);
428  stop.join(grid_data.cpu_stop || *(grid_data.gpu_stop));
429  if(is_leaf_node) {
430  if(threadIdx.x == 0) {
431  stop_diving.join(true);
432  }
433  }
434  else {
435  remaining_depth--;
436  if(threadIdx.x == 0) {
437  auto start = cp.stats.start_timer_device();
438  size_t branch_idx = (block_data.subproblem_idx & (size_t{1} << remaining_depth)) >> remaining_depth;
439  auto branches = cp.eps_split->split();
440  assert(branches.size() == 2);
441  cp.iprop->deduce(branches[branch_idx]);
442  cp.stats.stop_timer(Timer::SEARCH, start);
443  }
444  }
445  fp_engine.barrier();
446  }
447  return remaining_depth;
448 }

450 template <class S>
451 __device__ void solve_problem(BlockData<S>& block_data, GridData<S>& grid_data) {
452  using BlockCP = typename S::BlockCP;
453  BlockCP& cp = *block_data.root;
454  auto& fp_engine = *block_data.fp_engine;
455  auto& block_has_changed = *block_data.has_changed;
456  auto& stop = *block_data.stop;
457  block_has_changed.join(true);
458  stop.meet(false);
459  fp_engine.barrier();
460  auto start = cp.stats.start_timer_device();
461  // In the condition, we must only read variables that are local to this block.
462  // Otherwise, two threads might read different values if it is changed in between by another block.
463  while(block_has_changed && !stop) {
464  update_block_best_bound(block_data, grid_data);
465  cp.stats.stop_timer(Timer::SEARCH, start);
466  propagate(block_data, grid_data);
467  if(threadIdx.x == 0) {
468  start = cp.stats.start_timer_device();
469  stop.join(grid_data.cpu_stop || *(grid_data.gpu_stop));
470  // propagate induces a memory fence, therefore all threads are already past the "while" condition.
471  // auto t = cp.stats.start_timer_device();
472  block_has_changed.meet(cp.search_tree->deduce());
473  // cp.stats.stop_timer(Timer::ST_DEDUCE, t);
474  // auto t = cp.stats.start_timer_device();
475  // auto s = cp.search_tree->split->split();
476  // t = cp.stats.stop_timer(Timer::SPLIT, t);
477  // auto P = cp.search_tree->push(std::move(s));
478  // t = cp.stats.stop_timer(Timer::PUSH, t);
479  // block_has_changed.meet(cp.search_tree->pop(std::move(P)));
480  // cp.stats.stop_timer(Timer::POP, t);
481  }
482  fp_engine.barrier();
483  }
484  cp.stats.stop_timer(Timer::SEARCH, start);
485 }

496 template <class S>
497 __global__ void gpu_solve_kernel(GridData<S>* grid_data)
498 {
499  if(threadIdx.x == 0 && blockIdx.x == 0 && grid_data->root.config.verbose_solving) {
500  printf("%% GPU kernel started, starting solving...\n");
501  }
502  extern __shared__ unsigned char shared_mem[];
503  size_t num_subproblems = grid_data->root.stats.eps_num_subproblems;
504  BlockData<S>& block_data = grid_data->blocks[blockIdx.x];
505  block_data.allocate(*grid_data, shared_mem);
506  auto solve_start = block_data.root->stats.start_timer_device();
507  while(block_data.subproblem_idx < num_subproblems && !*(block_data.stop)) {
508  if(threadIdx.x == 0 && grid_data->root.config.verbose_solving) {
509  grid_data->print_lock->acquire();
510  printf("%% Block %d solves subproblem num %" PRIu64 "\n", blockIdx.x, block_data.subproblem_idx);
511  grid_data->print_lock->release();
512  }
513  block_data.restore();
514  __syncthreads();
515  auto dive_start = block_data.root->stats.start_timer_device();
516  size_t remaining_depth = dive(block_data, *grid_data);
517  block_data.root->stats.stop_timer(Timer::DIVE, dive_start);
518  if(remaining_depth == 0) {
519  solve_problem(block_data, *grid_data);
520  if(threadIdx.x == 0 && !*(block_data.stop)) {
521  block_data.root->stats.eps_solved_subproblems += 1;
522  }
523  }
524  else {
525  if(threadIdx.x == 0 && !*(block_data.stop)) {
526  size_t next_subproblem_idx = ((block_data.subproblem_idx >> remaining_depth) + size_t{1}) << remaining_depth;
527  grid_data->next_subproblem->meet(ZLB<size_t, bt::local_memory>(next_subproblem_idx));
528  // It is possible that several blocks skip similar subproblems. Hence, we only count the subproblems skipped by the block solving the left most subproblem.
529  if((block_data.subproblem_idx & ((size_t{1} << remaining_depth) - size_t{1})) == size_t{0}) {
530  block_data.root->stats.eps_skipped_subproblems += next_subproblem_idx - block_data.subproblem_idx;
531  }
532  }
533  }
534  // Load next problem.
535  if(threadIdx.x == 0 && !*(block_data.stop)) {
536  block_data.subproblem_idx = grid_data->next_subproblem->value();
537  grid_data->next_subproblem->meet(ZLB<size_t, bt::local_memory>(block_data.subproblem_idx + size_t{1}));
538  }
539  __syncthreads();
540  }
541  __syncthreads();
542  block_data.root->stats.stop_timer(Timer::SOLVE, solve_start);
543  if(threadIdx.x == 0 && !*(block_data.stop)) {
544  block_data.root->stats.num_blocks_done = 1;
545  }
546  if(threadIdx.x == 0) {
547  grid_data->print_lock->acquire();
548  if(!grid_data->blocks_reduced) {
549  int n = 0;
550  for(int i = 0; i < grid_data->blocks.size(); ++i) {
551  if(grid_data->blocks[i].root) { // `nullptr` could happen if we try to terminate the program before all blocks are even created.
552  n += grid_data->blocks[i].root->stats.num_blocks_done;
553  }
554  }
555  if(block_data.stop->value() || n == grid_data->blocks.size()) {
556  reduce_blocks(grid_data);
557  grid_data->blocks_reduced = true;
558  cuda::atomic_thread_fence(cuda::memory_order_seq_cst, cuda::thread_scope_system);
559  grid_data->ready_to_consume.test_and_set(cuda::std::memory_order_seq_cst);
560  grid_data->ready_to_consume.notify_one();
561  }
562  }
563  grid_data->print_lock->release();
564  }
565  // We must destroy all objects allocated in the shared memory, trying to destroy them anywhere else will lead to segfault.
566  block_data.deallocate_shared();
567 }

584 /** \returns the size of the shared memory and the kind of memory used. */
585 template <class S, class U>
586 MemoryConfig configure_memory(CP<U>& root) {
587  cudaDeviceProp deviceProp;
588  cudaGetDeviceProperties(&deviceProp, 0);
589  const auto& config = root.config;
590  size_t shared_mem_capacity = deviceProp.sharedMemPerBlock;
591 
592  // Copy the root to know how large are the abstract domains.
593  CP<U> root2(root);
594 
595  size_t store_alignment = 200; // The store does not create much alignment overhead since it is almost only a single array.
596 
597  MemoryConfig mem_config;
598  // Need a bit of shared memory for the fixpoint engine.
599  mem_config.shared_bytes = sizeof(FPEngine)+100;
600  mem_config.store_bytes = sizeof_store<S>(root2) + store_alignment;
601  // We add 20% extra memory due to the alignment of the shared memory which is not taken into account in the statistics.
602  // From limited experiments, alignment overhead is usually around 10%.
603  mem_config.pc_bytes = root2.prop_allocator.total_bytes_allocated();
604  mem_config.pc_bytes += mem_config.pc_bytes / 5;
605  if(config.only_global_memory || shared_mem_capacity < mem_config.shared_bytes + mem_config.store_bytes) {
606  if(!config.only_global_memory && config.verbose_solving) {
607  printf("%% The store of variables (%zuKB) cannot be stored in the shared memory of the GPU (%zuKB), therefore we use the global memory.\n",
608  mem_config.store_bytes / 1000,
609  shared_mem_capacity / 1000);
610  }
611  mem_config.mem_kind = MemoryKind::GLOBAL;
612  }
613  else if(shared_mem_capacity > mem_config.shared_bytes + mem_config.store_bytes + mem_config.pc_bytes) {
614  if(config.verbose_solving) {
615  printf("%% The store of variables and the propagators (%zuKB) are stored in the shared memory of the GPU (%zuKB).\n",
616  (mem_config.shared_bytes + mem_config.store_bytes + mem_config.pc_bytes) / 1000,
617  shared_mem_capacity / 1000);
618  }
619  mem_config.shared_bytes += mem_config.store_bytes + mem_config.pc_bytes;
620  mem_config.mem_kind = MemoryKind::STORE_PC_SHARED;
621  }
622  else {
623  if(config.verbose_solving) {
624  printf("%% The store of variables (%zuKB) is stored in the shared memory of the GPU (%zuKB).\n",
625  mem_config.store_bytes / 1000,
626  shared_mem_capacity / 1000);
627  }
628  mem_config.shared_bytes += mem_config.store_bytes;
629  mem_config.mem_kind = MemoryKind::STORE_SHARED;
630  }
631  root.stats.print_memory_statistics(config.verbose_solving, "store_memory_real", root2.store_allocator.total_bytes_allocated());
632  root.stats.print_memory_statistics(config.verbose_solving, "pc_memory_real", root2.prop_allocator.total_bytes_allocated());
633  root.stats.print_memory_statistics(config.verbose_solving, "other_memory_real", root2.basic_allocator.total_bytes_allocated());
634  return mem_config;
635 }

637 /** Wait the solving ends because of a timeout, CTRL-C or because the kernel finished. */
638 template<class S, class Timepoint>
639 bool wait_solving_ends(GridData<S>& grid_data, const Timepoint& start) {
640  cudaEvent_t event;
641  cudaEventCreateWithFlags(&event,cudaEventDisableTiming);
642  cudaEventRecord(event);
643  while(!must_quit(grid_data.root) && check_timeout(grid_data.root, start) && cudaEventQuery(event) == cudaErrorNotReady) {
644  std::this_thread::sleep_for(std::chrono::milliseconds(100));
645  }
646  if(cudaEventQuery(event) == cudaErrorNotReady) {
647  grid_data.cpu_stop = true;
648  grid_data.root.prune();
649  return true;
650  }
651  else {
652  cudaError error = cudaDeviceSynchronize();
653  if(error == cudaErrorIllegalAddress) {
654  printf("%% ERROR: CUDA kernel failed due to an illegal memory access. This might be due to a stack overflow because it is too small. Try increasing the stack size with the options -stack. If it does not work, please report it as a bug.\n");
655  exit(EXIT_FAILURE);
656  }
657  CUDAEX(error);
658  return false;
659  }

691 int threads_per_sm(cudaDeviceProp devProp) {
692  switch (devProp.major){
693  case 2: return (devProp.minor == 1) ? 48 : 32; // Fermi
694  case 3: return 192; // Kepler
695  case 5: return 128; // Maxwell
696  case 6: return (devProp.minor == 0) ? 64 : 128; // Pascal
697  case 7: return 64; // Volta and Turing

699  case 9: return 128; // Hopper
700  default: return 64;

704 template <class S, class U>
705 void configure_blocks_threads(CP<U>& root, const MemoryConfig& mem_config) {
706  int hint_num_blocks;
707  int hint_num_threads;
708  CUDAE(cudaOccupancyMaxPotentialBlockSize(&hint_num_blocks, &hint_num_threads, (void*) gpu_solve_kernel<S>, (int)mem_config.shared_bytes));
709 
710  cudaDeviceProp deviceProp;
711  cudaGetDeviceProperties(&deviceProp, 0);
712 
713  size_t total_global_mem = deviceProp.totalGlobalMem;
714  size_t num_sm = deviceProp.multiProcessorCount;
715  size_t num_threads_per_sm = threads_per_sm(deviceProp);
716 
717  auto& config = root.config;
718  config.or_nodes = (config.or_nodes == 0) ? hint_num_blocks : config.or_nodes;
719 
720  // The stack allocated depends on the maximum number of threads per SM, not on the actual number of threads per block.
721  size_t total_stack_size = num_sm * deviceProp.maxThreadsPerMultiProcessor * config.stack_kb * 1000;
722  size_t remaining_global_mem = total_global_mem - total_stack_size;
723  remaining_global_mem -= remaining_global_mem / 10; // We leave 10% of global memory free for CUDA allocations, not sure if it is useful though.
724 
725  // Basically the size of the store and propagator, and 100 bytes per variable.
726  // +1 for the root node in GridCP.
727  size_t heap_usage_estimation = (config.or_nodes + 1) * (mem_config.pc_bytes + mem_config.store_bytes + 100 * root.store->vars());
728  while(heap_usage_estimation > remaining_global_mem) {
729  config.or_nodes--;
730  }
731 
732  CUDAEX(cudaDeviceSetLimit(cudaLimitStackSize, config.stack_kb*1000));
733  CUDAEX(cudaDeviceSetLimit(cudaLimitMallocHeapSize, remaining_global_mem));
734 
735  root.stats.print_memory_statistics(config.verbose_solving, "stack_memory", total_stack_size);
736  root.stats.print_memory_statistics(config.verbose_solving, "heap_memory", remaining_global_mem);
737  root.stats.print_memory_statistics(config.verbose_solving, "heap_usage_estimation", heap_usage_estimation);
738  if(config.verbose_solving) {
739  printf("%% or_nodes=%zu\n", config.or_nodes);
740  }
741 }
