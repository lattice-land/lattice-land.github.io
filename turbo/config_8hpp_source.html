147  CUDA void print_commandline(const char* program_name) {
148  printf("%s -t %" PRIu64 " %s-n %" PRIu64 " %s%s%s%s%s",
149  program_name,
150  timeout_ms,
151  (print_intermediate_solutions ? "-a ": ""),
152  stop_after_n_solutions,
153  (print_intermediate_solutions ? "-i ": ""),
154  (free_search ? "-f " : ""),
155  (print_statistics ? "-s " : ""),
156  (verbose_solving ? "-v " : ""),
157  (print_ast ? "-ast " : "")
158  );
159  if(arch == Arch::GPU || arch == Arch::HYBRID) {
160  printf("-arch %s -or %" PRIu64 " -sub %" PRIu64 " -stack %" PRIu64 " ", name_of_arch(arch), or_nodes, subproblems_power, stack_kb);
161  if(only_global_memory) { printf("-globalmem "); }
162  }
163  else {
164  printf("-arch cpu -p %" PRIu64 " ", or_nodes);
165  }
166  if(disable_simplify) { printf("-disable_simplify "); }
167  if(force_ternarize) { printf("-force_ternarize "); }
168  if(network_analysis) { printf("-network_analysis "); }
169  printf("-fp %s ", name_of_fixpoint(fixpoint));
170  if(fixpoint == FixpointKind::WAC1) {
171  printf("-wac1_threshold %" PRIu64 " ", wac1_threshold);
172  }
173  if(version.size() != 0) {
174  printf("-version %s ", version.data());
175  }
176  if(hardware.size() != 0) {
177  printf("-hardware \'%s\' ", hardware.data());
178  }
179  printf("-cutnodes %" PRIu64 " ", stop_after_n_nodes == std::numeric_limits<size_t>::max() ? 0 : stop_after_n_nodes);
180  printf("%s", problem_path.data());
181  }
