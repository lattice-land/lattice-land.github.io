174  // When activated (`deps.is_shared_copy()`), we avoid copying the propagators and share them with the ones of the root `other`.
175  // This allows to save up memory and to avoid contention on L2 cache among blocks.
176  template<class A2, class Alloc2, class... Allocators>
177  CUDA static bytecodes_ptr init_bytecodes(const PIR<A2, Alloc2>& other, AbstractDeps<Allocators...>& deps) {
178  auto alloc = deps.template get_allocator<allocator_type>();
179  if constexpr(std::is_same_v<allocator_type, Alloc2>) {
180  if(deps.is_shared_copy()) {
181  assert(static_cast<bool>(other.bytecodes));
182  return other.bytecodes;
183  }
184  }
185  bytecodes_ptr r = battery::allocate_root<bytecodes_type, allocator_type>(alloc, *(other.bytecodes), alloc);
186  return std::move(r);

245  /** We interpret the formula `f` in the value `intermediate`, note that we only add one constraint to `intermediate` if the interpretation succeeds. */
246  template <IKind kind, bool diagnose, class F, class Env, class Intermediate>
247  CUDA bool interpret_formula(const F& f, Env& env, Intermediate& intermediate, IDiagnostics& diagnostics) const {
248  if(f.type() != aty() && !f.is_untyped()) {
249  RETURN_INTERPRETATION_ERROR("The type of the formula does not match the type of this abstract domain.");
250  }
251  if(f.is_binary()) {
252  Sig sig = f.sig();
253  // Expect constraint of the form X = Y <OP> Z, or Y <OP> Z = X.
254  int left = f.seq(0).is_binary() ? 1 : 0;
255  int right = f.seq(1).is_binary() ? 1 : 0;
256  if(sig == EQ && (left + right == 1)) {
257  auto& X = f.seq(left);
258  auto& Y = f.seq(right).seq(0);
259  auto& Z = f.seq(right).seq(1);
260  bytecode_type bytecode;
261  bytecode.op = f.seq(right).sig();
262  if(X.is_variable() && Y.is_variable() && Z.is_variable() &&
263  (bytecode.op == ADD || bytecode.op == MUL || bytecode.op == EDIV || bytecode.op == EMOD
264  || bytecode.op == MIN || bytecode.op == MAX
265  || bytecode.op == EQ || bytecode.op == LEQ))
266  {
267  if( env.template interpret<diagnose>(X, bytecode.x, diagnostics)
268  && env.template interpret<diagnose>(Y, bytecode.y, diagnostics)
269  && env.template interpret<diagnose>(Z, bytecode.z, diagnostics))
270  {
271  intermediate.bytecodes.push_back(bytecode);
272  return true;
273  }
274  RETURN_INTERPRETATION_ERROR("Could not interpret the variables in the environment.");
275  }
276  }
277  }
278  RETURN_INTERPRETATION_ERROR("The shape of this formula is not supported.");
279  }

282  template <IKind kind, bool diagnose = false, class F, class Env, class I>
283  CUDA NI bool interpret(const F& f, Env& env, I& intermediate, IDiagnostics& diagnostics) const {
284  size_t error_context = 0;
285  if constexpr(diagnose) {
286  diagnostics.add_suberror(IDiagnostics(false, name, "Uninterpretable formula in both PIR and its sub-domain.", f));
287  error_context = diagnostics.num_suberrors();
288  }
289  bool res = false;
290  AType current = f.type();
291  const_cast<F&>(f).type_as(sub->aty()); // We will restore the type after the call to sub->interpret.
292  if(sub->template interpret<kind, diagnose>(f, env, intermediate.sub_value, diagnostics)) {
293  res = true;
294  }
295  const_cast<F&>(f).type_as(current);
296  if(!res) {
297  res = interpret_formula<kind, diagnose>(f, env, intermediate, diagnostics);
298  }
299  if constexpr(diagnose) {
300  diagnostics.merge(res, error_context);
301  }
302  return res;
303  }

317  /** Similar limitations than `PC::deduce`. */
318  template <class Alloc2>
319  CUDA local::B deduce(const tell_type<Alloc2>& t) {
320  local::B has_changed = sub->deduce(t.sub_value);
321  if(t.bytecodes.size() > 0) {
322  bytecodes->reserve(bytecodes->size() + t.bytecodes.size());
323  for(int i = 0; i < t.bytecodes.size(); ++i) {
324  bytecodes->push_back(t.bytecodes[i]);
325  }
326  /** This is sorting the constraints `X = Y <op> Z` according to <OP>.
327  * Note that battery::sorti is much slower than std::sort, therefore the #ifdef. */
328  #ifdef __CUDA_ARCH__
329  battery::sorti(*bytecodes,
330  [&](int i, int j) { return (*bytecodes)[i].op < (*bytecodes)[j].op; });
331  #else
332  std::stable_sort(bytecodes->data(), bytecodes->data() + bytecodes->size(),
333  [](const bytecode_type& a, const bytecode_type& b) {
334  // return a.op < b.op;
335  return a.op == b.op ? (a.y.vid() == b.y.vid() ? (a.x.vid() == b.x.vid() ? a.z.vid() < b.z.vid() : a.x.vid() < b.x.vid()) : a.y.vid() < b.y.vid()) : a.op < b.op;
336  });
337  #endif
338  has_changed = true;
339  }
340  return has_changed;
341  }

348  CUDA local::B ask(bytecode_type bytecode) const {
349  // We load the variables.
350  local_universe_type r1((*sub)[bytecode.x]);
351  local_universe_type r2((*sub)[bytecode.y]);
352  local_universe_type r3((*sub)[bytecode.z]);
353 
354  // Reified constraint: X = (Y = Z) and X = (Y <= Z).
355  if(bytecode.op == EQ || bytecode.op == LEQ) {
356  // Y [op] Z
357  if(r1 <= ONE) {
358  return bytecode.op == EQ
359  ? (r2 == r3 && r2.lb().value() == r2.ub().value())
360  : r2.ub().value() <= r3.lb().value();
361  }
362  // not (Y [op] Z)
363  else if(r1 <= ZERO) {
364  return bytecode.op == EQ
365  ? fmeet(r2, r3).is_bot().value()
366  : r2.lb().value() > r3.ub().value();
367  }
368  // We need `r1` to be a singleton to decide whether the constraint is entailed.
369  return false;
370  }
371  // Arithmetic constraint: X = Y + Z, X = Y - Z, ...
372  else {
373  local_universe_type right;
374  right.project(bytecode.op, r2, r3);
375  return right == r1 && r1.lb().value() == r1.ub().value();
376  }
377  }

414  CUDA local::B deduce(bytecode_type bytecode) {
415  local::B has_changed = false;
416  // We load the variables.
417  local_universe_type r1((*sub)[bytecode.x]);
418  local_universe_type r2((*sub)[bytecode.y]);
419  local_universe_type r3((*sub)[bytecode.z]);
420  // Reified constraint: X = (Y = Z) and X = (Y <= Z).
421  if(bytecode.op == EQ || bytecode.op == LEQ) {
422  // Y [op] Z
423  if(r1 <= ONE) {
424  if(bytecode.op == LEQ) {
425  r3.join_lb(LB::top());
426  r2.join_ub(UB::top());
427  }
428  has_changed |= sub->embed(bytecode.y, r3);
429  has_changed |= sub->embed(bytecode.z, r2);
430  }
431  // not (Y [op] Z)
432  else if(r1 <= ZERO) {
433  if(bytecode.op == EQ) {
434  if(r2.lb().value() == r2.ub().value()) {
435  r1 = r3;
436  r1.meet_lb(LB::prev(r2.lb()));
437  r3.meet_ub(UB::prev(r2.ub()));
438  has_changed |= sub->embed(bytecode.z, fjoin(r1,r3));
439  }
440  else if(r3.lb().value() == r3.ub().value()) {
441  r1 = r2;
442  r1.meet_lb(LB::prev(r3.lb()));
443  r2.meet_ub(UB::prev(r3.ub()));
444  has_changed |= sub->embed(bytecode.y, fjoin(r1,r2));
445  }
446  }
447  else {
448  assert(bytecode.op == LEQ);
449  r2.meet_lb(LB::prev(r3.lb()));
450  r3.meet_ub(UB::prev(r2.ub()));
451  has_changed |= sub->embed(bytecode.y, r2);
452  has_changed |= sub->embed(bytecode.z, r3);
453  }
454  }
455  // X <- 1
456  else if(r2.ub().value() <= r3.lb().value() && (bytecode.op == LEQ || r2.lb().value() == r3.ub().value())) {
457  has_changed |= sub->embed(bytecode.x, ONE);
458  }
459  // X <- 0
460  else if(r2.lb().value() > r3.ub().value() || (bytecode.op == EQ && r2.ub().value() < r3.lb().value())) {
461  has_changed |= sub->embed(bytecode.x, ZERO);
462  }
463  }
464  // Arithmetic constraint: X = Y + Z, X = Y - Z, ...
465  else {
466  // X <- Y [op] Z
467  r1.project(bytecode.op, r2, r3);
468  has_changed |= sub->embed(bytecode.x, r1);
469 
470  // Y <- X <left residual> Z
471  switch(bytecode.op) {
472  case ADD: pc::GroupAdd<local_universe_type>::left_residual(r1, r3, r2); break;
473  case MUL: pc::GroupMul<local_universe_type, EDIV>::left_residual(r1, r3, r2); break;
474  case EDIV: pc::GroupDiv<local_universe_type, EDIV>::left_residual(r1, r3, r2); break;
475  case EMOD: {
476  if(r1.is_bot() || r3.is_bot()) {
477  r2.meet_bot();
478  }
479  break;
480  }
481  case MIN: pc::GroupMinMax<local_universe_type, MIN>::left_residual(r1, r3, r2); break;
482  case MAX: pc::GroupMinMax<local_universe_type, MAX>::left_residual(r1, r3, r2); break;
483  default: assert(false);
484  }
485  has_changed |= sub->embed(bytecode.y, r2);
486 
487  // Z <- X <right residual> Y
488  switch(bytecode.op) {
489  case ADD: pc::GroupAdd<local_universe_type>::right_residual(r1, r2, r3); break;
490  case MUL: pc::GroupMul<local_universe_type, EDIV>::right_residual(r1, r2, r3); break;
491  case EDIV: pc::GroupDiv<local_universe_type, EDIV>::right_residual(r1, r2, r3); break;
492  case EMOD: {
493  if(r1.is_bot() || r2.is_bot()) {
494  r3.meet_bot();
495  }
496  break;
497  }
498  case MIN: pc::GroupMinMax<local_universe_type, MIN>::right_residual(r1, r2, r3); break;
499  case MAX: pc::GroupMinMax<local_universe_type, MAX>::right_residual(r1, r2, r3); break;
500  default: assert(false);
501  }
502  has_changed |= sub->embed(bytecode.z, r3);
503  }
504  return has_changed;
505  }
