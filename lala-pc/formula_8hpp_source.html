33  template<bool negate>
34  CUDA local::BInc ask_impl(const A& a) const {
35  if constexpr(negate) {
36  return a.project(avar) >= U::eq_zero();
37  }

45  if constexpr(negate) {
46  a.tell(avar, U::eq_zero(), has_changed);
47  }
48  else {
49  a.tell(avar, U::eq_one(), has_changed);
50  }

53 public:
54  /** Given a variable `x` taking a value in a universe `U` denoted by \f$ a(x) \f$.
55  * - \f$ a \vDash x \f$ holds iff \f$ \lnot (a(x) \leq [\![x = 0]\!]_U) \f$.
56  * - \f$ a \vDash \lnot{x} \f$ holds iff \f$ a(x) \geq [\![x = 0]\!]_U \f$. */

231  Seq<Alloc> seq2{seq.get_allocator()};
232  for(int i = 0; i < seq.size(); ++i) {
233  seq2.push_back(map_avar(seq[i], t));

238  template <class Alloc>
239  CUDA F<Alloc> map_avar(const F<Alloc>& f, const F<Alloc>& t) const {
240  switch(f.index()) {
241  case F<Alloc>::V: return t;
242  case F<Alloc>::Seq: return F<Alloc>::make_nary(f.sig(), map_seq(f.seq(), t), f.type());
243  case F<Alloc>::ESeq: return F<Alloc>::make_nary(f.esig(), map_seq(f.eseq(), t), f.type());
244  default: return f;
245  }
246  }

254  auto tf = left.deinterpret(alloc, apc);
255  return map_avar(uf, tf);

681  * A term can occur in a formula, as it is usually done.
682  * Therefore, a formula is both a formula and a term.

685  * - Consists of four distinct values \f$\{[\![x = 0]\!]_U \sqcap [\![x = 1]\!]_U, [\![x = 0]\!]_U, [\![x = 1]\!]_U, \top_U \}\f$.
