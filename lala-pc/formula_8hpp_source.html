14/** A Boolean variable defined on a universe of discourse supporting `0` for false and `1` for true. */
15template<class AD, bool neg>
16class VariableLiteral {
17public:
18 using A = AD;
19 using U = typename A::local_universe;
20
21 template <class A2, bool neg2>
22 friend class VariableLiteral;
23
24private:
25 AVar avar;
26
27public:
28 VariableLiteral() = default;
29 CUDA VariableLiteral(AVar avar): avar(avar) {}
30
31 template <class A2, class Alloc>
32 CUDA VariableLiteral(const VariableLiteral<A2, neg>& other, const Alloc&): avar(other.avar) {}
33
34private:
35 template<bool negate>
36 CUDA local::BInc ask_impl(const A& a) const {
37 if constexpr(negate) {
38 return a.project(avar) >= U::eq_zero();
39 }
40 else {
41 return !(a.project(avar) <= U::eq_zero());
42 }
43 }
44
45 template <bool negate, class Mem>
46 CUDA void refine_impl(A& a, BInc<Mem>& has_changed) const {
47 if constexpr(negate) {
48 a.tell(avar, U::eq_zero(), has_changed);
49 }
50 else {
51 a.tell(avar, U::eq_one(), has_changed);
52 }
53 }
54
55public:
56 /** Given a variable `x` taking a value in a universe `U` denoted by \f$ a(x) \f$.
57 * - \f$ a \vDash x \f$ holds iff \f$ \lnot (a(x) \leq [\![x = 0]\!]_U) \f$.
58 * - \f$ a \vDash \lnot{x} \f$ holds iff \f$ a(x) \geq [\![x = 0]\!]_U \f$. */
59 CUDA local::BInc ask(const A& a) const {
60 return ask_impl<neg>(a);
61 }
62
63 /** Given a variable `x` taking a value in a universe `U` denoted by \f$ a(x) \f$.
64 * - \f$ a \nvDash x \f$ holds iff \f$ a(x) \geq [\![x = 0]\!]_U \f$.
65 * - \f$ a \nvDash \lnot{x} \f$ holds iff \f$ \lnot (a(x) \leq [\![x = 0]\!]_U) \f$. */
66 CUDA local::BInc nask(const A& a) const {
67 return ask_impl<!neg>(a);
68 }
69
70 template <class Mem>
71 CUDA void preprocess(A&, BInc<Mem>&) {}
72
73 /** Perform:
74 * * Positive literal: \f$ x = a(x) \sqcup [\![x = 1]\!]_U \f$
75 * * Negative literal: \f$ x = a(x) \sqcup [\![x = 0]\!]_U \f$. */
76 template <class Mem>
77 CUDA void refine(A& a, BInc<Mem>& has_changed) const {
78 refine_impl<neg>(a, has_changed);
79 }
80
81 /** Perform:
82 * * Positive literal: \f$ x = a(x) \sqcup [\![x = 0]\!]_U \f$
83 * * Negative literal: \f$ x = a(x) \sqcup [\![x = 1]\!]_U \f$. */
84 template <class Mem>
85 CUDA void nrefine(A& a, BInc<Mem>& has_changed) const {
86 refine_impl<!neg>(a, has_changed);
87 }
88
89 CUDA local::BInc is_top(const A& a) const {
90 return a.project(avar).is_top();
91 }
92
93 CUDA NI void print(const A& a) const {
94 if constexpr(neg) { printf("not "); }
95 printf("(%d,%d)", avar.aty(), avar.vid());
96 }
97
98 template <class Alloc>
99 CUDA NI TFormula<Alloc> deinterpret(const Alloc&, AType apc) const {
100 auto f = TFormula<Alloc>::make_avar(avar);
101 if(neg) {
102 f = TFormula<Alloc>::make_unary(NOT, f, apc);
103 }
104 return f;
105 }
106
107 CUDA size_t length() const { return 1; }
108};

188/** A predicate of the form `t >= u` where `t` is a term, `u` an element of an abstract universe, and `>=` the lattice order of this abstract universe. */
189template<class AD, class Allocator>
190class PLatticeOrderPredicate {
191public:
192 using A = AD;
193 using U = typename A::local_universe;
194 using allocator_type = Allocator;
195 using this_type = PLatticeOrderPredicate<A, allocator_type>;
196 using sub_type = Term<A, allocator_type>;
197
198 template <class A2, class Alloc2>
199 friend class PLatticeOrderPredicate;
200
201protected:
202 sub_type left;
203 U right;
204
205public:
206 CUDA PLatticeOrderPredicate(sub_type&& left, U&& right): left(std::move(left)), right(std::move(right)) {}
207 CUDA PLatticeOrderPredicate(this_type&& other): PLatticeOrderPredicate(std::move(other.left), std::move(other.right)) {}
208
209 template <class A2, class Alloc2>
210 CUDA PLatticeOrderPredicate(const PLatticeOrderPredicate<A2, Alloc2>& other, const allocator_type& alloc):
211 left(other.left, alloc),
212 right(other.right)
213 {}
214
215 CUDA local::BInc ask(const A& a) const {
216 return left.project(a) >= right;
217 }
218
219 CUDA local::BInc nask(const A&) const { assert(false); return local::BInc::bot(); }
220 template <class Mem> CUDA void nrefine(A&, BInc<Mem>&) const { assert(false); }
221 template <class Mem> CUDA void tell(A&, const U&, BInc<Mem>&) const { assert(false); }
222 CUDA U project(const A&) const { assert(false); return U::top(); }
223
224 CUDA local::BInc is_top(const A& a) const {
225 return left.is_top(a);
226 }
227
228 template <class Mem>
229 CUDA void preprocess(A& a, BInc<Mem>& has_changed) {
230 right.tell(left.project(a), has_changed);
231 }
232
233 template <class Mem>
234 CUDA void refine(A& a, BInc<Mem>& has_changed) const {
235 left.tell(a, right, has_changed);
236 }
237
238 CUDA NI void print(const A& a) const {
239 left.print(a);
240 printf(" >= ");
241 right.print();
242 }
243
244private:
245 template <class Alloc> using F = TFormula<Alloc>;
246 template <class Alloc> using Seq = typename F<Alloc>::Sequence;
247
248 template <class Alloc>
249 CUDA Seq<Alloc> map_seq(const Seq<Alloc>& seq, const F<Alloc>& t) const {
250 Seq<Alloc> seq2{seq.get_allocator()};
251 for(int i = 0; i < seq.size(); ++i) {
252 seq2.push_back(map_avar(seq[i], t));
253 }
254 return std::move(seq2);
255 }
256
257 template <class Alloc>
258 CUDA F<Alloc> map_avar(const F<Alloc>& f, const F<Alloc>& t) const {
259 switch(f.index()) {
260 case F<Alloc>::V: return t;
261 case F<Alloc>::Seq: return F<Alloc>::make_nary(f.sig(), map_seq(f.seq(), t), f.type());
262 case F<Alloc>::ESeq: return F<Alloc>::make_nary(f.esig(), map_seq(f.eseq(), t), f.type());
263 default: return f;
264 }
265 }
266
267public:
268 template <class Alloc>
269 CUDA NI TFormula<Alloc> deinterpret(const Alloc& alloc, AType apc) const {
270 // We deinterpret the constant with a placeholder variable that is then replaced by the interpretation of the left term.
271 VarEnv<Alloc> empty_env = VarEnv<Alloc>(alloc);
272 auto uf = right.deinterpret(AVar{}, empty_env);
273 auto tf = left.deinterpret(alloc, apc);
274 return map_avar(uf, tf);
275 }
276
277 CUDA size_t length() const { return left.length() + 2; }
278};

773 if(l.lb() == dual<LB>(l.ub())) {
774 if constexpr(U::complemented) {

776 }

778 right.tell(a, meet(
779 U(r).tell_ub(UB::next(l.ub())),
780 U(r).tell_lb(LB::next(l.lb()))), has_changed);

782 }

790 U(l).tell_lb(LB::next(r.lb()))), has_changed);
791 }
792 }

796 right.tell(a, left.project(a), has_changed);

850/**
