32 /** PC is an abstract transformer built on top of an abstract domain `A`.
33  It is expected that `A` has a projection function `u = project(x)`.
34  We also expect a `tell(x, u, has_changed)` function to join the abstract universe `u` in the domain of the variable `x`.
35  An example of abstract domain satisfying these requirements is `VStore<Interval<ZInc>>`. */

170  template<class Alloc, class F>
171  CUDA NI static tresult<Alloc, F> interpret_unary(const F& f, term_ptr<Alloc>&& a) {
172  using T = pc::Term<A, Alloc>;
173  Alloc alloc = a.get_allocator();
174  switch(f.sig()) {
175  case NEG: return T::make_neg(std::move(a));
176  case ABS: return T::make_abs(std::move(a));
177  default: return tresult<Alloc, F>(IError<F>(true, name, "Unsupported unary symbol.", f));

181  template<class Alloc, class F>
182  CUDA NI static tresult<Alloc, F> interpret_binary(const F& f, term_ptr<Alloc>&& x, term_ptr<Alloc>&& y)
183  {
184  using T = pc::Term<A, Alloc>;
185  switch(f.sig()) {
186  case ADD: return T::make_add(std::move(x), std::move(y));
187  case SUB: return T::make_sub(std::move(x), std::move(y));
188  case MUL: return T::make_mul(std::move(x), std::move(y));
189  case TDIV: return T::make_tdiv(std::move(x), std::move(y));
190  case FDIV: return T::make_fdiv(std::move(x), std::move(y));
191  case CDIV: return T::make_cdiv(std::move(x), std::move(y));
192  case EDIV: return T::make_ediv(std::move(x), std::move(y));
193  case MIN: return T::make_min(std::move(x), std::move(y));
194  case MAX: return T::make_max(std::move(x), std::move(y));
195  default: return tresult<Alloc, F>(IError<F>(true, name, "Unsupported binary symbol.", f));
196  }
197  }

210  template <bool is_tell, class F, class Env, class Alloc = typename Env::allocator_type>
211  CUDA NI tresult<Alloc, F> interpret_sequence(const F& f, Env& env)
212  {
213  using T = pc::Term<A, Alloc>;
214  Alloc alloc = env.get_allocator();
215  battery::vector<T, Alloc> subterms(alloc);
216  subterms.reserve(f.seq().size());
217  for(int i = 0; i < f.seq().size(); ++i) {
218  auto t = interpret_term<is_tell>(f.seq(i), env);
219  if(!t.has_value()) {
220  auto p = interpret_formula<is_tell>(f.seq(i), env, true);
221  if(!p.has_value()) {
222  return std::move(t.join_errors(std::move(p)));
223  }
224  t = tresult<Alloc, F>(T::make_formula(
225  battery::allocate_unique<pc::Formula<A, Alloc>>(alloc, std::move(p.value()))));
226  t.join_warnings(std::move(p));
227  }
228  subterms.push_back(std::move(t.value()));
229  }
230  if(subterms.size() == 1) {
231  return interpret_unary(f,
232  battery::allocate_unique<T>(alloc, std::move(subterms[0])));
233  }
234  else if(subterms.size() == 2) {
235  return interpret_binary(f,
236  battery::allocate_unique<T>(alloc, std::move(subterms[0])),
237  battery::allocate_unique<T>(alloc, std::move(subterms[1])));
238  }
239  else {
240  return interpret_nary(f, std::move(subterms));
241  }
242  }

244  template <class F, class Env>
245  CUDA NI IResult<AVar, F> interpret_var(const F& f, Env& env) {
246  if(f.is(F::V)) {
247  assert(!f.v().is_untyped());
248  return IResult<AVar, F>(f.v());
249  }
250  else {
251  auto x = var_in(f, env);
252  if(!x.has_value()) {
253  return IResult<AVar, F>(IError<F>(true, name, "The variable is not declared in the environment, and thus cannot be accessed by the propagator.", f));
254  }
255  auto avar = x->avar_of(sub->aty());
256  if(avar.has_value()) {
257  return IResult<AVar, F>(std::move(*avar));
258  }
259  else {
260  return IResult<AVar, F>(IError<F>(true, name, "The variable is not declared in the sub abstract domain of PC, and thus cannot be accessed by the propagator.", f));
261  }
262  }
263  }

265  template <bool is_tell, class F, class Env, class Alloc = typename Env::allocator_type>
266  CUDA NI tresult<Alloc, F> interpret_term(const F& f, Env& env) {
267  using T = pc::Term<A, Alloc>;
268  Alloc alloc = env.get_allocator();
269  if(f.is_variable()) {
270  auto avar = interpret_var(f, env);
271  if(avar.has_value()) {
272  return T::make_var(avar.value());
273  }
274  else {
275  return std::move(avar).template map_error<pc::Term<A, Alloc>>();
276  }
277  }
278  else if(f.is_constant()) {
279  using F2 = TFormula<typename Env::allocator_type>;
280  auto constant = F2::make_binary(F2::make_avar(AVar()), EQ, f, UNTYPED, env.get_allocator());
281  auto k = is_tell ? universe_type::interpret_tell(constant, env) : universe_type::interpret_ask(constant, env);
282  if(k.has_value()) {
283  return std::move(tresult<Alloc, F>(T::make_constant(std::move(k.value())))
284  .join_warnings(std::move(k)));
285  }
286  else {
287  return std::move(tresult<Alloc, F>(IError<F>(true, name, "Constant in a term could not be interpreted in the underlying abstract universe.", f))
288  .join_errors(std::move(k)));
289  }
290  }
291  else if(f.is(F::Seq)) {
292  return interpret_sequence<is_tell, F>(f, env);
293  }
294  else {
295  return tresult<Alloc, F>(IError<F>(true, name, "The shape of the formula is not supported in PC, and could not be interpreted as a term.", f));
296  }
297  }

310  template<bool is_tell, class Create, class F, class Env, class Alloc = typename Env::allocator_type>
311  CUDA NI fresult<Alloc, F> interpret_binary_logical_connector(const F& f, const F& g, Env& env, bool neg_context, Create&& create) {
312  using PF = pc::Formula<A, Alloc>;
313  Alloc alloc = env.get_allocator();
314  auto l = interpret_formula<is_tell>(f, env, neg_context);
315  if(l.has_value()) {
316  auto k = interpret_formula<is_tell>(g, env, neg_context);
317  if(k.has_value()) {
318  return std::move(fresult<Alloc, F>(create(
319  battery::allocate_unique<PF>(alloc, std::move(l.value())),
320  battery::allocate_unique<PF>(alloc, std::move(k.value())))
321  )
322  .join_warnings(std::move(l))
323  .join_warnings(std::move(k)));
324  }
325  else {
326  return std::move(k);
327  }
328  }
329  else {
330  return std::move(l);
331  }
332  }

409  template <bool is_tell, class F, class Env, class Alloc = typename Env::allocator_type>
410  CUDA NI fresult<Alloc, F> interpret_in_decomposition(const F& f, Env& env, bool neg_context) {
411  assert(f.seq(1).is(F::S));
412  using F2 = TFormula<typename Env::allocator_type>;
413  typename Env::allocator_type alloc = env.get_allocator();
414  // Decompose IN into disjunction.
415  typename F2::Sequence seq;
416  const auto& set = f.seq(1).s();
417  for(size_t i = 0; i < set.size(); ++i) {
418  if(battery::get<0>(set[i]) == battery::get<1>(set[i])) {
419  seq.push_back(F2::make_binary(f.seq(0), EQ, battery::get<0>(set[i]), f.type(), alloc));
420  }
421  else {
422  seq.push_back(
423  F2::make_binary(
424  F2::make_binary(f.seq(0), GEQ, battery::get<0>(set[i]), f.type(), alloc),
425  AND,
426  F2::make_binary(f.seq(0), LEQ, battery::get<1>(set[i]), f.type(), alloc),
427  f.type(),
428  alloc));
429  }
430  }
431  if(seq.size() == 1) {
432  return interpret_formula<is_tell>(seq[0], env, neg_context);
433  }
434  else {
435  return interpret_formula<is_tell>(F2::make_nary(OR, seq, f.type()), env, neg_context);
436  }
437  }

450  template <bool is_tell, class F, class Env, class Alloc = typename Env::allocator_type>
451  CUDA NI fresult<Alloc, F> interpret_formula(const F& f, Env& env, bool neg_context = false) {
452  assert(f.type() == aty() || f.is_untyped() || f.is_variable());
453  Alloc alloc = env.get_allocator();
454  using PF = pc::Formula<A, Alloc>;
455  if(f.is_false()) {
456  return PF::make_false();
457  }
458  else if(f.is_true()) {
459  return PF::make_true();
460  }
461  else if(f.is(F::Seq) && f.sig() == IN) {
462  return interpret_in_decomposition<is_tell>(f, env, neg_context);
463  }
464  else if(f.is(F::Seq) && f.seq().size() > 2 && (f.sig() == AND || f.sig() == OR || f.sig() == EQUIV)) {
465  return interpret_formula<is_tell>(binarize(f,0), env, neg_context);
466  }
467  else if(f.is_binary()) {
468  Sig sig = f.sig();
469  switch(sig) {
470  case AND: return interpret_conjunction<is_tell>(f.seq(0), f.seq(1), env, neg_context);
471  case OR: return interpret_disjunction<is_tell>(f.seq(0), f.seq(1), env);
472  case EQUIV: return interpret_biconditional<is_tell>(f.seq(0), f.seq(1), env);
473  case IMPLY: return interpret_implication<is_tell>(f.seq(0), f.seq(1), env);
474  case XOR: return interpret_xor<is_tell>(f.seq(0), f.seq(1), env);
475  case EQ: // Whenever an operand of `=` is a formula with logical connectors, we interpret `=` as `<=>`.
476  if(f.seq(0).is_logical() || f.seq(1).is_logical()) {
477  return interpret_biconditional<is_tell>(f.seq(0), f.seq(1), env);
478  }
479  // Form of the constraint `T <op> u` with `x <op> u` interpreted in the underlying universe.
480  default:
481  auto fn = move_constants_on_rhs(f);
482  using F2 = TFormula<typename Env::allocator_type>;
483  auto fu = F2::make_binary(F2::make_avar(AVar()), fn.sig(), fn.seq(1), fn.type(), alloc);
484  auto u = is_tell ? universe_type::interpret_tell(fu, env) : universe_type::interpret_ask(fu, env);
485  if(!u.has_value() && fn.sig() == NEQ) {
486  return interpret_neq_decomposition<is_tell>(fn, env, neg_context);
487  }
488  else if(u.has_value()) {
489  auto term = interpret_term<is_tell>(fn.seq(0), env);
490  if(term.has_value()) {
491  // In a context where the formula propagator can be asked for its negation, we must interpret the negation of the formula as well.
492  if(neg_context) {
493  auto nf_ = negate(fn);
494  if(nf_.has_value()) {
495  auto nf = interpret_formula<is_tell>(*nf_, env);
496  if(nf.has_value()) {
497  auto data = PF::make_nlop(std::move(term.value()), std::move(u.value()),
498  battery::allocate_unique<PF>(alloc, std::move(nf.value())));
499  return std::move(fresult<Alloc, F>(std::move(data))
500  .join_warnings(std::move(nf))
501  .join_warnings(std::move(u))
502  .join_warnings(std::move(term)));
503  }
504  else {
505  return std::move(fresult<Alloc, F>(IError<F>(true, name, "We must query this formula for disentailement, but we could not interpret its negation.", f))
506  .join_errors(std::move(nf))
507  .join_warnings(std::move(u))
508  .join_warnings(std::move(term)));
509  }
510  }
511  else {
512  return std::move(fresult<Alloc, F>(IError<F>(true, name, "We must query this formula for disentailement, but we could not compute its negation.", f))
513  .join_warnings(std::move(u))
514  .join_warnings(std::move(term)));
515  }
516  }
517  else {
518  auto data = PF::make_plop(std::move(term.value()), std::move(u.value()));
519  return std::move(fresult<Alloc, F>(std::move(data))
520  .join_warnings(std::move(u))
521  .join_warnings(std::move(term)));
522  }
523  }
524  else {
525  return std::move(fresult<Alloc, F>(IError<F>(true, name, "We cannot interpret the term on the LHS of the formula in PC.", f))
526  .join_warnings(std::move(u))
527  .join_errors(std::move(term)));
528  }
529  }
530  else {
531  return std::move(fresult<Alloc, F>(IError<F>(true, name, "We cannot interpret the constant on the RHS of the formula in the underlying abstract universe.", f))
532  .join_errors(std::move(u)));
533  }
534  }
535  }
536  // Negative literal
537  else if(f.is(F::Seq) && f.seq().size() == 1 && f.sig() == NOT &&
538  f.seq(0).is_variable())
539  {
540  return interpret_literal<true>(f.seq(0), env);
541  }
542  // Positive literal
543  else if(f.is_variable()) {
544  return interpret_literal<false>(f, env);
545  }
546  // Logical negation

550  else {
551  return fresult<Alloc, F>(IError<F>(true, name, "The shape of this formula is not supported.", f));

578  template <class R, bool is_tell, class F, class Env>
579  CUDA NI R interpret_in(const F& f, Env& env) {
580  using val_t = typename R::value_type;
581  // If the formula is untyped, we first try to interpret it in the sub-domain.
582  IResult<int, F> sub_err{0};
583  if(f.is_untyped() || f.type() != aty()) {
584  auto r = is_tell ? sub->interpret_tell_in(f, env) : sub->interpret_ask_in(f, env);
585  if(r.has_value()) {
586  return std::move(r).map(val_t(std::move(r.value()), env.get_allocator()));
587  }
588  if(f.type() != UNTYPED) {
589  return std::move(R(IError<F>(true, name,
590  "The formula could not be interpreted in the sub-domain, and it has an abstract type different from the one of the current element.", f))
591  .join_errors(std::move(r)));
592  }
593  sub_err = IResult<int, F>(std::move(r.error()));
594  }
595  assert(f.is_untyped() || f.type() == aty());
596  // Conjunction
597  if(f.is(F::Seq) && f.sig() == AND) {
598  auto split_formula = extract_ty(f, aty());
599  const auto& ipc_formulas = battery::get<0>(split_formula);
600  const auto& other_formulas = battery::get<1>(split_formula);
601  R res(val_t(env.get_allocator()));
602  // We need to interpret the formulas in the sub-domain first because it might handle existential quantifiers needed by formulas interpreted in this domain.
603  for(int i = 0; i < other_formulas.seq().size(); ++i) {
604  auto sub_tell = is_tell ? sub->interpret_tell_in(other_formulas.seq(i), env) : sub->interpret_ask_in(other_formulas.seq(i), env);
605  if(sub_tell.has_value()) {
606  res.value().sub_tells.push_back(std::move(sub_tell.value()));
607  res.join_warnings(std::move(sub_tell));
608  }
609  else if(other_formulas.seq(i).type() == UNTYPED) {
610  interpret_formula2<is_tell>(other_formulas.seq(i), env, res);
611  if(!res.has_value()) {
612  res.join_errors(std::move(sub_tell));
613  return std::move(res);
614  }
615  }
616  else {
617  return std::move(R(IError<F>(true, name,
618  "The formula could not be interpreted in the sub-domain and it has an abstract type different from the one of the current element.", f))
619  .join_errors(std::move(sub_tell)));
620  }
621  }
622  for(int i = 0; i < ipc_formulas.seq().size(); ++i) {
623  const auto& g = ipc_formulas.seq(i);
624  interpret_formula2<is_tell>(ipc_formulas.seq(i), env, res);
625  if(!res.has_value()) {
626  return std::move(std::move(res).join_errors(std::move(sub_err)));
627  }
628  }
629  return std::move(res);
630  }
631  else {
632  R res(val_t(env.get_allocator()));
633  interpret_formula2<is_tell>(f, env, res);
634  if(!res.has_value()) {
635  return std::move(std::move(res).join_errors(std::move(sub_err)));
636  }
637  return std::move(res);
638  }
639  }
