226 template<class Universe, Sig divsig>
227 struct GroupMul {
228  using U = Universe;
229 
230  CUDA static U op(const U& a, const U& b) {
231  return U::template fun<MUL>(a, b);
232  }
233 
234  CUDA static bool is_absorbing(const U& a) {
235  return a == U::eq_zero();
236  }
237 
238  /** \pre `is_absorbing(b)` must be `false`. */
239  CUDA static U rev_op(const U& a, const U& b) {
240  return U::template fun<divsig>(a, b);
241  }
242 
243  /** If `a` or `b` contains 0, then we cannot say anything on the inverse since 0 is absorbing and the inverse could be anything. */
244  CUDA static U inv1(const U& a, const U& b) {
245  return b <= U::eq_zero() && a <= U::eq_zero()
246  ? U::bot()
247  : U::template fun<divsig>(a, b);
248  }
249 
250  CUDA static U inv2(const U& a, const U& b) {
251  return inv1(a, b);
252  }
253 
254  static constexpr bool prefix_symbol = false;
255  CUDA static char symbol() { return '*'; }
256  CUDA static Sig sig() { return MUL; }
257 };

348  /** Enforce `x <op> y >= u` where >= is the lattice order of the underlying abstract universe.
349  For instance, over the interval abstract universe, `x + y >= [2..5]` will ensure that `x + y` is eventually at least `2` and at most `5`. */
350  CUDA void tell(A& a, const U& u, local::BInc& has_changed) const {
351  auto xt = x().project(a);
352  auto yt = y().project(a);
353  if(xt.is_top() || yt.is_top()) { return; }
354  if(!x().is(sub_type::IConstant)) {
355  x().tell(a, G::inv1(u, yt), has_changed); // x <- u <inv> y
356  }
357  if(!y().is(sub_type::IConstant)) {
358  y().tell(a, G::inv2(u, x().project(a)), has_changed); // y <- u <inv> x
359  }
360  }
361 
362  CUDA U project(const A& a) const {
363  return G::op(x().project(a), y().project(a));
364  }
365 
366  CUDA local::BInc is_top(const A& a) const {
367  return x().is_top(a) || y().is_top(a);
368  }
369 
370  CUDA NI void print(const A& a) const {
371  if constexpr(G::prefix_symbol) {
372  printf("%s(", G::symbol());
373  x().print(a);
374  printf(", ");
375  x().print(a);
376  printf(")");
377  }
378  else {
379  x().print(a);
380  printf(" %c ", G::symbol());
381  y().print(a);
382  }
383  }

396 // Nary is only valid for commutative group (e.g., addition and multiplication).
397 template <class Combinator>
398 class Nary {
399 public:
400  using this_type = Nary<Combinator>;
401  using allocator_type = typename Combinator::allocator_type;
402  using A = typename Combinator::A;
403  using U = typename Combinator::U;
404  using G = typename Combinator::G;
405 
406  template <class Combinator2>
407  friend class Nary;
408 private:
409  using sub_type = Term<A, allocator_type>;
410  battery::vector<sub_type, allocator_type> terms;
411 
412  CUDA INLINE const sub_type& t(size_t i) const {
413  return terms[i];
414  }
415 
416 public:
417  CUDA Nary(battery::vector<sub_type, allocator_type>&& terms): terms(std::move(terms)) {}
418  CUDA Nary(this_type&& other): Nary(std::move(other.terms)) {}
419 
420  template <class Combinator2>
421  CUDA Nary(const Nary<Combinator2>& other, const allocator_type& allocator)
422  : terms(other.terms, allocator)
423  {}
424 
425  CUDA U project(const A& a) const {
426  U accu = t(0).project(a);
427  for(int i = 1; i < terms.size(); ++i) {
428  accu = G::op(accu, t(i).project(a));
429  }
430  return accu;
431  }
432 
433  CUDA void tell(A& a, const U& u, local::BInc& has_changed) const {
434  U all = project(a);
435  if(G::is_absorbing(all)) {
436  return;
437  }
438  for(int i = 0; i < terms.size(); ++i) {
439  t(i).tell(a, G::inv1(u, G::rev_op(all, t(i).project(a))), has_changed);
440  }
441  }
442 
443  CUDA local::BInc is_top(const A& a) const {
444  for(int i = 0; i < terms.size(); ++i) {
445  if(t(i).is_top(a)) {
446  return local::BInc::top();
447  }
448  }
449  return local::BInc::bot();
450  }
451 
452  CUDA NI void print(const A& a) const {
453  t(0).print(a);
454  for(int i = 1; i < terms.size(); ++i) {
455  printf(" %c ", G::symbol());
456  t(i).print(a);
457  }
458  }
459 
460  template <class Alloc>
461  CUDA NI TFormula<Alloc> deinterpret(const Alloc& alloc, AType apc) const {
462  using F = TFormula<Alloc>;
463  typename F::Sequence seq{alloc};
464  for(int i = 0; i < terms.size(); ++i) {
465  seq.push_back(t(i).deinterpret(alloc, apc));
466  }
467  return TFormula<Alloc>::make_nary(G::sig(), std::move(seq), apc);
468  }
