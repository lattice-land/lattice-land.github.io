154 template<class F>
155 CUDA NI AType type_of_conjunction(const typename F::Sequence& seq) {
156  AType ty = UNTYPED;
157  for(int i = 0; i < seq.size(); ++i) {
158  if(seq[i].type() == UNTYPED) {
159  return UNTYPED;
160  }
161  else if(ty == UNTYPED) {
162  ty = seq[i].type();
163  }
164  else if(ty != seq[i].type()) {
165  return UNTYPED;
166  }
167  }
168  return ty;

171 /** Given a conjunctive formula `f` of the form \f$ c_1 \land ... \land c_n \f$, it returns a pair \f$ \langle c_i \land .. \land c_j, c_k \land ... \land c_l \rangle \f$ such that the first component contains all formulas with the type `ty`, and the second component, all other formulas. */
172 template<class F>
173 CUDA NI battery::tuple<F,F> extract_ty(const F& f, AType ty) {
174  assert(f.is(F::Seq));
175  const auto& seq = f.seq();
176  typename F::Sequence fty;
177  typename F::Sequence other;
178  for(int i = 0; i < seq.size(); ++i) {
179  // In case of nested conjunction.
180  if(seq[i].is(F::Seq) && seq[i].sig() == AND) {
181  auto r = extract_ty(seq[i], ty);
182  auto& fty_ = battery::get<0>(r).seq();
183  auto& other_ = battery::get<1>(r).seq();
184  for(int i = 0; i < fty_.size(); ++i) {
185  fty.push_back(std::move(fty_[i]));
186  }
187  for(int i = 0; i < other_.size(); ++i) {
188  other.push_back(std::move(other_[i]));
189  }
190  }
191  else if(seq[i].type() == ty) {
192  fty.push_back(seq[i]);
193  }
194  else {
195  other.push_back(seq[i]);
196  }
197  }
198  AType other_ty = type_of_conjunction<F>(other);
199  return battery::make_tuple(
200  F::make_nary(AND, std::move(fty), ty),
201  F::make_nary(AND, std::move(other), other_ty));
202 }

207 /** not(f1 \/ ... \/ fn) --> not(f1) /\ ... /\ not(fn)
208  not(f1 /\ ... /\ fn) --> not(f1) \/ ... \/ not(fn) */
209 template <class F>
210 CUDA NI thrust::optional<F> de_morgan_law(Sig sig_neg, const F& f) {
211  auto seq = f.seq();
212  typename F::Sequence neg_seq(seq.size());
213  for(int i = 0; i < f.seq().size(); ++i) {
214  auto neg_i = negate(seq[i]);
215  if(neg_i.has_value()) {
216  neg_seq[i] = *neg_i;
217  }
218  else {
219  return {};
220  }
221  }
222  return F::make_nary(sig_neg, neg_seq, f.type());
223 }

225 template <class F>
226 CUDA NI thrust::optional<F> negate_eq(const F& f) {
227  assert(f.is_binary() && f.sig() == EQ);
228  if(f.seq(0).is(F::B)) {
229  auto b = f.seq(0);
230  b.b() = !b.b();
231  return F::make_binary(b, EQ, f.seq(1), f.type(), f.seq().get_allocator());
232  }
233  else if(f.seq(1).is(F::B)) {
234  auto b = f.seq(1);
235  b.b() = !b.b();
236  return F::make_binary(f.seq(0), EQ, b, f.type(), f.seq().get_allocator());
237  }
238  return F::make_nary(NEQ, f.seq(), f.type());

241 template <class F>
242 CUDA NI thrust::optional<F> negate(const F& f) {
243  if(f.is(F::Seq)) {
244  Sig neg_sig;
245  switch(f.sig()) {
246  case EQ: return negate_eq(f);
247  // Total order predicates can be reversed.
248  case LEQ: neg_sig = GT; break;
249  case GEQ: neg_sig = LT; break;
250  case NEQ: neg_sig = EQ; break;
251  case LT: neg_sig = GEQ; break;
252  case GT: neg_sig = LEQ; break;
253  // Partial order predicates cannot simply be reversed.
254  case SUBSET:
255  case SUBSETEQ:
256  case SUPSET:
257  case SUPSETEQ:
258  return F::make_unary(NOT, f, f.type());
259  case AND:
260  return de_morgan_law(OR, f);
261  case OR:
262  return de_morgan_law(AND, f);
263  default:
264  return {};
265  }
266  return F::make_nary(neg_sig, f.seq(), f.type());
267  }
268  return {};
269 }

294 CUDA NI inline Sig converse_comparison(Sig sig) {
295  switch(sig) {

313 /** Given a predicate of the form `t <op> u` (e.g., `x + y <= z + 4`), it transforms it into an equivalent predicate of the form `s <op> k` where `k` is a constant (e.g., `x + y - (z + 4) <= 0`).
314 If the formula is not a predicate, it is returned unchanged. */
315 template <class F>
316 CUDA NI F move_constants_on_rhs(const F& f) {
317  if(is_comparison(f) && !f.seq(1).is_constant()) {
318  AType aty = f.type();
319  if(f.seq(0).is_constant()) {
320  return F::make_binary(f.seq(1), converse_comparison(f.sig()), f.seq(0), aty);
321  }
322  else {
323  return F::make_binary(
324  F::make_binary(f.seq(0), SUB, f.seq(1), aty),
325  f.sig(),
326  F::make_z(0),
327  aty);
328  }
329  }
330  return f;
331 }
