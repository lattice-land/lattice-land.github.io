19 /** A pre-abstract universe is a lattice (with usual operations join, order, ...) equipped with a simple logical interpretation function and a next/prev functions.
20  We consider totally ordered pre-abstract universes with an upset semantics.
21  For any lattice \f$ L \f$, we consider an element \f$ a \in L \f$ to represent all the concrete elements equal to or above it.
22  This set is called the upset of \f$ a \f$ and is denoted \f$ \mathord{\uparrow}{a} \f$.
23  The concretization function \f$ \gamma \f$ formalizes this idea: \f$ \gamma(a) = \{x \mapsto b \;|\; b \in \mathord{\uparrow}{a} \cap U \} \f$ where \f$ U \f$ is the universe of discourse.
24  The intersection with \f$ U \f$ is necessary to remove potential elements in the abstract universe that are not in the concrete universe of discourse (e.g., \f$ -\infty, \infty \f$ below).
25 
26  The upset semantics associates each element of a lattice to its concrete upset.
27  It is possible to decide that each element is associated to the concrete downset instead.
28  Doing so will reverse our usage of the lattice-theoretic operations (join instead of meet, <= instead of >=, etc.).
29  Instead of considering the upset semantics, it is more convenient to consider the downset semantics of the dual lattice.
30 
31  Example:
32  * The lattice of increasing integer \f$ \mathit{ZInc} = \langle \{-\infty, \ldots, -2, -1, 0, 1, 2, \ldots, \infty\}, \leq \rangle \f$ is ordered by the natural arithmetic comparison operator.
33  Using the upset semantics, we can represent simple constraints such as \f$ x \geq 3 \f$, in which case the upset \f$ \mathord{\uparrow}{3} = \{3, 4, \ldots\} \f$ represents all the values of \f$ x \f$ satisfying the constraints \f$ x \geq 3 \f$, that is, the solutions of the constraints.
34  * By taking the downset semantics of \f$ \mathit{ZInc} \f$, we can represent constraints such as \f$ x \leq 3 \f$.
35  * Alternatively, we can take the dual lattice of decreasing integers \f$ \mathit{ZDec} = \langle \{\infty, \ldots, 2, 1, 0, -1, -2, \ldots, -\infty\}, \geq \rangle \f$.
36  The upset semantics of \f$ \mathit{ZDec} \f$ corresponds to the downset semantics of \f$ \mathit{ZInc} \f$.
37 
38  From a pre-abstract universe, we obtain an abstract universe using the `Universe` class below.
39  We also define various aliases to abstract universes such as `ZInc`, `ZDec`, etc.
40 */

104 /** This function is useful when we need to convert a value to its dual.
105  The dual is the downset of the current element, therefore, if we have \f$ x >= 10 \f$, the dual is given by the formula \f$ x <= 10 \f$ interpreted in the dual lattice.
106  In that case, it just changes the type of the lattice without changing the value.
107  A difference occurs on the bottom and top element.
108  Indeed, by our representation of bot and top, the bottom value in a lattice L equals the top value in its dual, but we need them to remain the same, so the dual of `L::bot()` is `LDual::bot()`.*/
109 template <class LDual, class L>
110 CUDA constexpr LDual dual(const L& x) {
111  if(x.is_bot()) return LDual::bot();
112  if(x.is_top()) return LDual::top();
113  return LDual(x.value());
114 }

286  /** \return \f$ x \geq i \f$ where `x` is a variable's name and `i` the current value.
287  If `U` preserves bottom `true` is returned whenever \f$ a = \bot \f$, if it preserves top `false` is returned whenever \f$ a = \top \f$.
288  We always return an exact approximation, hence for any formula \f$ \llbracket \varphi \rrbracket = a \f$, we must have \f$ a = \llbracket \rrbracket a \llbracket \rrbracket \f$ where \f$ \rrbracket a \llbracket \f$ is the deinterpretation function. */
289  template<class Env>
290  CUDA NI TFormula<typename Env::allocator_type> deinterpret(AVar avar, const Env& env) const {
291  using F = TFormula<typename Env::allocator_type>;
292  if(preserve_top && is_top()) {
293  return F::make_false();
294  }
295  else if(preserve_bot && is_bot()) {
296  return F::make_true();
297  }
298  return F::make_binary(
299  deinterpret<F>(),
300  U::sig_order(),
301  F::make_avar(avar),
302  UNTYPED, env.get_allocator());
303  }

332  /** Interpret a formula of the form `k <sig> x`. */
333  template<class F>
334  CUDA NI static iresult<F> interpret_tell_k_op_x(const F& f, const F& k, Sig sig) {
335  auto r = pre_universe::interpret_tell(k);
336  if(!r.has_value()) {
337  return r;
338  }
339  else if(sig == EQ || sig == U::sig_order()) { // e.g., x <= 4 or x >= 4.24
340  return r;
341  }
342  else if(sig == U::sig_strict_order()) { // e.g., x < 4 or x > 4.24
343  if constexpr(preserve_concrete_covers) {
344  return std::move(r).map(pre_universe::next(r.value()));
345  }
346  else {
347  return r;
348  }
349  }
350  else {
351  return iresult<F>(IError<F>(true, name, "The symbol `" + LVar<typename F::allocator_type>(string_of_sig(sig)) + "` is not supported in the language of this universe.", f));
352  }

355  /** Interpret a formula of the form `k <sig> x`. */
356  template<class F>
357  CUDA NI static iresult<F> interpret_ask_k_op_x(const F& f, const F& k, Sig sig) {
358  auto r = pre_universe::interpret_ask(k);
359  if(!r.has_value()) {
360  return r;
361  }
362  else if(sig == U::sig_order()) {
363  return r;
364  }
365  else if(sig == NEQ || sig == U::sig_strict_order()) {
366  // We could actually do a little bit better in the case of FInc/FDec.
367  // If the real number `k` is approximated by `[f, g]`, it actually means `]f, g[` so we could safely choose `r` since it already under-approximates `k`.
368  return std::move(r).map(pre_universe::next(r.value()));
369  }
370  else {
371  return iresult<F>(IError<F>(true, name, "The symbol `" + LVar<typename F::allocator_type>(string_of_sig(sig)) + "` is not supported in the ask language of this universe.", f));
372  }

375  template<class F>
376  CUDA NI static iresult<F> interpret_tell_set(const F& f, const F& k) {
377  const auto& set = k.s();
378  if(set.size() == 0) {
379  return top();
380  }
381  value_type meet_s = pre_universe::top();
382  constexpr int bound_index = increasing ? 0 : 1;
383  for(int i = 0; i < set.size(); ++i) {
384  auto bound = battery::get<bound_index>(set[i]);
385  auto res = pre_universe::interpret_tell(bound);
386  if(!res.has_value()) {
387  return std::move(iresult<F>(IError<F>(true, name, "Could not interpret an element of a set in the interval's bound.", f))
388  .join_errors(std::move(res)));
389  }
390  meet_s = pre_universe::meet(meet_s, res.value());
391  }
392  return iresult<F>(this_type(meet_s));
393  }

395 public:
396  /** Expects a predicate of the form `x <op> k`, `k <op> x` or `x in k`, where `x` is any variable's name, and `k` a constant.
397  * The symbol <op> is expected to be `U::sig_order()`, `U::sig_strict_order()` and `=`.
398  * Existential formula \f$ \exists{x:T} \f$ can also be interpreted (only to bottom) depending on the underlying pre-universe.
399  */
400  template<class F, class Env>
401  CUDA NI static iresult<F> interpret_tell(const F& f, const Env&) {
402  if(f.is_true()) {
403  if constexpr(preserve_bot) {
404  return bot();
405  }
406  else {
407  return iresult<F>(IError<F>(true, name, "Bottom is not preserved, hence we cannot over-approximate `true`.", f));
408  }
409  }
410  else if(f.is_false()) {
411  return top();
412  }
413  else if(f.is(F::E)) {
414  return pre_universe::interpret_type(f);
415  }
416  else {
417  if(f.is_binary()) {
418  int idx_constant = f.seq(0).is_constant() ? 0 : (f.seq(1).is_constant() ? 1 : 100);
419  int idx_variable = f.seq(0).is_variable() ? 0 : (f.seq(1).is_variable() ? 1 : 100);
420  if(idx_constant + idx_variable != 1) {
421  return iresult<F>(IError<F>(true, name, "Only binary formulas of the form `t1 <sig> t2` where if t1 is a constant and t2 is a variable (or conversely) are supported.", f));
422  }
423  const auto& k = f.seq(idx_constant);
424  if(f.sig() == IN) {
425  if(idx_constant == 0) { // `k in x` is equivalent to `{k} \subseteq x`.
426  return iresult<F>(IError<F>(true, name, "The formula `k in x` is not supported in this abstract universe (`x in k` is supported).", f));
427  }
428  else { // `x in k` is equivalent to `x >= meet k` where `>=` is the lattice order `U::sig_order()`.
429  return interpret_tell_set(f, k);
430  }
431  }
432  else {
433  Sig sig = idx_constant == 1 ? converse_comparison(f.sig()) : f.sig();
434  return interpret_tell_k_op_x(f, k, sig);
435  }
436  }
437  else {
438  return iresult<F>(IError<F>(true, name, "Only binary constraints are supported.", f));
439  }
440  }
441  }
442 
443  /** Expects a predicate of the form `x <op> k` or `k <op> x`, where `x` is any variable's name, and `k` a constant.
444  * The symbol <op> is expected to be `U::sig_order()`, `U::sig_strict_order()` or `!=`.
445  */
446  template<class F, class Env>
447  CUDA NI static iresult<F> interpret_ask(const F& f, const Env&) {
448  if(f.is_true()) {
449  return bot();
450  }
451  else if(f.is_false()) {
452  if constexpr(preserve_top) {
453  return top();
454  }
455  else {
456  return iresult<F>(IError<F>(true, name, "Top is not preserved, hence we cannot under-approximate `false`.", f));
457  }
458  }
459  else {
460  if(f.is_binary()) {
461  int idx_constant = f.seq(0).is_constant() ? 0 : (f.seq(1).is_constant() ? 1 : 100);
462  int idx_variable = f.seq(0).is_variable() ? 0 : (f.seq(1).is_variable() ? 1 : 100);
463  if(idx_constant + idx_variable != 1) {
464  return iresult<F>(IError<F>(true, name, "Only binary formulas of the form `t1 <sig> t2` where if t1 is a constant and t2 is a variable (or conversely) are supported.", f));

489  /** Unary function of type `Sig: FlatUniverse -> PrimitiveUpset`.
490  * \return If `a` is `bot`, we return the bottom element of the upset lattice; and dually for `top`.
491  * Otherwise, we apply the function `Sig` to `a` and return the result.
492  * \remark The result of the function is always over-approximated (or exact when possible).
493  */
494  template <Sig sig, class M>
495  CUDA static constexpr local_type fun(const flat_type<M>& a) {
496  using local_flat_type = flat_type<battery::local_memory>;
497  local_flat_type r1(a);
498  if(r1.is_top()) {
499  return local_type::top();
500  }
501  else if(r1.is_bot()) {
502  return local_type::bot();
503  }
504  return pre_universe::template fun<sig>(r1);
505  }

507  /** Binary functions of type `Sig: FlatUniverse x FlatUniverse -> PrimitiveUpset`.
508  * \return If `a` or `b` is `bot`, we return the bottom element of the upset lattice; and dually for `top`.
509  * Otherwise, we apply the function `Sig` to `a` and `b` and return the result.
510  * \remark The result of the function is always over-approximated (or exact when possible).
511  */
512  template<Sig sig, class M1, class M2>
513  CUDA static constexpr local_type fun(const flat_type<M1>& a, const flat_type<M2>& b) {
514  using local_flat_type = flat_type<battery::local_memory>;
515  local_flat_type r1(a);
516  local_flat_type r2(b);
517  if(r1.is_top() || r2.is_top()) {
518  return local_type::top();
519  }
520  else if(r1.is_bot() || r2.is_bot()) {
521  return local_type::bot();
522  }
523  if constexpr(is_division(sig)) {
524  if(r2.value() == pre_universe::zero()) {
525  return local_type::top();
526  }
527  }
528  return pre_universe::template fun<sig>(r1, r2);
529  }

531  /** Given two values `a` and `b`, we perform the division while taking care of the case where `b == 0`.
532  * When `b != 0`, the result is `fun<sig>(a, b)`.
533  * Otherwise, the result depends on the type of `a` and `b`:
534  * `a` | `b` | local_type | Result
535  * ------ | ------ | ---------- | ------
536  * Inc | Inc | Inc | 0
537  * Inc | Dec | Dec | 0
538  * Dec | Dec | Inc | 0
539  * Dec | Inc | Dec | 0
540  * - | - | - | bot()
541  */
542  template<Sig sig, class Pre1, class Mem1, class Pre2, class Mem2>
543  CUDA static constexpr local_type guarded_div(
544  const PrimitiveUpset<Pre1, Mem1>& a, const PrimitiveUpset<Pre2, Mem2>& b)
545  {
546  using A = PrimitiveUpset<Pre1, Mem1>;
547  using B = PrimitiveUpset<Pre2, Mem2>;
548  using local_flat_type = flat_type<battery::local_memory>;
549  local_flat_type r1(a);
550  local_flat_type r2(b);
551  if (r2 != B::pre_universe::zero())
552  {
553  return fun<sig>(r1, r2);
554  }
555  else {
556  if constexpr(B::preserve_concrete_covers) {
557  // When `b` is "integer-like" we can just skip the value 0 and go to `-1` or `1` depending on the type `B`.
558  return fun<sig>(r1, local_flat_type(B::next(r2.value())));
559  }
560  else if constexpr(
561  (A::increasing && B::increasing == increasing) || // Inc X T -> T where T is either Inc or Dec.
562  (!A::increasing && !B::increasing && increasing) || // Dec X Dec -> Inc
563  (!A::increasing && B::increasing && !increasing)) // Dec X Inc -> Dec
564  {
565  return local_type(pre_universe::zero());
566  }
567  else {
568  return bot();
569  }
570  }
571  }
572 
573  template <Sig sig, class M1, class M2>
574  CUDA static constexpr local_type fun(const this_type2<M1> &a, const this_type2<M2> &b)
575  {
576  static_assert(pre_universe::is_supported_fun(sig), "Function unsupported by the current upset universe.");
577  static_assert(sig == MIN || sig == MAX, "Only MIN and MAX are supported on Upset elements.");
578  using local_flat_type = flat_type<battery::local_memory>;
579  local_flat_type r1(a);
580  local_flat_type r2(b);
581  if (r1.is_top() || r2.is_top())
582  {
583  return local_type::top();
584  }
585  else if (r1.is_bot() || r2.is_bot())
586  {
587  if constexpr((sig == MAX && increasing) || (sig == MIN && !increasing)) {
588  return r1.is_bot() ? b : a;
589  }
590  else {
591  return local_type::bot();
592  }
593  }
594  return pre_universe::template fun<sig>(r1, r2);
595  }
