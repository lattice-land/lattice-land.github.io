59 CUDA NI static iresult<F> interpret(const F& f) {
60 if(f.is(F::Z)) {
61 auto z = f.z();
62 if(z == bot() || z == top()) {
63 return iresult<F>(IError<F>(true, name, "Constant of sort `Int` with the minimal or maximal representable value of the underlying integer type. We use those values to model negative and positive infinities. Example: Suppose we use a byte type, `x >= 256` is interpreted as `x >= INF` which is always false and thus is different from the intended constraint.", f));
64 }
65 return iresult<F>(z);
66 }
67 else if(f.is(F::R)) {
68 if constexpr(is_tell) {
69 return iresult<F>(battery::ru_cast<value_type>(battery::get<0>(f.r())));
70 }
71 else {
72 return iresult<F>(battery::ru_cast<value_type>(battery::get<1>(f.r())));
73 }
74 }
75 else if(f.is(F::B)) {
76 return iresult<F>(f.b() ? one() : zero());
77 }
78 return iresult<F>(IError<F>(true, name, "Only constant of sorts `Int` and `Real` can be interpreted by an integer abstract universe.", f));

82 /** Interpret a constant in the lattice of increasing integers according to the upset semantics (see universe.hpp for explanation).
83 Overflows are not verified (issue #1).
84 Interpretations:
85 * Formulas of kind `F::Z` are interpreted exactly: \f$ [\![ x:\mathbb{Z} \geq k:\mathbb{Z} ]\!] = k \f$.
86 * Formulas of kind `F::R` are over-approximated: \f$ [\![ x:\mathbb{Z} \geq [l..u]:\mathbb{R} ]\!] = \lceil l \rceil \f$.
87 Note that all elements in \f$ [l..\lceil l \rceil[\f$ do not belong to \f$ \mathbb{Z} \f$, so they can be safely ignored.
88 Examples:
89 * \f$ [\![x >= [2.5..2.5]:R ]\!] = 3 \f$.
90 * \f$ [\![x >= [2.9..3.1]:R ]\!] = 3 \f$.
91 */
92 template<class F>
93 CUDA static iresult<F> interpret_tell(const F& f) {
94 return interpret<true>(f);
95 }

103 /** Verify if the type of a variable, introduced by an existential quantifier, is compatible with the current abstract universe.
104 Variables of type `Int` are interpreted exactly (\f$ \mathbb{Z} = \gamma(\bot) \f$). */

159 /** From a lattice perspective, `next: ZInc -> ZInc` returns an element `next(x)` such that `x` is covered by `next(x)`.
160 \param x The element covering the returned element.
161 \return The next value of `x` in the discrete increasing chain `bot, ..., -2, -1, 0, 1, ..., top`.
162 \remark The next value of `bot` is `bot` and the next value of `top` is `top`.
163 \remark There is no element \f$ x \neq \top \f$ such that \f$ \mathit{next}(x) = \top \f$, but it can occur in case of overflow which is not checked. */
164 CUDA static constexpr value_type next(value_type x) {
165 return x + (x != top() && x != bot());

168 /** From a lattice perspective, `prev: ZInc -> ZInc` returns an element `prev(x)` such that `x` covers `prev(x)`.
169 \param x The element covered by the returned element.
170 \return The previous value of `x` in the discrete increasing chain `bot, ..., -2, -1, 0, 1, ..., top`.
171 \remark The previous value of `bot` is `bot` and the previous value of `top` is `top`.
172 \remark There is no element \f$ x \neq \bot \f$ such that \f$ \mathit{prev}(x) = \bot \f$, but it can occur in case of overflow which is not checked. */

206 /** `fun: value_type -> ZInc` is an abstract function on `ZInc` over-approximating the function denoted by `sig` on the concrete domain.
207 * \tparam sig The signature of the function to over-approximate, can be either `NEG` or `ABS`.
208 * \param x The argument of the function, which is a constant value in the underlying universe of discourse.
209 * \note Since `x` is a constant, we do not check for equality with `bot()` or `top()`.
210 */
211 template<Sig sig>
212 CUDA static constexpr value_type fun(value_type x) {
213 static_assert(sig == NEG || sig == ABS, "Unsupported unary function.");
214 switch(sig) {
215 case NEG: return -x;
216 case ABS: return abs(x);
217 default: assert(0); return x;
218 }
219 }

221 /** `fun: value_type X value_type -> ZInc` is similar to its unary version but with an arity of 2. */
222 template<Sig sig>
223 CUDA NI static constexpr value_type fun(value_type x, value_type y) {
224 static_assert(
225 sig == ADD || sig == SUB || sig == MUL || sig == TDIV || sig == TMOD || sig == FDIV || sig == FMOD || sig == CDIV || sig == CMOD || sig == EDIV || sig == EMOD || sig == POW || sig == MIN || sig == MAX || sig == EQ || sig == NEQ || sig == LEQ || sig == GEQ || sig == LT || sig == GT,
226 "Unsupported binary function.");
227 switch(sig) {
228 case ADD: return x + y;
229 case SUB: return x - y;
230 case MUL: return x * y;
231 // Truncated division and modulus, by default in C++.
232 case TDIV: return x / y;
233 case TMOD: return x % y;
234 // Floor division and modulus, see (Leijen D. (2003). Division and Modulus for Computer Scientists).
235 case FDIV: return x / y - (battery::signum(x % y) == -battery::signum(y));
236 case FMOD: return x % y + y * (battery::signum(x % y) == -battery::signum(y));
237 // Ceil division and modulus.
238 case CDIV: return x / y + (battery::signum(x % y) == battery::signum(y));
239 case CMOD: return x % y - y * (battery::signum(x % y) == battery::signum(y));
240 // Euclidean division and modulus, see (Leijen D. (2003). Division and Modulus for Computer Scientists).
241 case EDIV: return x / y - ((x % y >= 0) ? 0 : battery::signum(y));
242 case EMOD: return x % y + y * ((x % y >= 0) ? 0 : battery::signum(y));
243 case POW: return battery::ipow(x, y);
244 case MIN: return battery::min(x, y);
245 case MAX: return battery::max(x, y);
246 case EQ: return x == y;
247 case NEQ: return x != y;
248 case LEQ: return x <= y;
249 case GEQ: return x >= y;
250 case LT: return x < y;
251 case GT: return x >= y;
252 default: assert(0); return x;
253 }
