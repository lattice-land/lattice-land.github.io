13/** This abstract domain works at the level of logical formulas.
14 * It refines the formula by performing a number of simplifications w.r.t. an underlying abstract domain including:
15 * 1. Removing assigned variables.
16 * 2. Removing unused variables.
17 * 3. Removing entailed formulas.
18 * 4. Removing variable equality by tracking equivalence classes.
19 *
20 * The simplified formula can be obtained by calling `deinterpret()`.
21 * Given a solution to the simplified formula, the extended model (with the variables deleted) can be obtained by calling `representative()` to obtain the representative variable of each equivalence class.
22 */
23template<class A, class Allocator>
24class Simplifier {

31 using this_type = Simplifier<sub_type, allocator_type>;

42 AType atype;

44 // We keep a copy of the variable environment in which the formula has been initially interpreted.
45 // This is necessary to project the variables and ask constraints in the subdomain during refinement.
46 VarEnv<allocator_type> env;
47 // Read-only conjunctive formula, where each is treated independently.
48 formula_sequence formulas;
49 // Write-only (accessed in only 1 thread because this is not a parallel lattice entity) conjunctive formula, the main operation is a map between formulas and simplified_formulas.
50 formula_sequence simplified_formulas;
51 // eliminated_variables[i] is `true` when the variable `i` can be removed because it is assigned to a constant.
52 battery::dynamic_bitset<memory_type, allocator_type> eliminated_variables;
53 // eliminated_formulas[i] is `true` when the formula `i` is entailed.
54 battery::dynamic_bitset<memory_type, allocator_type> eliminated_formulas;
55 // `equivalence_classes[i]` contains the index of the representative variable in the equivalence class of the variable `i`.
56 battery::vector<ZDec<int, memory_type>, allocator_type> equivalence_classes;
57 // `constants[i]` contains the universe value of the representative variables `i`, aggregated by join on the values of all variables in the equivalence class.
58 battery::vector<universe_type, allocator_type> constants;

113private:
114 template <class F, class Env>
115 CUDA NI void interpret_in_impl(const F& f, Env& env, iresult<F>& res) const {
116 if(res.is_error()) {
117 return;
118 }
119 if(f.is(F::E)) {
120 auto avar = env.interpret(f.map_atype(aty()));
121 if(avar.has_value()) {
122 res.value().num_vars++;
123 }
124 else {
125 res = std::move(avar).template map_error<tell_type>();
126 }
127 }
128 else if(f.is(F::Seq) && f.sig() == AND) {
129 const auto& seq = f.seq();
130 for(int i = 0; i < seq.size(); ++i) {
131 interpret_in_impl(seq[i], env, res);
132 }
133 }
134 else {
135 res.value().formulas.push_back(f);
136 }
137 }

139public:
140 template <class F, class Env>
141 CUDA NI iresult_tell<F> interpret_tell_in(const F& f, Env& env) const {
142 auto snap = env.snapshot();
143 iresult_tell<F> res{get_allocator()};
144 interpret_in_impl(f, env, res);
145 if(!res.has_value()) {
146 env.restore(snap);
147 }
148 else {
149 res.value().env = env;
150 }
151 return std::move(res);
152 }

154 CUDA this_type& tell(tell_type&& t) {
155 env = std::move(t.env);
156 eliminated_variables.resize(t.num_vars);
157 eliminated_formulas.resize(t.formulas.size());
158 constants.resize(t.num_vars);
159 equivalence_classes.resize(t.num_vars);
160 for(int i = 0; i < equivalence_classes.size(); ++i) {
161 equivalence_classes[i].tell(local::ZDec(i));
162 }
163 formulas = std::move(t.formulas);
164 simplified_formulas.resize(formulas.size());
165 return *this;
166 }

210 // We eliminate the representative of the variable `i` if it is a singleton.
211 template <class Mem>
212 CUDA void vrefine(size_t i, BInc<Mem>& has_changed) {
213 const auto& u = sub->project(to_sub_var(i));
214 size_t j = equivalence_classes[i];
215 constants[j].tell(u, has_changed);
216 if(constants[j].lb().value() == constants[j].ub().value()) {
217 eliminate(eliminated_variables, j, has_changed);
218 }
219 }
220
221 template <class Mem>
222 CUDA void crefine(size_t i, BInc<Mem>& has_changed) {
223 using F = TFormula<allocator_type>;
224 // Eliminate constraint of the form x = y, and add x,y in the same equivalence class.
225 if(is_var_equality(formulas[i])) {
226 AVar x = var_of(formulas[i].seq(0));
227 AVar y = var_of(formulas[i].seq(1));
228 equivalence_classes[x.vid()].tell(local::ZDec(equivalence_classes[y.vid()]), has_changed);
229 equivalence_classes[y.vid()].tell(local::ZDec(equivalence_classes[x.vid()]), has_changed);
230 eliminate(eliminated_formulas, i, has_changed);
231 }
232 else {
233 // Eliminate entailed formulas.
234 auto r = sub->interpret_ask_in(formulas[i], env);
235 if(r.has_value()) {
236 if(sub->ask(r.value())) {
237 eliminate(eliminated_formulas, i, has_changed);
238 return;
239 }
240 }
241 // Replace assigned variables by constants.
242 // Note that since everything is in a fixed point loop, both the constant and the equivalence class might be updated later on.
243 // This is one of the reasons we cannot update `formulas` in-place: we would not be able to update the constant a second time (since the variable would be eliminated).
244 auto f = formulas[i].map([&](const F& f) {
245 if(f.is_variable()) {
246 AVar x = var_of(f);
247 if(eliminated_variables.test(x.vid())) {
248 return constants[x.vid()].template deinterpret<F>();
249 }
250 else if(equivalence_classes[x.vid()] != x.vid()) {
251 return F::make_lvar(UNTYPED, env.name_of(AVar{aty(), equivalence_classes[x.vid()]}));
252 }
253 return f.map_atype(UNTYPED);
254 }
255 return f;
256 });
257 if(f != simplified_formulas[i]) {
258 simplified_formulas[i] = f;
259 has_changed.tell_top();
260 }
261 }
262 }

265 /** We have one refinement operator per variable and one per constraint in the interpreted formula. */
266 CUDA size_t num_refinements() const {
267 return constants.size() + formulas.size();
268 }

270 template <class Mem>
271 CUDA void refine(size_t i, BInc<Mem>& has_changed) {
272 assert(i < num_refinements());
273 if(i < constants.size()) {
274 vrefine(i, has_changed);
275 }
276 else {
277 crefine(i - constants.size(), has_changed);
278 }
279 }

323 for(int i = 0; i < equivalence_classes.size(); ++i) {
324 if(equivalence_classes[i] == i && !eliminated_variables.test(i)) {
325 const auto& x = env[AVar{aty(), i}];
326 seq.push_back(F::make_exists(UNTYPED, x.name, x.sort));
327 auto domain_constraint = constants[i].deinterpret(AVar(aty(), i), env);
328 map_avar_to_lvar(domain_constraint, env, true);
329 seq.push_back(domain_constraint);

333 // Deinterpret the simplified formulas.

335 if(!eliminated_formulas.test(i)) {

337 }
338 }

340 return F::make_true();
341 }
