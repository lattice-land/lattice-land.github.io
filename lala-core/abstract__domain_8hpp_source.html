12 /** An abstract domain is a [lattice](https://en.wikipedia.org/wiki/Lattice_(order)) with additional operations including an interpretation function, a refinement operator and a split operator.
13 We now explain the main idea of this abstract framework, although it is not possible to be very precise here, for more information on the theory, please contact Pierre Talbot (pierre.talbot@uni.lu).
14 We have three entities playing a role in our framework: logical formulas, abstract domains and the concrete domain.
15 The concrete domain is a mathematical object, possibly infinite and not computable, which is not explicitly represented in the code.
16 The concrete domain is however crucial to establish the proofs of soundness and completeness of our abstract domains.
17 We will write the concrete domain \f$ D^\flat \f$ and an abstract domain \f$ A \f$.
18 We can interpret a logical formula \f$ \varphi \f$ in the concrete domain with \f$[\![\varphi]\!]^\flat\f$, and this element contains all the solutions of \f$\varphi\f$.
19 We connect the concrete and abstract worlds using a monotone concretization function \f$\gamma: A \to D^\flat\f$ which turns an abstract element into a concrete element.
20 The concretization is useful to establish two properties.
21 Let \f$ \varphi \f$ be a formula and \f$ a \in A \f$ an abstract element, then:
22  - \f$a\f$ is an under-approximation if \f$\gamma(a) \subseteq [\![\varphi]\!]^\flat \f$ (soundness).
23  - \f$a\f$ is an over-approximation if \f$\gamma(a) \supseteq [\![\varphi]\!]^\flat \f$ (completeness).
24  - \f$a \f$ is an exact representation of \f$\varphi\f$ if it is both an under- and over-approximation.
25 
26 In brief, an under-approximating element guarantees we have only solutions of \f$\varphi\f$ represented in the abstract element \f$a\f$, but not necessarily all.
27 Whereas an over-approximating element guarantees we have all solutions of \f$\varphi\f$ but possibly with extra non-solution elements.
28 
29 This class exists only for documentation purposes as abstract domains will be combined using templates, and not by relying on inheritance from an abstract class.
30 Ideally, `AbstractDomain` should be a C++20 concept, but it is not yet possible because neither NVCC or Doxygen support concepts yet.
31 */
32 template <typename LE, typename VD, typename Alloc>

35  /** `LogicalElement` is an intermediate representation between a logical formula (see `Formula`) and an abstract element.
36  The design rational is that we want to avoid manipulating `Formula` during solving for efficiency purposes (since Formula contains dynamic arrays, string representation of operators, ...).
37  Therefore, we can create the logical elements at the beginning, and adding them in the abstract element later on, when appropriate. */
38  typedef LE LogicalElement;

55  /** A partial interpretation function \f$[\![\varphi]\!]_a^\updownarrow \f$ turns the logical formula \f$\varphi\f$ (`f`) into a logical element according to the current abstract element \f$a\f$ and the approximation kind \f$\updownarrow\f$ (`appx`).
56  See also `LogicalElement`.
57  The approximation kind is not necessarily bound to an abstract element.
58  For instance, let an abstract element \f$a\f$ under-approximating a logical formula \f$\varphi\f$.
59  We can still add over-approximating _redundant constraints_ in \f$a\f$, which will not impact the under-approximating property of \f$a\f$ w.r.t. \f$\varphi\f$.
60  The returned logical elements must be joined later in the current abstract element `this` and not in another abstract element.
61  \return An empty optional if the formula cannot be interpreted in the abstract domain, or if \f$\bot\f$ would be trivially returned in case of over-approximation (dually for \f$ \top \f$ and under-approximation).
62  Otherwise, it returns the interpreted formula. */
63  CUDA virtual thrust::optional<LogicalElement> interpret(Approx appx, const Formula& f) = 0;
64 
65  /** Compute \f$ a \sqcup [\![\varphi]\!] \f$ where \f$a\f$ is the current element and \f$ [\![\varphi]\!] \f$ an interpreted formula. */
66  CUDA virtual this_type& join(const LogicalElement& other) = 0;
67

71  /** `refine` is an extensive function (\f$\forall{a \in A},~\mathit{refine}(a) \geq a \f$) refining an abstract element \f$a\f$.
72  It can have additional properties such as under- or over-approximation depending on the abstract domain.
73  \return `true` if the abstract element has changed and `false` if we reached a fixed point. */
74  CUDA virtual bool refine() = 0;
75 
76  /** The entailment, formally written \f$a \models \varphi\f$, is `true` whenever we can deduce a formula \f$\varphi\f$ from an abstract element \f$a\f$, i.e., \f$\gamma(a) \subseteq [\![\varphi]\!]\f$.
77  Note that if it returns `false`, it can either mean \f$\lnot\varphi\f$ is entailed, or that we do not know yet if it is entailed or not.
78  Therefore, to test for _disentailment_, you should ask if the negation of the formula is entailed. */
79  CUDA virtual bool entailment(const LogicalElement& element) const = 0;
80 
81  /** The projection of term onto the underlying variable domain `VarDom`.
82  A common example is to project the domain of a variable `x` or a term such as `x + y` onto an interval or set variable domain.
83  If you want to project a formula onto a Boolean, you should use `entailment` instead. */
84  CUDA virtual VarDom project(const LogicalElement& x) const = 0;
85 
86  /** The function `embed(x, dom)` is similar to \f$ a \sqcup [\![\varphi]\!] \f$ where \f$\varphi\f$ is a formula with a single variable equals to \f$ x \f$ and interpretable in `VarDom`.
87  Here, the underlying element `VarDom` has already been created. */
88  CUDA virtual void embed(AVar x, const VarDom& dom) = 0;
89 
90  /** `split` is an extensive function, i.e., \f$ \forall{a \in A},~\forall{b \in \mathit{split}(a)},~a \leq b \f$, that divides an abstract element into a set of subelements.
91  We call _unsplittable elements_ the elements such that \f$\mathit{split}(a) \f$ is a singleton.
92  We require \f$\mathit{split}(a) = \{a\} \f$ for all unsplittable elements \f$a \in A \f$.
93  An additional usage of `split` is to detect unsatisfiability of over-approximation, and satisfiability of under-approximation:
94  - In case of an over-approximating element \f$a\f$, we have \f$\mathit{split}(a) = \{\} \Rightarrow \gamma(a) = \{\} \f$.
95  - In case of an under-approximating element \f$a\f$, we have \f$\mathit{split}(a) \neq \{\} \Rightarrow \gamma(a) \neq \{\} \land \gamma(a) \subseteq [\![\varphi]\!]^\flat\f$.
96 
97  For the special case of _eventually under-approximating_ abstract domain, \f$a\f$ is an under-approximation whenever \f$\mathit{split}(a) = \{a\}\f$.
98 
99  \return A list of logical elements (possibly complementary, but not necessarily) that can be joined in an abstract element to further refine its state.
100  */
101  CUDA virtual DArray<LogicalElement, Allocator> split(/*const SearchStrategy& strat*/) const = 0;
102 
103  /** This method resets the current abstract element to an anterior state \f$b \f$.
104  Therefore, this operation is similar to computing \f$ a \sqcap b \f$ where \f$ a \geq b \f$. */
105  CUDA virtual void reset(const this_type& b) = 0;
106 
107  /** \return A copy of the current abstract element. */
108  CUDA virtual this_type* clone() const = 0;
109 
110  /** This function is the inverse of `interpret`, but directly maps to a general `Formula`.
111  Let \f$ a = [\![\varphi]\!]_A \f$, then we must have \f$ \gamma(a) = [\![[\![a]\!]^{-1}]\!]^\flat \f$. */
112  CUDA virtual Formula deinterpret() const = 0;
113 
114  /** Print the current element with the logical name of the variables. */
115  CUDA virtual void print() const = 0;
116 };
