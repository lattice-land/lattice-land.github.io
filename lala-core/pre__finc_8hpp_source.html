13/** `PreFInc` is a pre-abstract universe \f$ \langle \mathbb{F}\setminus\{NaN\}, \leq \rangle \f$ totally ordered by the floating-point arithmetic comparison operator.
14 We work on a subset of floating-point numbers without NaN.
15 It is used to represent (and possibly approximate) constraints of the form \f$ x \geq k \f$ where \f$ k \f$ is a real number.

41 template<bool is_tell, class F>
42 CUDA NI static iresult<F> interpret(const F& f) {
43 if(f.is(F::Z)) {
44 auto z = f.z();
45 // We do not consider the min and max values of integers to be infinities when they are part of the logical formula.
46 if constexpr(is_tell) {
47 return iresult<F>(battery::rd_cast<value_type, decltype(z), false>(z));
48 }
49 else {
50 return iresult<F>(battery::ru_cast<value_type, decltype(z), false>(z));
51 }
52 }
53 else if(f.is(F::R)) {
54 if constexpr(is_tell) {
55 return iresult<F>(battery::rd_cast<value_type>(battery::get<0>(f.r())));
56 }
57 else {
58 return iresult<F>(battery::ru_cast<value_type>(battery::get<1>(f.r())));
59 }
60 }
61 return iresult<F>(IError<F>(true, name, "Only a constant of sort `Int` or `Real` can be interpreted by a floating-point abstract universe.", f));
62 }

65 /** Interpret a constant in the lattice of increasing floating-point numbers `FInc` according to the upset semantics (see universe.hpp for explanation).
66 Interpretations:
67 * Formulas of kind `F::Z` might be over-approximated (if the integer cannot be represented in a floating-point number because it is too large).
68 * Formulas of kind `F::R` might be over-approximated to the lower bound of the interval (if the real number is represented by an interval [lb..ub] where lb != ub).
69 * Other kind of formulas are not supported. */
70 template<class F>
71 CUDA static iresult<F> interpret_tell(const F& f) {
72 return interpret<true>(f);
73 }

81 /** Verify if the type of a variable, introduced by an existential quantifier, is compatible with the current abstract universe.
82 Interpretations:
83 * Variables of type `Int` are always over-approximated (\f$ \mathbb{Z} \subseteq \gamma(\bot) \f$).
84 * Variables of type `Real` are represented exactly (only initially because \f$ \mathbb{R} = \gamma(\bot) \f$). */
85 template<class F>
86 CUDA NI static iresult<F> interpret_type(const F& f) {
87 assert(f.is(F::E));
88 const auto& vname = battery::get<0>(f.exists());
89 const auto& cty = battery::get<1>(f.exists());
90 if(cty.is_int()) {
91 return iresult<F>(bot(), IError<F>(false, name, "Variable `" + vname + "` of sort `Int` is over-approximated in a floating-point abstract universe.", f));
92 }
93 else if(cty.is_real()) {
94 return iresult<F>(bot());
95 }
96 else {
97 return iresult<F>(IError<F>(true, name, "Variable `" + vname + "` can only be of sort `Real`, or be over-approximated if the sort is `Bool` or `Int`.", f));
98 }
99 }

143 /** From a lattice perspective, this function returns an element \f$ y \f$ such that \f$ y \f$ is a cover of \f$ x \f$.
144
145 \return The next value of \f$ x \f$ in the floating-point increasing chain \f$ -\infty, \ldots, prev(-2.0), -2.0, next(-2.0), \ldots, \infty \f$ is the next representable value of \f$ x \f$ when \f$ x \not\in \{\infty, -\infty\} \f$ and \f$ x \f$ otherwise.
146 Note that \f$ 0 \f$ is considered to be represented by a single value. */
147 CUDA static constexpr value_type next(value_type x) {
148 if(x == bot() || x == top()) {
149 return x;
150 }
151 if(x == value_type{-0.0}) {
152 return battery::nextafter(value_type{+0.0}, top());
153 }
154 return battery::nextafter(x, top());
155 }

192 template<Sig sig>
193 CUDA static constexpr value_type fun(value_type x) {
194 static_assert(is_supported_fun(sig), "Unsupported unary function.");
195 // Negation and absolute functions are exact in floating-point arithmetic.
196 switch(sig) {
197 case NEG: return -x;
198 case ABS: return abs(x);
199 default: assert(0); return x;
200 }
201 }

203 template<Sig sig>
204 CUDA NI static constexpr value_type fun(value_type x, value_type y) {
205 static_assert(is_supported_fun(sig), "Unsupported binary function.");
206 switch(sig) {
207 case ADD: return battery::add_down(x, y);
208 case SUB: return battery::sub_down(x, y);
209 case MUL: return battery::mul_down(x, y);
210 case DIV: return battery::div_down(x, y);
211 case MIN: return battery::min(x, y);
212 case MAX: return battery::max(x, y);
213 case EQ: return x == y;
214 case NEQ: return x != y;
215 case LEQ: return x <= y;
216 case GEQ: return x >= y;
217 case LT: return x < y;
218 case GT: return x >= y;
219 default: assert(0); return x;
220 }
