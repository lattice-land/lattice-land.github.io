100  template <class F>
101  CUDA NI iresult<F> interpret_existential(const F& f) {
102  const auto& vname = battery::get<0>(f.exists());
103  if(f.type() == UNTYPED) {
104  return iresult<F>(IError<F>(true, name, "Untyped abstract type: variable `" + vname + "` has no abstract type.", f));
105  }
106  if(contains(vname)) {

108  }
109  else {

116  template <class F>
117  CUDA NI iresult<F> interpret_lv(const F& f) {
118  const auto& vname = f.lv();
119  auto var = variable_of(vname);
120  if(var.has_value()) {
121  if(f.type() != UNTYPED) {
122  auto avar = var->avar_of(f.type());
123  if(avar.has_value()) {
124  return AVar(*avar);
125  }
126  else {
127  return iresult<F>(IError<F>(true, name, "Variable `" + vname + "` has not been declared in the abstract domain `" + fstring<F>::from_int(f.type()) + "`.", f));
128  }
129  }
130  else {
131  if(var->avars.size() == 1) {
132  return AVar(var->avars[0]);
133  }
134  else {
135  return iresult<F>(IError<F>(true, name, "Variable occurrence `" + vname + "` is untyped, but exists in multiple abstract domains.", f));
136  }
137  }
138  }
139  else {
140  return iresult<F>(IError<F>(true, name, "Undeclared variable `" + vname + "`.", f));
141  }

182  /** A variable environment can interpret formulas of two forms:
183  * - Existential formula with a valid abstract type (`f.type() != UNTYPED`).
184  * - Variable occurrence.
185  * It returns an abstract variable (`AVar`) corresponding to the variable created (existential) or already presents (occurrence). */
186  template <class F>
187  CUDA NI iresult<F> interpret(const F& f) {
188  if(f.is(F::E)) {
189  return interpret_existential(f);
190  }
191  else if(f.is(F::LV)) {
192  return interpret_lv(f);
193  }
194  else if(f.is(F::V)) {
195  if(contains(f.v())) {
196  return f.v();
197  }
198  else {
199  return iresult<F>(IError<F>(true, name, "Undeclared abstract variable `" + fstring<F>::from_int(f.v().aty()) + ", " + fstring<F>::from_int(f.v().vid()) + "`.", f));
200  }
201  }
202  else {
203  return iresult<F>(IError<F>(true, name, "Unsupported formula: `VarEnv` can only interpret quantifiers and occurrences of variables.", f));
204  }
205  }

270  /** Restore the environment to its previous state `snap`. */
271  CUDA NI void restore(const snapshot_type& snap) {
272  assert(lvars.size() >= snap.lvars_snap.size());
273  assert(avar2lvar.size() >= snap.avar2lvar_snap.size());
274  while(lvars.size() > snap.lvars_snap.size()) {
275  lvars.pop_back();

289 /** Given a formula `f` and an environment, return the first variable occurring in `f` or `{}` if `f` has no variable in `env`. */
290 template <class F, class Env>
291 CUDA NI thrust::optional<const typename Env::variable_type&> var_in(const F& f, const Env& env) {
292  const auto& g = var_in(f);
293  switch(g.index()) {
294  case F::V:
295  if(g.v().is_untyped()) { return {}; }
296  else { return env[g.v()]; }
297  case F::E:
298  return env.variable_of(battery::get<0>(g.exists()));
299  case F::LV:
300  return env.variable_of(g.lv());
301  }
302  return {};
303 }
