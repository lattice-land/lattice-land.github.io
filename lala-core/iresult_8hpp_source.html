16/** The representation of an error (or warning) obtained when interpreting a formula in an abstract universe or domain. */
17template<class F>
18class IError {
19public:
20 using allocator_type = typename F::allocator_type;
21 using this_type = IError<F>;
22
23private:
24 battery::string<allocator_type> ad_name;
25 battery::string<allocator_type> description;
26 F uninterpretable_formula;
27 AType aty;
28 battery::vector<IError<F>, allocator_type> suberrors;
29 bool fatal;
30
31 CUDA void print_indent(int indent) const {
32 for(int i = 0; i < indent; ++i) {
33 printf(" ");
34 }
35 }
36
37 CUDA void print_line(const char* line, int indent) const {
38 print_indent(indent);
39 printf("%s", line);
40 }
41
42public:
43 // If fatal is false, it is considered as a warning.
44 CUDA NI IError(bool fatal, battery::string<allocator_type> ad_name,
45 battery::string<allocator_type> description,
46 F uninterpretable_formula,
47 AType aty = UNTYPED)
48 : ad_name(std::move(ad_name)),
49 description(std::move(description)),
50 uninterpretable_formula(std::move(uninterpretable_formula)),
51 aty(aty),
52 fatal(fatal)
53 {}
54
55 CUDA NI this_type& add_suberror(IError<F>&& suberror) {
56 suberrors.push_back(suberror);
57 return *this;
58 }
59
60 CUDA NI void print(int indent = 0) const {
61 if(fatal) {
62 print_line("[error] ", indent);
63 }
64 else {
65 print_line("[warning] ", indent);
66 }
67 printf("Uninterpretable formula.\n");
68 print_indent(indent);
69 printf(" Abstract domain: %s\n", ad_name.data());
70 print_line(" Abstract type: ", indent);
71 if(aty == UNTYPED) {
72 printf("untyped\n");
73 }
74 else {
75 printf("%d\n", aty);
76 }
77 print_line(" Formula: ", indent);
78 uninterpretable_formula.print(true);
79 printf("\n");
80 print_indent(indent);
81 printf(" Description: %s\n", description.data());
82 for(int i = 0; i < suberrors.size(); ++i) {
83 suberrors[i].print(indent + 2);
84 printf("\n");
85 }
86 }

88 CUDA bool is_fatal() const { return fatal; }
89};
