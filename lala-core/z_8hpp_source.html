56  /** Expects a predicate of the form `x <op> i` where `x` is any variable's name, and `i` an integer.
57  - If `appx` is EXACT: `op` can be >= or >.
58  - If `appx` is UNDER: `op` can be >=, > or !=.
59  - If `appx` is OVER: `op` can be >=, > or ==.
60  An empty optional is also returned if the element would overflow in `int_type`.
61  */
62  CUDA thrust::optional<LogicalElement> interpret(Approx appx, const Formula& f) {
63  typedef Formula F;
64  if(f.tag == F::TRUE) {
65  return bot();
66  }
67  else if(f.tag == F::FALSE) {
68  return top();
69  }
70  if(SHAPE(f, F::GEQ, F::AVAR, F::LONG)) { // x >= 4
71  return ZInc(f.children[1].i);
72  }
73  else if(SHAPE(f, F::GT, F::AVAR, F::LONG)) { // x > 4
74  return ZInc(f.children[1].i + 1);
75  }
76  // Under-approximation of `x != 4` as `5`.
77  else if(SHAPE(f, F::NEQ, F::AVAR, F::LONG) && appx == UNDER) {
78  return ZInc(f.children[1].i + 1);
79  }
80  // Over-approximation of `x == 4` as `4`.
81  else if(SHAPE(f, F::EQ, F::AVAR, F::LONG) && appx == OVER) {
82  return ZInc(f.children[1].i);
83  }
84  return {};
85  }
