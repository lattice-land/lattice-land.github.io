51  /** Expects a predicate of the form `x <op> i` where `x` is any variable's name, and `i` an integer.
52  - If `appx` is EXACT: `op` can be >= or >.
53  - If `appx` is UNDER: `op` can be >=, > or !=.
54  - If `appx` is OVER: `op` can be >=, > or ==.
55  An empty optional is also returned if the element would overflow in `int_type`.
56  */
57  CUDA thrust::optional<LogicalElement> interpret(Approx appx, const Formula& f) {
58  typedef Formula F;
59  if(f.tag == F::TRUE) {
60  return bot();
61  }
62  else if(f.tag == F::FALSE) {
63  return top();
64  }
65  if(SHAPE(f, F::GEQ, F::AVAR, F::LONG)) { // x >= 4
66  return z_inc(f.children[1].i);
67  }
68  else if(SHAPE(f, F::GT, F::AVAR, F::LONG)) { // x > 4
69  return z_inc(f.children[1].i + 1);
70  }
71  // Under-approximation of `x != 4` as `5`.
72  else if(SHAPE(f, F::NEQ, F::AVAR, F::LONG) && appx == UNDER) {
73  return z_inc(f.children[1].i + 1);
74  }
75  // Over-approximation of `x == 4` as `4`.
76  else if(SHAPE(f, F::EQ, F::AVAR, F::LONG) && appx == OVER) {
77  return z_inc(f.children[1].i);
78  }
79  return {};
80  }
