201  /** \return \f$ x = k \f$ where `x` is a variable's name and `k` the current value.
202  `true` is returned whenever \f$ a = \bot \f$, and `false` is returned whenever \f$ a = \top \f$.
203  We always return an exact approximation, hence for any formula \f$ \llbracket \varphi \rrbracket = a \f$, we must have \f$ a = \llbracket \rrbracket a \llbracket \rrbracket \f$ where \f$ \rrbracket a \llbracket \f$ is the deinterpretation function. */
204  template<class Env>
205  CUDA NI TFormula<typename Env::allocator_type> deinterpret(AVar avar, const Env& env) const {
206  using F = TFormula<typename Env::allocator_type>;
207  if(is_top()) {
208  return F::make_false();
209  }
210  else if(is_bot()) {
211  return F::make_true();
212  }
213  return F::make_binary(
214  F::make_avar(avar),
215  EQ,
216  deinterpret<F>(),
217  avar.aty(), env.get_allocator());
218  }

246 public:
247  /** Expects a predicate of the form `x = k` or `k = x`, where `x` is any variable's name, and `k` a constant.
248  Existential formula \f$ \exists{x:T} \f$ can also be interpreted (only to bottom). */
249  template<class F, class Env>
250  CUDA NI static iresult<F> interpret_tell(const F& f, const Env& env) {
251  if(f.is_true()) {
252  return bot();
253  }
254  else if(f.is_false()) {
255  return top();
256  }
257  else if(f.is(F::E)) {
258  return pre_universe::interpret_type(f);
259  }
260  else {
261  if(f.is_binary() && f.sig() == EQ) {
262  int idx_constant = f.seq(0).is_constant() ? 0 : (f.seq(1).is_constant() ? 1 : 100);
263  int idx_variable = f.seq(0).is_variable() ? 0 : (f.seq(1).is_variable() ? 1 : 100);
264  if(idx_constant + idx_variable == 1) {
265  const auto& k = f.seq(idx_constant);
266  const auto& x = f.seq(idx_variable);
267  auto t = pre_universe::interpret_tell(k);
268  auto a = pre_universe::interpret_ask(k);
269  if(t.has_value()) {
270  if(a.has_value()) {
271  if(a.value() == t.value()) {
272  return t;
273  }
274  else {
275  return std::move(iresult<F>(IError<F>(true, name, "The constant has no exact interpretation which is required in this abstract universe.", f))
276  .join_errors(std::move(t))
277  .join_errors(std::move(a)));
278  }
279  }
280  else {
281  return a;
282  }
283  }
284  else {
285  return t;
286  }
287  }
288  }
289  return iresult<F>(IError<F>(true, name,
290  "Only interpretations of existential quantifier and binary formulas of the form `t1 = t2` where if t1 is a constant and t2 is a variable (or conversely) are supported.", f));
291  }
292  }

294  /** Same as `interpret_tell` without the support for existential quantifier. */

323  /** Binary functions over `value_type`. */
324  template<Sig sig, class M1, class M2>
325  CUDA static constexpr local_type fun(const this_type2<M1>& l, const this_type2<M2>& k) {
326  static_assert(is_supported_fun(sig));
327  auto a = l.value();
328  auto b = k.value();
329  if(U::top() == a || U::top() == b) {
330  return local_type::top();
331  }
332  else if(U::bot() == a || U::bot() == b) {
333  return local_type::bot();
334  }
335  if constexpr(is_division(sig) && is_arithmetic) {
336  if(b == pre_universe::zero()) {
337  return local_type::top();
338  }
339  }
340  return pre_universe::template fun<sig>(a, b);
341  }
