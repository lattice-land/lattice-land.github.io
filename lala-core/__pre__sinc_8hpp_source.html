41  /** Interpret a constant in the lattice of increasing sets of integers according to the upset semantics (see universe.hpp for explanation).
42  Only formulas of kind `F::S(F::Z)` are supported: \f$ [\![ x:\mathcal{P}(\mathbb{Z}) \supseteq \{k_1,\ldots,k_n\}:\mathcal{P}(\mathbb{Z}) ]\!] = \{k_1,\ldots,k_n\} \f$.
43  */
44  template<class F>
45  CUDA static iresult<F> interpret_tell(const F& f) {
46  using bstring = battery::string<typename F::allocator_type>;
47  if(f.is(F::S)) {
48  if(f.s().empty()) {
49  return iresult<F>(bot());
50  }
51  else if(f.sort() == Sort(Sort::Set, Sort(Sort::Int))) {
52  const auto& s = f.s();
53  value_type bitset = bot();
54  for(int i = 0; i < s.size(); ++i) {
55  for(int x = battery::get<0>(s[i]); x <= battery::get<1>(s[i]); ++x) {
56  if(x >= bitset.size()) {
57  return iresult<F>(IError<F>(true, name,
58  "The element `" + bstring::from_int(x)
59  + "` cannot be represented in the current finite set abstract universe, which has a maximum size of `"
60  + bstring::from_int(bitset.size()) + "`.", f));
61  }
62  bitset.set(x);
63  }
64  }
65  return iresult<F>(s);
66  }
67  }
68  return iresult<F>(IError<F>(true, name, "Only constant of sort `Set(Int)` can be interpreted by a finite set abstract universe.", f));
69  }

125  /** `fun: value_type -> SInc` is an abstract function on `SInc` over-approximating the function denoted by `sig` on the concrete domain.
126  * \tparam sig The signature of the function to over-approximate, can be either `CARD` or `HULL`.
127  * \param x The argument of the function, which is a constant value in the underlying universe of discourse.
128  * \note Since `x` is a constant, we do not check for equality with `bot()` or `top()`.
129  */
130  template<Sig sig>
131  CUDA static constexpr value_type fun(value_type x) {
132  static_assert(sig == NEG || sig == ABS, "Unsupported unary function.");
133  switch(sig) {
134  case NEG: return -x;
135  case ABS: return abs(x);
136  default: assert(0); return x;
137  }
138  }
139 
140  /** `fun: value_type X value_type -> ZInc` is similar to its unary version but with an arity of 2. */
141  template<Sig sig>
142  CUDA static constexpr value_type fun(value_type x, value_type y) {
143  static_assert(
144  sig == ADD || sig == SUB || sig == MUL || sig == TDIV || sig == TMOD || sig == FDIV || sig == FMOD || sig == CDIV || sig == CMOD || sig == EDIV || sig == EMOD || sig == POW || sig == MIN || sig == MAX || sig == EQ || sig == NEQ || sig == LEQ || sig == GEQ || sig == LT || sig == GT,
145  "Unsupported binary function.");
146  switch(sig) {
147  case ADD: return x + y;
148  case SUB: return x - y;
149  case MUL: return x * y;
150  // Truncated division and modulus, by default in C++.
151  case TDIV: return x / y;
152  case TMOD: return x % y;
153  // Floor division and modulus, see (Leijen D. (2003). Division and Modulus for Computer Scientists).
154  case FDIV: return x / y - (battery::signum(x % y) == -battery::signum(y));
155  case FMOD: return x % y + y * (battery::signum(x % y) == -battery::signum(y));
156  // Ceil division and modulus.
157  case CDIV: return x / y + (battery::signum(x % y) == battery::signum(y));
158  case CMOD: return x % y - y * (battery::signum(x % y) == battery::signum(y));
159  // Euclidean division and modulus, see (Leijen D. (2003). Division and Modulus for Computer Scientists).
160  case EDIV: return x / y - ((x % y >= 0) ? 0 : battery::signum(y));
161  case EMOD: return x % y + y * ((x % y >= 0) ? 0 : battery::signum(y));
162  case POW: return battery::ipow(x, y);
163  case MIN: return battery::min(x, y);
164  case MAX: return battery::max(x, y);
165  case EQ: return x == y;
166  case NEQ: return x != y;
167  case LEQ: return x <= y;
168  case GEQ: return x >= y;
169  case LT: return x < y;
170  case GT: return x >= y;
171  default: assert(0); return x;
172  }
173  }
