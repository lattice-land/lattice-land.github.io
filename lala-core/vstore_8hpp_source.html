20/** The variable store abstract domain is a _domain transformer_ built on top of an abstract universe `U`.
21Concretization function: \f$ \gamma(\rho) \sqcap_{x \in \pi(\rho)} \gamma_{U_x}(\rho(x)) \f$.
22The top element is smashed and the equality between two stores is represented by the following equivalence relation, for two stores \f$ S \f$ and \f$ T \f$:
23\f$ S \equiv T \Leftrightarrow \forall{x \in \mathit{Vars}},~S(x) = T(x) \lor \exists{x \in \mathit{dom}(S)},\exists{y \in \mathit{dom}(T)},~S(x) = \top \land T(y) = \top \f$.
24Intuitively, it means that either all elements are equal or both stores have a top element, in which case they "collapse" to the top element, and are considered equal.
25
26The bottom element is the element \f$ \langle \bot, \ldots \rangle \f$, that is an infinite number of variables initialized to bottom.
27In practice, we cannot represent infinite collections, so we represent bottom either as the empty collection or with a finite number of bottom elements.
28Any finite store \f$ \langle x_1, \ldots, x_n \rangle \f$ should be seen as the concrete store \f$ \langle x_1, \ldots, x_n, \bot, \ldots \rangle \f$.
29
30This semantics has implication when joining or merging two elements.
31For instance, \f$ \langle 1 \rangle.\mathit{dtell}(\langle \bot, 4 \rangle) \f$ will be equal to bottom, in that case represented by \f$ \langle \bot \rangle \f$.
32
33Limitation:
34 - The size of the variable store is fixed at initialization and cannot be modified afterwards.
35 The reason is that resizing an array does not easily fit the PCCP model of computation.
36 - You can only tell store that are smaller or equal in size (unless the extra elements are set to bottom).
37
38Template parameters:
39 - `U` is the type of the abstract universe.
40 - `Allocator` is the allocator of the underlying array of universes. */

179 template <class F, class Env>
180 CUDA NI iresult<F, Env> interpret_existential(const F& f, Env& env) const {
181 using TellType = tell_type<typename Env::allocator_type>;
182 assert(f.is(F::E));
183 auto u = local_universe::interpret_tell(f, env);
184 if(u.has_value()) {
185 if(env.num_vars_in(atype) >= vars()) {
186 return iresult<F, Env>(IError<F>(true, name, "The variable could not be interpreted because the store is full.", f));
187 }
188 auto avar = env.interpret(f.map_atype(atype));
189 if(avar.has_value()) {
190 if(u.value().is_bot()) {
191 return std::move(iresult<F, Env>(TellType(env.get_allocator())).join_warnings(std::move(u)));
192 }
193 else {
194 TellType res(env.get_allocator());
195 res.push_back(var_dom(avar.value().vid(), u.value()));
196 return std::move(iresult<F, Env>(std::move(res)).join_warnings(std::move(u)));
197 }
198 }
199 else {
200 return std::move(avar).template map_error<TellType>();
201 }
202 }
203 else {
204 return std::move(u).template map_error<TellType>();
205 }
206 }

208 /** Interpret a predicate without variables. */
209 template <class F, class Env>
210 CUDA NI iresult<F, Env> interpret_zero_predicate(const F& f, const Env& env) const {
211 using TellType = tell_type<typename Env::allocator_type>;
212 if(f.is_true()) {

225 /** Interpret a predicate with a single variable occurrence. */
226 template <bool is_tell, class F, class Env>
227 CUDA NI iresult<F, Env> interpret_unary_predicate(const F& f, const Env& env) const {
228 using TellType = tell_type<typename Env::allocator_type>;
229 auto u = is_tell ? local_universe::interpret_tell(f, env) : local_universe::interpret_ask(f, env);
230 if(u.has_value()) {
231 TellType res(env.get_allocator());
232 const auto& varf = var_in(f);
233 // When it is not necessary, we try to avoid using the environment.
234 // This is for instance useful when refinement operators add new constraints but do not have access to the environment (e.g., split()), and to avoid passing the environment around everywhere.
235 if(varf.is(F::V)) {
236 res.push_back(var_dom(varf.v().vid(), u.value()));
237 }
238 else {
239 auto var = var_in(f, env);
240 if(!var.has_value()) {
241 return std::move(iresult<F, Env>(IError<F>(true, name, "Undeclared variable.", f)).join_warnings(std::move(u)));
242 }
243 if(!u.value().is_bot()) {
244 auto avar = var->avar_of(atype);
245 if(!avar.has_value()) {
246 return std::move(iresult<F, Env>(IError<F>(true, name,
247 "The variable was not declared in the current abstract element (but exists in other abstract elements).", f))
248 .join_warnings(std::move(u)));
249 }
250 res.push_back(var_dom(avar->vid(), u.value()));
251 }
252 }
253 return std::move(iresult<F, Env>(std::move(res)).join_warnings(std::move(u)));
254 }
255 else {
256 return std::move(iresult<F, Env>(IError<F>(true, name, "Could not interpret a unary predicate in the underlying abstract universe.", f)).join_errors(std::move(u)));
257 }
258 }

260 template <bool is_tell, class F, class Env>
261 CUDA NI iresult<F, Env> interpret_predicate(const F& f, Env& env) const {
262 if(f.type() != UNTYPED && f.type() != aty()) {
263 return iresult<F, Env>(IError<F>(true, name, "The predicate is not of the right type.", f));
264 }
265 if constexpr(is_tell) {
266 if(f.is(F::E)) {
267 return interpret_existential(f, env);
268 }
269 }
270 if(f.is(F::Seq) && f.sig() == AND) {
271 return interpret_in_impl<is_tell>(f, env);
272 }
273 else {
274 switch(num_vars(f)) {
275 case 0: return interpret_zero_predicate(f, env);
276 case 1: return interpret_unary_predicate<is_tell>(f, env);
277 default: return iresult<F, Env>(IError<F>(true, name, "Interpretation of n-ary predicate is not supported in VStore.", f));
278 }
279 }
280 }

282 template <bool is_tell, class F, class Env>
283 CUDA NI iresult<F, Env> interpret_in_impl(const F& f, Env& env) const {
284 using TellType = tell_type<typename Env::allocator_type>;
285 if(f.is_untyped() || f.type() == aty()) {
286 if(f.is(F::Seq) && f.sig() == AND) {
287 const typename F::Sequence& seq = f.seq();
288 auto res = iresult<F, Env>(TellType(env.get_allocator()));
289 for(int i = 0; i < seq.size(); ++i) {
290 auto r = interpret_predicate<is_tell>(seq[i], env);
291 if(r.has_value()) {
292 for(int j = 0; j < r.value().size(); ++j) {
293 res.value().push_back(r.value()[j]);
294 }
295 res.join_warnings(std::move(r));
296 }
297 else {
298 return std::move(iresult<F, Env>(IError<F>(true, name, "Could not interpret a component of the conjunction", f))
299 .join_errors(std::move(r))
300 .join_warnings(std::move(res)));
301 }
302 }
303 return res;
304 }
305 else {
306 return interpret_predicate<is_tell>(f, env);
307 }
308 }
309 else {
310 return iresult<F, Env>(IError<F>(true, name, "Interpretation of a formula with a different type.", f));
311 }
312 }

315 /** The store of variables lattice expects a conjunctive formula \f$ c_1 \land \ldots \land c_n \f$ in which all components \f$ c_i \f$ are formulas with a single variable (including existential quantifiers) that can be handled by the abstract universe `U`.
316 * If one component of the conjunction cannot be interpreted by the abstract universe, the whole formula is considered as uninterpretable.
317 *
318 * Variables must be existentially quantified before a formula containing variables can be interpreted.
319 * Variables are immediately assigned to an index of `VStore` and initialized to \f$ \bot_U \f$, if the universe's interpretation is different from bottom, the result of the VStore interpretation need to be `tell` later on in the store.
320 * Shadowing/redeclaration of variables with existential quantifier is not supported.
321 * Variable mappings are added to the environment only if `interpret(f).has_value()`.
322
323 * There is a small quirk: different stores might be produced if quantifiers do not appear in the same order.
324 * This is because we attribute the first available index to variables when interpreting the quantifier.
325 * The store will only be equivalent when considering the `env` structure.
326 */
327 template <class F, class Env>
328 CUDA NI iresult<F, Env> interpret_tell_in(const F& f, Env& env) const {
329 auto snap = env.snapshot();
330 auto r = interpret_in_impl<true>(f, env);
331 if(!r.has_value()) {
332 env.restore(snap);
333 }
334 return std::move(r);
335 }
336
337 /** The static version of interpret creates a store with exactly the number of existentially quantified variables occurring in `f`.
338 * All the existentially quantified variables must be untyped.
339 * The UID of the store will be an UID that is free in `env`. */
340 template <class F, class Env>
341 CUDA NI static IResult<this_type, F> interpret_tell(const F& f, Env& env, allocator_type alloc = allocator_type()) {
342 auto snap = env.snapshot();
343 size_t ty = env.extends_abstract_dom();
344 this_type store(ty,
345 num_quantified_vars(f, UNTYPED) + num_quantified_vars(f, ty),
346 alloc);
347 auto r = store.interpret_tell_in(f, env);
348 if(r.has_value()) {
349 store.tell(r.value());
350 return std::move(r).map(std::move(store));
351 }
352 else {
353 env.restore(snap);
354 return std::move(r).template map_error<this_type>();
355 }
356 }

459 template <class U2, class Alloc2, class Mem>
460 CUDA this_type& dtell(const VStore<U2, Alloc2>& other, BInc<Mem>& has_changed) {
461 if(other.is_top()) {
462 return *this;
463 }
464 int min_size = battery::min(vars(), other.vars());
465 is_at_top.dtell(other.is_at_top, has_changed);
466 for(int i = 0; i < min_size; ++i) {
467 data[i].dtell(other[i], has_changed);
468 }

475 return *this;

497 /** An abstract element is extractable when it is not equal to top.
498 * If the strategy is `atoms`, we check the domains are singleton.
499 */
500 template<class ExtractionStrategy = NonAtomicExtraction>
501 CUDA bool is_extractable(const ExtractionStrategy& strategy = ExtractionStrategy()) const {
502 if(is_top()) {
503 return false;
504 }
505 if constexpr(ExtractionStrategy::atoms) {
506 for(int i = 0; i < data.size(); ++i) {
507 if(data[i].lb() < dual<typename universe_type::LB>(data[i].ub())) {
508 return false;
509 }
510 }
511 }
512 return true;

526 template<class Env>
527 CUDA NI TFormula<typename Env::allocator_type> deinterpret(const Env& env) const {
528 using F = TFormula<typename Env::allocator_type>;
529 typename F::Sequence seq{env.get_allocator()};
530 for(int i = 0; i < data.size(); ++i) {
531 AVar v(aty(), i);
532 seq.push_back(F::make_exists(aty(), env.name_of(v), env.sort_of(v)));
533 auto f = data[i].deinterpret(v, env);
534 f.type_as(aty());
535 map_avar_to_lvar(f, env);
536 seq.push_back(std::move(f));
537 }
538 return F::make_nary(AND, std::move(seq), aty());
539 }

542// Lattice operations.
543// Note that we do not consider the logical names.
544// These operations are only considering the indices of the elements.

546template<class L, class K, class Alloc>
547CUDA auto join(const VStore<L, Alloc>& a, const VStore<K, Alloc>& b)
548{
549 using U = decltype(join(a[0], b[0]));
550 if(a.is_top() || b.is_top()) {
551 return VStore<U, Alloc>::top(UNTYPED, a.get_allocator());
552 }
553 int max_size = battery::max(a.vars(), b.vars());
554 int min_size = battery::min(a.vars(), b.vars());
555 VStore<U, Alloc> res(UNTYPED, max_size, a.get_allocator());
556 for(int i = 0; i < min_size; ++i) {
557 res.tell(i, join(a[i], b[i]));
558 }
559 for(int i = min_size; i < a.vars(); ++i) {
560 res.tell(i, a[i]);
561 }
562 for(int i = min_size; i < b.vars(); ++i) {
563 res.tell(i, b[i]);
564 }
565 return res;
566}

568template<class L, class K, class Alloc>
569CUDA auto meet(const VStore<L, Alloc>& a, const VStore<K, Alloc>& b)
570{
571 using U = decltype(meet(a[0], b[0]));
572 if(a.is_top()) {
573 if(b.is_top()) {
574 return VStore<U, Alloc>::top(UNTYPED, a.get_allocator());
575 }
576 else {
577 return VStore<U, Alloc>(b);
578 }
579 }
580 else if(b.is_top()) {
581 return VStore<U, Alloc>(a);
582 }
583 else {
584 int min_size = battery::min(a.vars(), b.vars());
585 VStore<U, Alloc> res(UNTYPED, min_size, a.get_allocator());
586 for(int i = 0; i < min_size; ++i) {
587 res.tell(i, meet(a[i], b[i]));
588 }
589 return res;
590 }
