26 /** An interval is a Cartesian product of a lower and upper bounds, themselves represented as lattices.
27  One difference, is that the \f$ \top \f$ can be represented by multiple interval elements, whenever \f$ l > u \f$, therefore some operations are different than on the Cartesian product, e.g., \f$ [3..2] \equiv [4..1] \f$ in the interval lattice. */

118  template <bool is_tell, class F, class Env>
119  CUDA static iresult<F> forward_to_cp(const F& f, const Env& env) {
120  auto cp_res = is_tell ? CP::interpret_tell(f, env) : CP::interpret_ask(f, env);
121  if(cp_res.has_value()) {
122  local_type itv(cp_res.value());
123  return std::move(iresult<F>(std::move(itv)).join_warnings(std::move(cp_res)));
124  }
125  return std::move(cp_res).template map_error<local_type>();
126  }

128 public:
129  /** Support the same language than the Cartesian product, and more:
130  * * `x == k` is over-approximated by interpreting `x == k` in both bounds.
131  * * `x in k` is over-approximated by interpreting `x in k` in both bounds.
132  * * `var x:B` when the underlying universe is arithmetic and preserve concrete covers.
133  * Therefore, the element `k` is always in \f$ \gamma(lb) \cap \gamma(ub) \f$. */
134  template<class F, class Env>
135  CUDA NI static iresult<F> interpret_tell(const F& f, const Env& env) {
136  if(f.is_binary() &&
137  (f.sig() == EQ ||
138  (f.sig() == IN && f.seq(1).is(F::S))))
139  {
140  auto cp_res = CP::interpret_tell(f, env);
141  if(cp_res.has_value()) {
142  local_type itv(cp_res.value());
143  return std::move(iresult<F>(std::move(itv)).join_warnings(std::move(cp_res)));
144  }
145  }
146  if constexpr(LB::preserve_concrete_covers && LB::is_arithmetic) {
147  if(f.is(F::E)) {
148  auto sort = f.sort();
149  if(sort.has_value() && sort->is_bool()) {
150  return local_type(LB::geq_k(LB::pre_universe::zero()), UB::leq_k(UB::pre_universe::one()));
151  }
152  }
153  }
154  return forward_to_cp<true>(f, env);
155  }
156 
157  /** Support the same language than the Cartesian product, and more:
158  * * `x != k` is under-approximated by interpreting `x != k` in the lower bound.
159  * * `x == k` is interpreted by over-approximating `x == k` in both bounds and then verifying both bounds are the same. */
160  template<class F, class Env>
161  CUDA NI static iresult<F> interpret_ask(const F& f, const Env& env) {
162  if(f.is_binary() && f.sig() == NEQ) {
163  auto lb = LB::interpret_ask(f, env);
164  if(lb.has_value()) {
165  local_type itv(lb.value(), UB::bot());
166  return std::move(iresult<F>(std::move(itv)).join_warnings(std::move(lb)));
167  }
168  }
169  else if(f.is_binary() && f.sig() == EQ) {
170  auto lb = LB::interpret_tell(f, env);
171  if(lb.has_value()) {
172  auto ub = UB::interpret_tell(f, env);
173  if(ub.has_value()) {
174  if(lb.value() == ub.value()) {
175  return iresult<F>(local_type(lb.value(), ub.value()));
176  }
177  else {
178  return iresult<F>(IError<F>(true, name, "When interpreting equality, the underlying bounds LB and UB failed to agree on the same value.", f));
179  }
180  }
181  }
182  }
183  return forward_to_cp<false>(f, env);
184  }

345  // This operation preserves top, i.e., \f$ abs(x) \in [\top] \f$ if \f$ x \in [\top] \f$, \f$ [\top] \f$ being the equivalence class of top elements.
346  template<class L>
347  CUDA constexpr static local_type abs(const Interval<L>& x) {
348  switch(sign(x)) {
349  case PP: return x;
350  case NP: return local_type( // [0..max(-lb, ub)]
351  LB2::geq_k(LB2::pre_universe::zero()),
352  meet(dual<UB2>(LB2::template fun<ABS>(typename L::template flat_type<battery::local_memory>(x.lb()))), x.ub()));
353  case NN: return neg(x);
354  case PN: return local_type(x.lb(), meet(UB2::leq_k(UB2::pre_universe::zero()), x.ub()));
355  }
356  assert(0); // all cases should be covered:
357  return top();
358  }

389  /** The sign function is monotone w.r.t. the order of `Interval<A>` and `bounds_sign`. */
390  template<class A>
391  CUDA constexpr static bounds_sign sign(const Interval<A>& a) {
392  if(a.lb() >= LB2::geq_k(LB2::pre_universe::zero())) {
393  if(a.ub() > UB2::leq_k(UB2::pre_universe::zero())) {
394  return PN;
395  }
396  else {
397  return PP;
398  }
399  }
400  else {
401  if(a.ub() >= UB2::leq_k(UB2::pre_universe::zero())) {
402  return NN;
403  }
404  else {
405  return NP;
406  }
407  }

422 public:
423  template<class L, class K>
424  CUDA constexpr static local_type mul(const Interval<L>& l, const Interval<K>& k) {
425  auto a = typename Interval<L>::local_type(l);
426  auto b = typename Interval<K>::local_type(k);
427  // Interval multiplication case, [al..au] * [bl..bu]
428  switch(sign(a)) {
429  case PP:
430  switch(sign(b)) {
431  case PP: return mul2(a, b);
432  case NP: return mul2(a.ub2(), b);
433  case NN: return mul2(reverse(a), b);
434  case PN:
435  if(b.as_product().is_top()) { return top(); }
436  else { return mul2(a.lb2(), b); }
437  }
438  case NP:
439  switch(sign(b)) {
440  case PP: return mul2(a, b.ub2());
441  // Note: we use meet for both bounds because UB is the dual of LB (e.g., if meet in LB is min, then meet in UB is max).
442  case NP: return local_type(
443  meet(flat_fun2<MUL, LB2>(a.lb(), b.ub()), flat_fun2<MUL, LB2>(a.ub(), b.lb())),
444  meet(flat_fun2<MUL, UB2>(a.lb(), b.lb()), flat_fun2<MUL, UB2>(a.ub(), b.ub())));
445  case NN: return mul2(reverse(a), b.lb2());
446  case PN:
447  if(b.as_product().is_top()) { return top(); }
448  else { return eq_zero(); }
449  }
450  case NN:
451  switch(sign(b)) {
452  case PP: return mul2(a, reverse(b));
453  case NP: return mul2(a.lb2(), reverse(b));
454  case NN: return mul2(reverse(a), reverse(b));
455  case PN:
456  if(b.as_product().is_top()) { return top(); }
457  else { return mul2(a.ub2(), reverse(b)); }
458  }
459  case PN:
460  if(a.as_product().is_top()) { return top(); }
461  else {
462  switch(sign(b)) {
463  case PP: return mul2(a, b.lb2());
464  case NP: return eq_zero();
465  case NN: return mul2(reverse(a), b.ub2());
466  case PN:
467  if(b.as_product().is_top()) { return top(); }
468  else {
469  return local_type(
470  join(flat_fun2<MUL, LB2>(a.lb(), b.lb()), flat_fun2<MUL, LB2>(a.ub(), b.ub())),
471  join(flat_fun2<MUL, UB2>(a.lb(), b.ub()), flat_fun2<MUL, UB2>(a.ub(), b.lb())));
472  }
473  }
474  }
475  }
476  assert(0); // All cases should be covered.
477  return top();
478  }

488 public:
489  template<Sig divsig, class L, class K>
490  CUDA constexpr static local_type div(const Interval<L>& l, const Interval<K>& k) {
491  auto a = typename Interval<L>::local_type(l);
492  auto b = typename Interval<K>::local_type(k);
493  // Below, you can find cases where a or b are top, similar to multiplication above, but this has bugs and need to be studied in more depth.
494  if(a.is_top() || b.is_top()) { return top(); }
495  using UB_K = typename Interval<K>::UB::local_type;
496  constexpr auto leq_zero = UB_K::leq_k(UB_K::pre_universe::zero());
497  // Interval division, [al..au] / [bl..bu]
498  switch(sign(b)) {
499  case PP:
500  if(b.ub() >= leq_zero) { return top(); } // b is a singleton equal to zero.
501  switch(sign(a)) {
502  case PP: return div2<divsig>(a.lb(), a.ub(), b.ub(), b.lb());
503  case NP: return div2<divsig>(a.lb(), a.ub(), b.lb(), b.lb());
504  case NN: return div2<divsig>(a.lb(), a.ub(), b.lb(), b.ub());
505  case PN:
506  if(a.as_product().is_top()) { return top(); }
507  else { return div2<divsig>(a.lb(), a.ub(), b.ub(), b.ub()); }
508  }
509  case NP:
510  if(a.is_top()) { return top(); }
511  else {
512  if constexpr(L::preserve_concrete_covers && K::preserve_concrete_covers) { // In the discrete case, division can be more precise.
513  switch(sign(a)) {
514  case PP: return div2<divsig>(a.ub(), a.ub(), b.lb(), b.ub());
515  case NP: return local_type(
516  meet(LB2::template guarded_div<divsig>(a.lb(), b.ub()), LB2::template guarded_div<divsig>(a.ub(), b.lb())),
517  meet(UB2::template guarded_div<divsig>(a.lb(), b.lb()), UB2::template guarded_div<divsig>(a.ub(), b.ub())));
518  case NN: return div2<divsig>(a.lb(), a.lb(), b.ub(), b.lb());
519  case PN: return (a.as_product().is_top()) ? top() : eq_zero();
520  }
521  }
522  else {
523  return bot();
524  }
525  }
526  case NN:
527  switch(sign(a)) {
528  case PP: return div2<divsig>(a.ub(), a.lb(), b.ub(), b.lb());
529  case NP: return div2<divsig>(a.ub(), a.lb(), b.ub(), b.ub());
530  case NN: return div2<divsig>(a.ub(), a.lb(), b.lb(), b.ub());
531  case PN:
532  if(a.as_product().is_top()) { return top(); }
533  else { return div2<divsig>(a.ub(), a.lb(), b.lb(), b.lb()); }
534  }
535  case PN:
536  if(b.as_product().is_top()) { return top(); }
537  if constexpr(L::preserve_concrete_covers && K::preserve_concrete_covers) {
538  switch(sign(a)) {
539  case PP: return div2<divsig>(a.lb(), a.lb(), b.ub(), b.lb());
540  case NP: return eq_zero();
541  case NN: return div2<divsig>(a.ub(), a.ub(), b.ub(), b.lb());
542  case PN:
543  if(a.as_product().is_top()) { return top(); }
544  else {
545  return local_type(
546  join(LB2::template guarded_div<divsig>(a.lb(), b.ub()), LB2::template guarded_div<divsig>(a.ub(), b.lb())),
547  join(UB2::template guarded_div<divsig>(a.lb(), b.lb()), UB2::template guarded_div<divsig>(a.ub(), b.ub())));
548  }
549  }
550  }
551  else {
552  return top(); /* This is by duality with the case above, but I am unsure if it makes sense. */
553  }
554  }
555  assert(0); // All cases should be covered.
556  return top();
557  }
