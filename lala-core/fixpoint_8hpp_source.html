51 /** A simple form of fixpoint computation based on Kleene fixpoint.
52  * At each iteration, the refinement operations \f$ f_1, \ldots, f_n \f$ are composed by parallel composition \f$ f = f_1 \| \ldots \| f_n \f$ meaning they are executed in parallel by different threads.
53  * This is called an asynchronous iteration and it is due to (Cousot, Asynchronous iterative methods for solving a fixed point system of monotone equations in a complete lattice, 1977).
54  * The underlying lattice on which we iterate must provide two methods:
55  * - `a.refine(int, BInc&)`: call the ith refinement functions and set `has_changed` to `true` if `a` has changed. Note that if `a.is_top()` is `true`, then `has_changed` must stay false for all refinement functions.
56  * - `a.num_refinements()`: return the number of refinement functions.
57  * \tparam Group is a CUDA cooperative group class.
58  * \tparam Memory is an atomic memory, that must be compatible with the cooperative group chosen (e.g., don't use atomic_memory_block if the group contains multiple blocks). */

106  template <class A, class M>
107  CUDA size_t fixpoint(A& a, BInc<M>& has_changed) {
108  #ifndef __CUDA_ARCH__
109  assert_cuda_arch();
110  return 0;
111  #else
112  reset();
113  barrier();
114  size_t i;
115  for(i = 1; changed[(i-1)%3]; ++i) {
116  iterate(a, changed[i%3]);
117  changed[(i+1)%3].dtell_bot(); // reinitialize changed for the next iteration.
118  barrier();
119  }
120  // It changes if we performed several iteration, or if the first iteration changed the abstract domain.
121  has_changed.tell(changed[1]);
122  has_changed.tell(changed[2]);
123  return i - 1;
124  #endif
125  }
