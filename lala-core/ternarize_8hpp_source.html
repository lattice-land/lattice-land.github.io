145 /** Create the ternary formula `x = y <sig> z`. */
146 F push_ternary(F x, F y, Sig sig, F z) {
147 if(simplify_to_unary(x,y,sig,z)) {
148 return x;
149 }
150 if(sig == SUB) {
151 /** We simplify x = y - z into y = x + z. */
152 conjunction.push_back(F::make_binary(y, EQ, F::make_binary(x, ADD, z)));
153 return x;
154 }
155 /** If the simplification was not possible, we add the ternary constraint. */
156 conjunction.push_back(F::make_binary(x, EQ, F::make_binary(y, sig, z)));
157 return x;
158 }

160 F ternarize_unary(const F& f, bool toplevel = false) {
161 F x = ternarize(f.seq(0));
162 switch(f.sig()) {
163 /** -x ~~> t = 0 - x */
164 case NEG: {
165 F t = push_ternary(introduce_int_var(), ternarize(F::make_z(0)), SUB, x);
166 if(toplevel) {
167 return ternarize(F::make_binary(t, NEQ, F::make_z(0)), true);
168 }
169 return t;
170 }
171 /** |x| ~~> t1 = 0 - x /\ t2 = max(x, t1) /\ t2 >= 0 */
172 case ABS: {
173 F t1 = ternarize(F::make_unary(NEG, x));
174 F t2 = introduce_int_var();

176 compute(F::make_binary(t2, GEQ, F::make_z(0)));
177 if(toplevel) {

179 }
180 return t2;
181 }

183 case NOT: return ternarize(F::make_binary(x, EQ, F::make_z(0)), toplevel);

223 F ternarize_binary(F f, bool toplevel) {
224 if(f.sig() == IN && f.is_binary() && f.seq(0).is_variable() && f.seq(1).is(F::S)) {
225 if(toplevel) {
226 compute(decompose_in_constraint(f));
227 /** The decomposition of x in S does not capture the approximation x >= min(S) /\ x <= max(S).
228 * Therefore, we still give this unary constraint to the interval store in order to over-approximate it.
229 * We avoid adding the over-approximation if the decomposition is also a unary constraint. */
230 if(f.seq(1).s().size() > 1) {
231 conjunction.push_back(f);
232 }
233 return F::make_true(); /* unused anyways. */
234 }
235 else {
236 return ternarize(decompose_in_constraint(f), toplevel);
237 }
238 }
239 /** We introduce a new temporary variable `t0`.
240 * The type of `t0` is decided by the return type of the operator and whether we are at toplevel or not.
241 */
242 F t0;
243 F t1;
244 F t2;
245 bool almost_ternary = false;
246 /** We first handle "almost ternarized" constraint.
247 * If the symbol is already an equality with a variable on one side, we only need to ternarize the other half.
248 * We set t0 to be the variable and proceeds. */
249 if((((f.seq(0).is_variable() || f.seq(0).is_constant()) && f.seq(1).is_binary())
250 ||((f.seq(1).is_variable() || f.seq(1).is_constant()) && f.seq(0).is_binary()))
251 && (f.sig() == EQUIV || f.sig() == EQ))
252 {
253 int left = f.seq(0).is_binary();
254 int right = f.seq(1).is_binary();
255 /** If a IN constraint appears on the right side, we decompose it here and immediately call ternarize again. */
256 if(f.seq(right).sig() == IN) {
257 return ternarize(F::make_binary(f.seq(left), EQ, decompose_in_constraint(f.seq(right))), toplevel);
258 }
259 t0 = ternarize(f.seq(left));
260 f = f.seq(right);
261 toplevel = false;
262 almost_ternary = true;
263 }
264 t1 = ternarize(f.seq(0));
265 t1 = booleanize(t1, f.sig());
266 t2 = ternarize(f.seq(1));
267 t2 = booleanize(t2, f.sig());
268 if(!almost_ternary) {
269 /** We don't need to create t0 for these formulas at toplevel. */
270 if(toplevel && (f.sig() == NEQ || f.sig() == XOR || f.sig() == IMPLY || f.sig() == GT || f.sig() == LT)) {}
271 else if(is_logical(f.sig()) || is_predicate(f.sig())
272 || ((f.sig() == MIN || f.sig() == MAX) && is_boolean(t1) && is_boolean(t2)))
273 {
274 t0 = toplevel ? ternarize_constant(F::make_z(1)) : introduce_bool_var();
275 }
276 else {
277 t0 = toplevel ? ternarize_constant(F::make_z(1)) : introduce_int_var();
278 }
279 }
280 switch(f.sig()) {
281 case AND:
282 case MIN: return push_ternary(t0, t1, MIN, t2);
283 case OR:
284 case MAX: return push_ternary(t0, t1, MAX, t2);
285 case EQ:
286 case EQUIV: return push_ternary(t0, t1, EQ, t2);
287 // x xor y ~~> t1 = not t2 /\ t1 = (x = y)
288 case NEQ:
289 case XOR: {
290 if(toplevel) {
291 return push_ternary(ternarize_constant(F::make_z(0)), t1, EQ, t2);
292 }
293 push_ternary(ternarize(F::make_unary(NOT, t0)), t1, EQ, t2);
294 return t0;
295 }
296 case IMPLY: return ternarize(F::make_binary(F::make_unary(NOT, t1), OR, t2), toplevel);
297 case LEQ: return push_ternary(t0, t1, LEQ, t2);
298 // x >= y ~~> y <= x
299 case GEQ: return push_ternary(t0, t2, LEQ, t1);
300 // x > y ~~> !(x <= y)
301 case GT: {
302 if(toplevel) {
303 return push_ternary(ternarize_constant(F::make_z(0)), t1, LEQ, t2);
304 }
305 push_ternary(ternarize(F::make_unary(NOT, t0)), t1, LEQ, t2);
306 return t0;
307 }
308 // x < y ~~> y > x ~~> !(y <= x)
309 case LT: {
310 if(toplevel) {
311 return push_ternary(ternarize_constant(F::make_z(0)), t2, LEQ, t1);
312 }
313 push_ternary(ternarize(F::make_unary(NOT, t0)), t2, LEQ, t1);
314 return t0;
315 }
316 default: {
317 return push_ternary(t0, t1, f.sig(), t2);
318 }

322 std::pair<F, F> binarize_middle(const F& f) {
323 assert(f.is(F::Seq) && f.seq().size() > 2);
324 battery::vector<F, allocator_type> left;
325 battery::vector<F, allocator_type> right;
326 int i;
327 for(i = 0; i < f.seq().size() / 2; ++i) {
328 left.push_back(f.seq(i));
329 }
330 for(; i < f.seq().size(); ++i) {
331 right.push_back(f.seq(i));
332 }
333 return {

335 ternarize(right.size() == 1 ? right.back() : F::make_nary(f.sig(), std::move(right)))};
336 }

338 F ternarize_nary(const F& f, bool toplevel) {
339 if(is_associative(f.sig())) {
340 auto [t1, t2] = binarize_middle(f);
341 return ternarize(F::make_binary(t1, f.sig(), t2), toplevel);
342 }
343 else {
344 F tmp = ternarize(F::make_binary(f.seq(0), f.sig(), f.seq(1)));
345 for (int i = 2; i < f.seq().size() - 1; i++) {
346 tmp = ternarize(F::make_binary(tmp, f.sig(), f.seq(i)));
347 }
348 tmp = ternarize(F::make_binary(tmp, f.sig(), f.seq(f.seq().size() - 1)), toplevel);
349 return tmp;
350 }

353 F ternarize(const F& f, bool toplevel = false) {
354 if (f.is_variable()) {
355 if(toplevel) {
356 return ternarize(F::make_binary(f, NEQ, F::make_z(0)), true);
357 }
358 return f;
359 }
360 else if (f.is(F::Z) || f.is(F::B)) {
361 if(toplevel) {
362 return f.to_z() != 0 ? F::make_true() : F::make_false();
363 }
364 return ternarize_constant(f);
365 }
366 else if (f.is(F::S)) {
367 return f;
368 }
369 else if (f.is_unary()) {
370 return ternarize_unary(f, toplevel);
371 }
372 else if (f.is_binary()) {
373 return ternarize_binary(f, toplevel);
374 }
375 else if (f.is(F::Seq) && f.seq().size() > 2) {
376 return ternarize_nary(f, toplevel);
377 }
378 printf("%% Detected during ternarization (during preprocessing): unsupported formula.");
379 f.print();
380 printf("\n");
381 exit(EXIT_FAILURE);
382 }

385 void compute(const F& f) {
386 if (f.is(F::Seq) && f.sig() == AND) {
387 const auto& seq = f.seq();
388 for (int i = 0; i < seq.size(); ++i) {
389 compute(f.seq(i));
390 }
391 }
392 else if(f.is(F::E)) {
393 existentials.push_back(f);
394 std::string varname(battery::get<0>(f.exists()).data());
395 name2exists[varname] = existentials.size() - 1;
396 /** If ternarize has been called before, some temporary variables __VAR_Z_* and __VAR_B_* might already have been created.
397 * In that case, we need to update the counters to avoid conflicts.
398 * This is not perfect, because it requires existential quantifier to be in the beginning of the formula (before we introduce any variable).
399 * For more robustness, we should rename the variables in the formula, if introduced_int_vars >= X in __VAR_Z_X.
400 */
401 introduce_existing_var(varname);
402 }
403 else if(!f.is(F::ESeq) && !is_extended_ternary_form(f)) {
404 ternarize(f, true);
405 }
406 // Either it is unary, an extended formula or already in ternary form.
407 else {
408 conjunction.push_back(f);
409 }
410 }

426/**
427 * Given a formula `f`, we transform it into a conjunction of formulas of this form:
428 * 1. `x <op> c` where `c` is a constant.
429 * 2. `x = (y <op> z)` where `<op>` is a binary operator, either arithmetic or a comparison (`=`, `<=`).
430 * This ternary form is used by the lala-pc/PIR solver.
431 */
432template <class F, class Env = VarEnv<battery::standard_allocator>>
433F ternarize(const F& f, const Env& env = Env(), const std::vector<int>& constants = {}) {
434 impl::Ternarizer<F, Env> ternarizer(env);
435 for(int c : constants) {
436 ternarizer.ternarize_constant(F::make_z(c));
437 }
438 ternarizer.compute(f);
439 return std::move(ternarizer).create();
440}
