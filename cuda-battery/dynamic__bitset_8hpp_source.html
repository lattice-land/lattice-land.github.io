12/** \file dynamic_bitset.hpp
13 * This class is similar to `bitset` but the size is specified at runtime in the constructor.
14 * For simplicity of implementation, the size is a multiple of sizeof(T); that could be improved in future version if needed.
15 */

17namespace battery {

19/**
20 * \tparam `Mem` is a memory consistency, for instance `local_memory`, see `memory.hpp`.
21 * \tparam `T` is the underlying type defining the blocks of the bitset.
22*/
23template <class Mem, class Allocator = standard_allocator, class T = unsigned long long>
24class dynamic_bitset {
25private:
26 constexpr static const size_t BITS_PER_BLOCK = sizeof(T) * CHAR_BIT;
27
28 // Would be better to have constexpr, but it gives an error "undefined in device code", probably because of the call to the constructor of `T`.
29 #define ZERO (T{0})
30 #define ONES (~T{0})
31
32 using block_type = typename Mem::template atomic_type<T>;
33 using allocator_type = Allocator;
34 using this_type = dynamic_bitset<Mem, Allocator, T>;
35
36 /** Suppose T = char, with 2 blocks. Then the bitset "0000 00100000" is represented as:
37 *
38 * blocks index: 0 1
39 * blocks: 00100000 00000000
40 * indexes: 76543210 ...98
41 *
42 * We have `bitset.test(5) == true`.
43 *
44 * Note that the last block is the one carrying the most significant bits, and also the one that is potentially padded with zeroes.
45 * */
46 vector<block_type, allocator_type> blocks;
47
48private:
49 CUDA constexpr size_t num_blocks(size_t n) {
50 return n / BITS_PER_BLOCK + (n % BITS_PER_BLOCK != 0);
51 }
52
53public:
54 template <class Mem2, class Allocator2, class T2>
55 friend class dynamic_bitset;
56
57 static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>, "Block of a bitset must be defined on an unsigned integer type.");
58
59 CUDA dynamic_bitset(const allocator_type& alloc = allocator_type())
60 : blocks(alloc) {}
61
62 /** Create a bitset with a size of at least `num_bits`. */
63 CUDA dynamic_bitset(size_t at_least_num_bits, const allocator_type& alloc = allocator_type())
64 : blocks(num_blocks(at_least_num_bits), ZERO, alloc) {}
65
66 CUDA dynamic_bitset(const char* bit_str, const allocator_type& alloc = allocator_type())
67 : blocks(num_blocks(strlen(bit_str)), ZERO, alloc)
68 {
69 for(size_t i = strlen(bit_str), j = 0; i > 0; --i, ++j) {
70 if(bit_str[i-1] == '1') {
71 set(j);
72 }
73 }
74 }
75
76 /** Set all bits in the range [start..end] to `1` and create a bitset with at least `end+1` bits.
77 * `end >= start`. */
78 CUDA dynamic_bitset(size_t start, size_t end, const allocator_type& alloc = allocator_type())
79 : dynamic_bitset(end+1, alloc) {
80 assert(end >= start);
81 int block_start = start / BITS_PER_BLOCK;
82 int block_end = end / BITS_PER_BLOCK;
83 store(blocks[block_start], ONES << (start % BITS_PER_BLOCK));
84 for(int k = block_start + 1; k <= block_end; ++k) {
85 store(blocks[k], ONES);
86 }
87 store(blocks[block_end], Mem::load(blocks[block_end]) & (ONES >> ((BITS_PER_BLOCK-(end % BITS_PER_BLOCK)-1))));
88 }

91 CUDA dynamic_bitset(const dynamic_bitset<Mem2, Alloc2, T>& other,
92 const allocator_type& alloc = allocator_type())
93 : blocks(other.blocks.size(), ZERO, alloc)
94 {
95 for(int i = 0; i < blocks.size(); ++i) {
96 Mem::store(blocks[i], Mem2::load(other.blocks[i]));
97 }

139 assert(pos < size());

143 CUDA constexpr bool all() const {
144 for(int i = 0; i < blocks.size(); ++i) {
145 if(Mem::load(blocks[i]) != ONES) {
146 return false;
147 }

170 CUDA constexpr size_t size() const {
171 return BITS_PER_BLOCK * blocks.size();
172 }

174 // Only the values of the first `at_least_num_bits` are copied in the new resized bitset.
175 CUDA void resize(size_t at_least_num_bits) {
176 // NOTE: We cannot call vector.resize because it does not support resizing non-copyable, non-movable types such as atomics.
177 // Therefore, we implement our own resizing function using explicit load and store operations.
178 this_type bitset2(at_least_num_bits, get_allocator());
179 for(int i = 0; i < at_least_num_bits && i < size(); ++i) {
180 bitset2.set(i, test(i));
181 }
182 blocks.swap(bitset2.blocks);
183 }

185 CUDA constexpr size_t count() const {
186 size_t bits_at_one = 0;
187 for(int i = 0; i < blocks.size(); ++i){
188 bits_at_one += popcount(blocks[i]);
189 }
190 return bits_at_one;
191 }

193 CUDA constexpr dynamic_bitset& set(size_t pos) {
194 assert(pos < size());

196 return *this;
197 }

205 for(int i = 0; i < blocks.size(); ++i) {

230 template <class Mem2, class Alloc2>
231 CUDA constexpr bool is_subset_of(const dynamic_bitset<Mem2, Alloc2, T>& other) const {
232 size_t min_size = min(blocks.size(), other.blocks.size());
233 for(int i = 0; i < min_size; ++i) {
234 T block = Mem::load(blocks[i]);
235 if((block & Mem2::load(other.blocks[i])) != block) {
236 return false;
237 }
238 }
239 if(size() > other.size()) {
240 for(int i = min_size; i < blocks.size(); ++i) {
241 if(Mem::load(blocks[i]) != ZERO) {
242 return false;
243 }
244 }
245 }
246 return true;
247 }

249 template <class Mem2, class Alloc2>
250 CUDA constexpr bool is_proper_subset_of(const dynamic_bitset<Mem2, Alloc2, T>& other) const {
251 bool proper = false;
252 size_t min_size = min(blocks.size(), other.blocks.size());
253 for(int i = 0; i < min_size; ++i) {
254 T block = Mem::load(blocks[i]);
255 T block2 = Mem2::load(other.blocks[i]);
256 if((block & block2) != block) {
257 return false;
258 }
259 proper |= block != block2;
260 }
261 if(proper && blocks.size() > other.blocks.size()) {
262 for(int i = min_size; i < blocks.size(); ++i) {
263 if(Mem::load(blocks[i]) != ZERO) {
264 return false;
265 }
266 }
267 }
268 else if(!proper && blocks.size() < other.blocks.size()) {
269 for(int i = min_size; i < other.blocks.size(); ++i) {
270 if(Mem2::load(other.blocks[i]) != ZERO) {
271 return true;
272 }
273 }
274 }
275 return proper;
276 }
