12 /** \file dynamic_bitset.hpp
13  * This class is similar to `bitset` but the size is specified at runtime in the constructor.
14  * For simplicity of implementation, the size is a multiple of sizeof(T); that could be improved in future version if needed.
15  */

19 /**
20  * \tparam `Mem` is a memory consistency, for instance `local_memory`, see `memory.hpp`.
21  * \tparam `T` is the underlying type defining the blocks of the bitset.
22 */
23 template <class Mem, class Allocator = standard_allocator, class T = unsigned long long>
24 class dynamic_bitset {
25 private:
26  constexpr static size_t BITS_PER_BLOCK = sizeof(T) * CHAR_BIT;
27  constexpr static T ZERO = T{0};
28  constexpr static T ONES = ~T{0};
29 
30  using block_type = typename Mem::template atomic_type<T>;
31  using allocator_type = Allocator;
32  using this_type = dynamic_bitset<Mem, Allocator, T>;
33 
34  /** Suppose T = char, with 2 blocks. Then the bitset "0000 00100000" is represented as:
35  *
36  * blocks index: 0 1
37  * blocks: 00100000 00000000
38  * indexes: 76543210 ...98
39  *
40  * We have `bitset.test(5) == true`.
41  *
42  * Note that the last block is the one carrying the most significant bits, and also the one that is potentially padded with zeroes.
43  * */
44  vector<block_type, allocator_type> blocks;
45 
46 private:
47  CUDA constexpr size_t num_blocks(size_t n) {
48  return n / BITS_PER_BLOCK + (n % BITS_PER_BLOCK != 0);
49  }
50 
51 public:
52  static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>, "Block of a bitset must be defined on an unsigned integer type.");
53 
54  CUDA dynamic_bitset(const allocator_type& alloc = allocator_type())
55  : blocks(alloc) {}

57  /** Create a bitset with a size of at least `num_bits`. */
58  CUDA dynamic_bitset(size_t at_least_num_bits, const allocator_type& alloc = allocator_type())
59  : blocks(num_blocks(at_least_num_bits), ZERO, alloc) {}

61  CUDA dynamic_bitset(const char* bit_str, const allocator_type& alloc = allocator_type())
62  : blocks(num_blocks(strlen(bit_str)), ZERO, alloc)
63  {
64  for(size_t i = strlen(bit_str), j = 0; i > 0; --i, ++j) {
65  if(bit_str[i-1] == '1') {
66  set(j);
67  }
68  }
69  }

71  /** Set all bits in the range [start..end] to `1` and create a bitset with at least `end+1` bits.
72  * `end >= start`. */
73  CUDA dynamic_bitset(size_t start, size_t end, const allocator_type& alloc = allocator_type())
74  : dynamic_bitset(end+1, alloc) {
75  assert(end >= start);
76  int block_start = start / BITS_PER_BLOCK;
77  int block_end = end / BITS_PER_BLOCK;
78  store(blocks[block_start], ONES << (start % BITS_PER_BLOCK));
79  for(int k = block_start + 1; k <= block_end; ++k) {
80  store(blocks[k], ONES);
81  }
82  store(blocks[block_end], Mem::load(blocks[block_end]) & (ONES >> ((BITS_PER_BLOCK-(end % BITS_PER_BLOCK)-1))));
83  }

209  template <class Mem2, class Alloc2>
210  CUDA constexpr bool is_subset_of(const dynamic_bitset<Mem2, Alloc2, T>& other) const {
211  size_t min_size = min(blocks.size(), other.blocks.size());
212  for(int i = 0; i < min_size; ++i) {
213  T block = Mem::load(blocks[i]);
214  if((block & Mem2::load(other.blocks[i])) != block) {
215  return false;
216  }
217  }
218  if(size() > other.size()) {
219  for(int i = min_size; i < blocks.size(); ++i) {
220  if(Mem::load(blocks[i]) != ZERO) {
221  return false;
222  }
223  }
224  }
225  return true;
226  }

228  template <class Mem2, class Alloc2>
229  CUDA constexpr bool is_proper_subset_of(const dynamic_bitset<Mem2, Alloc2, T>& other) const {
230  bool proper = false;
231  size_t min_size = min(blocks.size(), other.blocks.size());
232  for(int i = 0; i < min_size; ++i) {
233  T block = Mem::load(blocks[i]);
234  T block2 = Mem2::load(other.blocks[i]);
235  if((block & block2) != block) {
236  return false;
237  }
238  proper |= block != block2;
239  }
240  if(proper && blocks.size() > other.blocks.size()) {
241  for(int i = min_size; i < blocks.size(); ++i) {
242  if(Mem::load(blocks[i]) != ZERO) {
243  return false;
244  }
245  }
246  }
247  else if(!proper && blocks.size() < other.blocks.size()) {
248  for(int i = min_size; i < other.blocks.size(); ++i) {
249  if(Mem2::load(other.blocks[i]) != ZERO) {
250  return true;
251  }
252  }
253  }
254  return proper;
255  }
