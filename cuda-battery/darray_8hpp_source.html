12 /** `TypeAllocatorDispatch` is useful for `DArray` to enforce the copy semantics mentioned there. */
13 template<typename T>
14 struct TypeAllocatorDispatch {
15  template<typename Allocator>
16  CUDA static void build(T* placement, const T& from, Allocator& allocator) {
17  if constexpr(std::is_pointer_v<T> && std::is_polymorphic_v<std::remove_pointer_t<T>>) {
18  *placement = from->clone(allocator);
19  }
20  else if constexpr(std::is_constructible<T, const T&, const Allocator&>{}) {
21  new(placement) T(from, allocator);
22  }
23  else {
24  new(placement) T(from);
25  }
26  }
27 
28  template<typename Allocator>
29  CUDA static void destroy(T& data, Allocator& allocator) {
30  if constexpr(std::is_pointer_v<T> && std::is_polymorphic_v<std::remove_pointer_t<T>>) {
31  typedef std::remove_pointer_t<T> U;
32  data->~U();
33  delete data;
34  }
35  if constexpr(std::is_destructible_v<T>) {
36  data.~T();
37  }
38  }
39 };

43 /** `DArray<T, Allocator>` is a dynamic array (so the size can depends on runtime value), but once created, the size cannot be modified (no `push_back` method).
44 It has some dedicated support for collection of _polymorphic objects_.
45 `DArray` is also usable on CPU using the `StandardAllocator` (see allocator.hpp).
46 
47 In order to make the usage of collection as uniform as possible in host and device code, and with polymorphic objects or not, a `DArray` is initialized and copied following a set of rules:
48 - Case of non-polymorphic types `T` (i.e., `is_polymorphic_v<remove_pointer_t<T>>` is false):
49  - Elements are copied using `T(const T&, const Allocator&)` if such a copy constructor is available, otherwise the normal copy constructor `T(const T&)` is used.
50 - Case of polymorphic type `T`: The objects of type `T` are always copied using a method `T* clone(const Allocator&)` where the `clone` method is supposed to be overloaded for the relevant/necessary allocator types.
51  Note that for `T* clone(GlobalAllocator)`, we suppose the object vtable is initialized on the device.
52  The polymorphic objects are automatically freed in the destructor of DArray.
53 
54 Hence, _polymorphic objects are owned by DArray_.
55 If you use this class to store an array of polymorphic objects, the polymorphic objects will be deleted when the destructor of the array is called.
56 See also `tests/polymorphic_darray_test.cpp` for example on how to implement the `clone` methods and using DArray as a polymorphic container. */
