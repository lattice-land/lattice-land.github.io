112 template<class T, class Alloc, class... Args>
113 CUDA unique_ptr<T, Alloc> allocate_unique(const Alloc& alloc, Args&&... args) {
114  Alloc allocator(alloc);
115  T* ptr = static_cast<T*>(allocator.allocate(sizeof(T)));
116  assert(ptr != nullptr);
117  if constexpr(std::is_constructible<T, Args&&..., const Alloc&>{}) {
118  new(ptr) T(std::forward<Args>(args)..., allocator);
119  }
120  else {
121  new(ptr) T(std::forward<Args>(args)...);
122  }
123  return unique_ptr<T, Alloc>(ptr, allocator);
124 }
