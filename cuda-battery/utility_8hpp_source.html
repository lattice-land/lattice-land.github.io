156 /** Cast the variable `x` from type `From` to type `To` following upper rounding rule (cast in the direction of infinity).
157  Minimal and maximal values of `From` are interpreted as infinities, and are therefore mapped to the infinities of the new types accordingly (e.g., float INF maps to int MAX_INT).
158 
159  * On CPU: Rounding mode is UPWARD after this operation.
160  * On GPU: CUDA intrinsics are used.
161 
162  Overflow: Nothing is done to prevent overflow, it mostly behaves as with `static_cast`. */
163 template<class To, class From>
164 CUDA To ru_cast(From x) {
165  if constexpr(std::is_same_v<To, From>) {
166  return x;
167  }
168  MAP_LIMITS(x, From, To)
169  #ifdef __CUDA_ARCH__
170  // Integer to floating-point number cast.
171  if constexpr(std::is_integral_v<From> && std::is_floating_point_v<To>) {
172  if constexpr(std::is_same_v<From, unsigned long long>) {
173  if constexpr(std::is_same_v<To, float>) {
174  return __ull2float_ru(x);
175  }
176  else if constexpr(std::is_same_v<To, double>) {
177  return __ull2double_ru(x);
178  }
179  else {
180  static_assert(std::is_same_v<To, float>, "Unsupported combination of types in ru_cast.");
181  }
182  }
183  else if constexpr(std::is_same_v<From, int>) {
184  if constexpr(std::is_same_v<To, float>) {
185  return __int2float_ru(x);
186  }
187  else if constexpr(std::is_same_v<To, double>) {
188  return __int2double_rn(x);
189  }
190  else {
191  static_assert(std::is_same_v<To, float>, "Unsupported combination of types in ru_cast.");
192  }
193  }
194  else {
195  static_assert(sizeof(long long int) >= sizeof(From));
196  if constexpr(std::is_same_v<To, float>) {
197  return __ll2float_ru(x);
198  }
199  else if constexpr(std::is_same_v<To, double>) {
200  return __ll2double_ru(x);
201  }
202  else {
203  static_assert(std::is_same_v<To, float>, "Unsupported combination of types in ru_cast.");
204  }
205  }
206  }
207  // Floating-point number to integer number.
208  else if constexpr(std::is_floating_point_v<From> && std::is_integral_v<To>) {
209  if constexpr(std::is_same_v<From, float>) {
210  return static_cast<To>(__float2ll_ru(x));
211  }
212  else if constexpr(std::is_same_v<From, double>) {
213  return static_cast<To>(__double2ll_ru(x));
214  }
215  else {
216  static_assert(std::is_same_v<From, float>, "Unsupported combination of types in ru_cast.");
217  }
218  }
219  // Floating-point to floating-point.
220  else if constexpr(std::is_same_v<From, double> && std::is_same_v<To, float>) {
221  return __double2float_ru(x);
222  }
223  #else
224  // Integer to floating-point number cast.
225  if constexpr(std::is_integral_v<From> && std::is_floating_point_v<To>) {
226  #ifndef __GNUC__
227  #pragma STDC FENV_ACCESS ON
228  #endif
229  int r = std::fesetround(FE_UPWARD);
230  assert(r == 0);
231  return static_cast<To>(x);
232  }
233  // Floating-point number to integer number.
234  else if constexpr(std::is_floating_point_v<From> && std::is_integral_v<To>) {
235  return static_cast<To>(std::ceil(x));
236  }
237  // Floating-point to floating-point.
238  else if constexpr(std::is_same_v<From, double> && std::is_same_v<To, float>) {
239  #ifndef __GNUC__
240  #pragma STDC FENV_ACCESS ON
241  #endif
242  int r = std::fesetround(FE_UPWARD);
243  assert(r == 0);
244  return static_cast<To>(x);
245  }
246  #endif
247  return static_cast<To>(x);
248 }

369 template<class T>
370 CUDA int countl_zero(T x) {
371  static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>, "countl_zero only works on unsigned integers");
372  #ifdef __CUDA_ARCH__
373  // If the size of `T` is smaller than `int` or `long long int` we must remove the extra zeroes that are added after conversion.
374  if constexpr(sizeof(T) <= sizeof(int)) {
375  return __clz(x) - ((sizeof(int) - sizeof(T)) * CHAR_BIT);
376  }
377  else if constexpr(sizeof(T) <= sizeof(long long int)) {
378  return __clzll(x) - ((sizeof(long long int) - sizeof(T)) * CHAR_BIT);
379  }
380  else {
381  static_assert(sizeof(T) < sizeof(long long int), "countX_Y (CUDA version) only supports types smaller than long long int.");
382  }
383  #elif __cpp_lib_bitops
384  return std::countl_zero(x);
385  #else
386  int c = 0;
387  constexpr int bits = sizeof(T) * CHAR_BIT;
388  constexpr T mask = (T)1 << (bits - 1);
389  for(int i = 0; i < bits && (x & mask) == 0; ++i) {
390  c += (x & mask) == 0;
391  x <<= 1;
392  }
393  return c;
394  #endif
395 }

472 template <class T>
473 CUDA T ipow(T a, T b) {
474  static_assert(std::is_integral_v<T>, "ipow is only working on integer value.");
475  assert(b >= 0);
476  if(b == 2) {
477  return a*a;
478  }
479  // Code taken from GeCode implementation.
480  T p = 1;
481  do {
482  if (b % 2 == 0) {
483  a *= a;
484  b >>= 1;
485  } else {
486  p *= a;
487  b--;
488  }
489  } while (b > 0);
490  return p;
491 }
