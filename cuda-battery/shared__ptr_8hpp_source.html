11 /** Similar to std::shared_ptr with small differences:
12  * - There is no specialization for arrays (e.g., shared_ptr<T[]>).
13  - We rely on an allocator to allocate/deallocate the memory.
14  - No support for aliasing constructors.
15  - No special thread-safety support.
16  Similarly to vector, the allocator is scoped, meaning it is propagated to the underlying type constructor if it takes one. */

169 template<class T, class Alloc, class... Args>
170 CUDA shared_ptr<T, Alloc> allocate_shared(const Alloc& alloc, Args&&... args) {
171  Alloc allocator(alloc);
172  T* ptr = static_cast<T*>(allocator.allocate(sizeof(T)));
173  assert(ptr != nullptr);
174  if constexpr(std::is_constructible<T, Args&&..., const Alloc&>{}) {
175  new(ptr) T(std::forward<Args>(args)..., allocator);
176  }
177  else {
178  new(ptr) T(std::forward<Args>(args)...);
179  }
180  return shared_ptr<T, Alloc>(ptr, allocator);
181 }
