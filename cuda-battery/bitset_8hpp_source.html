11/** \file bitset.hpp
12 * An extension of the C++ STL bitset class with support for set operations (e.g., union and intersection) and for atomic read and write on the bitset blocks.
13 */
14
15namespace battery {
16
17/**
18 * \tparam `N` is the number of bits in the bitset.
19 * \tparam `Mem` is a memory consistency, for instance `local_memory`, see `memory.hpp`.
20 * \tparam `T` is the underlying type defining the blocks of the bitset.
21*/
22template <size_t N, class Mem, class T = unsigned long long>
23class bitset {
24 constexpr static size_t MAX_SIZE = N;
25private:
26 constexpr static size_t BITS_PER_BLOCK = sizeof(T) * CHAR_BIT;
27 constexpr static size_t BITS_LAST_BLOCK = (N % BITS_PER_BLOCK) == 0 ? BITS_PER_BLOCK : (N % BITS_PER_BLOCK);
28 constexpr static size_t PADDING_LAST_BLOCK = BITS_PER_BLOCK - BITS_LAST_BLOCK;
29 constexpr static size_t BLOCKS = N / BITS_PER_BLOCK + (N % BITS_PER_BLOCK != 0);
30 constexpr static T ZERO = T{0};
31 constexpr static T ONES = ~T{0};
32 /** The last block might not be fully used.
33 If we have two blocks of size 8 each, but N = 10, only 2 bits are relevant in the last block.
34 We have 10 % 8 = 2, then ONES << 2 gives 11111100, and ~(ONES << 2) = 00000011. */
35 constexpr static T ONES_LAST = PADDING_LAST_BLOCK == 0 ? ONES : (T)(~(ONES << BITS_LAST_BLOCK));
36
37 using block_type = typename Mem::template atomic_type<T>;
38
39 /** Suppose T = char, with 2 blocks. Then the bitset "0000 00100000" is represented as:
40 *
41 * blocks index: 0 1
42 * blocks: 00100000 00000000
43 * indexes: 76543210 ...98
44 *
45 * We have `bitset.test(5) == true`.
46 *
47 * Note that the last block is the one carrying the most significant bits, and also the one that is potentially padded with zeroes.
48 * */
49 block_type blocks[BLOCKS];
50
51public:
52 static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>, "Block of a bitset must be defined on an unsigned integer type.");
53
54 CUDA constexpr bitset(): blocks() {}
55
56 CUDA constexpr bitset(const char* bit_str): blocks() {
57 size_t n = min(strlen(bit_str), MAX_SIZE);
58 for(int i = n, j = 0; i > 0; --i, ++j) {
59 if(bit_str[i-1] == '1') {
60 set(j);
61 }
62 }
63 }

65 template<class Mem2>
66 CUDA constexpr bitset(const bitset<N, Mem2, T>& other) {
67 for(int i = 0; i < BLOCKS; ++i) {
68 store(blocks[i], Mem2::load(other[i]));
69 }
70 }

101 CUDA constexpr T bit_of(size_t pos) const {
102 return static_cast<T>(1) << (pos % BITS_PER_BLOCK);
103 }

106 CUDA constexpr bool test(size_t pos) const {
107 assert(pos < MAX_SIZE);
108 return load_block(pos) & bit_of(pos);
109 }

111 CUDA constexpr bool all() const {
112 int i = 0;
113 for(; i < BLOCKS - 1; ++i) {
114 if(Mem::load(blocks[i]) != ONES) {
115 return false;
116 }
117 }
118 return Mem::load(blocks[i]) == ONES_LAST;
119 }

122 for(int i = 0; i < BLOCKS; ++i) {

124 return true;
125 }
126 }
127 return false;
128 }

151 CUDA constexpr bitset& set() {
152 int i = 0;
153 for(; i < BLOCKS - 1; ++i) {
154 store(blocks[i], ONES);
155 }
156 store(blocks[i], ONES_LAST);
157 return *this;
158 }

160 CUDA constexpr bitset& set(size_t pos) {
161 assert(pos < MAX_SIZE);
162 store_block(pos, load_block(pos) | bit_of(pos));
163 return *this;
164 }

166 CUDA constexpr bitset& set(size_t pos, bool value) {
167 assert(pos < MAX_SIZE);
168 return value ? set(pos) : reset(pos);
169 }

171 CUDA constexpr bitset& reset() {
172 for(int i = 0; i < BLOCKS; ++i) {
173 store(blocks[i], ZERO);
174 }
175 return *this;
176 }
177
178 CUDA constexpr bitset& reset(size_t pos) {
179 assert(pos < MAX_SIZE);
180 store_block(pos, load_block(pos) & ~bit_of(pos));
181 return *this;
182 }

184 CUDA constexpr bitset& flip() {
185 int i = 0;
186 for(; i < BLOCKS - 1; ++i) {
187 store(blocks[i], ~Mem::load(blocks[i]));
188 }
189 store(blocks[i], ONES_LAST & ~Mem::load(blocks[i]));
190 return *this;
191 }
192
193 CUDA constexpr bitset& flip(size_t pos) {
194 assert(pos < MAX_SIZE);
195 store_block(pos, load_block(pos) ^ bit_of(pos));
196 return *this;
197 }

199 template <class Mem2>
200 CUDA constexpr bool is_subset_of(const bitset<N, Mem2, T>& other) const {
201 for(int i = 0; i < BLOCKS; ++i) {
202 T block = Mem::load(blocks[i]);
203 if((block & Mem2::load(other.blocks[i])) != block) {
204 return false;
205 }
206 }
207 return true;
208 }

210 template <class Mem2>
211 CUDA constexpr bool is_proper_subset_of(const bitset<N, Mem2, T>& other) const {
212 bool proper = false;
213 for(int i = 0; i < BLOCKS; ++i) {
214 T block = Mem::load(blocks[i]);
215 T block2 = Mem2::load(other.blocks[i]);
216 if((block & block2) != block) {
217 return false;
218 }
219 proper |= block != block2;
220 }
221 return proper;
222 }

267 CUDA constexpr int countl_zero() const {
268 int k = BLOCKS - 1;
269 if(blocks[k] > ZERO) {
270 return ::battery::countl_zero(blocks[k]) - PADDING_LAST_BLOCK;
271 }
272 --k;
273 int i = 0;
274 for(; k >= 0 && blocks[k] == ZERO; --k, ++i) {}
275 return i * BITS_PER_BLOCK
276 + BITS_LAST_BLOCK
277 + (k == -1 ? 0 : ::battery::countl_zero(blocks[k]));
278 }
279
280 CUDA constexpr int countl_one() const {
281 int k = BLOCKS - 1;
282 if(blocks[k] != ONES_LAST) {
283 return battery::countl_one((T)(blocks[k] << PADDING_LAST_BLOCK));
284 }
285 --k;
286 int i = 0;
287 for(; k >= 0 && blocks[k] == ONES; --k, ++i) {}
288 return i * BITS_PER_BLOCK
289 + BITS_LAST_BLOCK
290 + (k == -1 ? 0 : ::battery::countl_one(blocks[k]));
291 }
292
293 CUDA constexpr int countr_zero() const {
294 int k = 0;
295 for(; k < BLOCKS && blocks[k] == ZERO; ++k) {}
296 return k * BITS_PER_BLOCK
297 + (k == BLOCKS ? -PADDING_LAST_BLOCK : ::battery::countr_zero(blocks[k]));
298 }
299
300 CUDA constexpr int countr_one() const {
301 int k = 0;
302 for(; k < BLOCKS && blocks[k] == ONES; ++k) {}
303 return k * BITS_PER_BLOCK
304 + (k == BLOCKS ? 0 : ::battery::countr_one(blocks[k]));
305 }
