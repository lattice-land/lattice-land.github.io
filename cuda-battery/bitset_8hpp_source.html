11 /** \file bitset.hpp
12  * An extension of the C++ STL bitset class with support for set operations (e.g., union and intersection) and for atomic read and write on the bitset blocks.
13  */

17 /**
18  * \tparam `N` is the number of bits in the bitset.
19  * \tparam `Mem` is a memory consistency, for instance `local_memory`, see `memory.hpp`.
20  * \tparam `T` is the underlying type defining the blocks of the bitset.
21 */
22 template <size_t N, class Mem, class T = unsigned long long>
23 class bitset {
24 private:
25  constexpr static size_t BITS_PER_BLOCK = sizeof(T) * CHAR_BIT;
26  constexpr static size_t BITS_LAST_BLOCK = (N % BITS_PER_BLOCK) == 0 ? BITS_PER_BLOCK : (N % BITS_PER_BLOCK);
27  constexpr static size_t PADDING_LAST_BLOCK = BITS_PER_BLOCK - BITS_LAST_BLOCK;
28  constexpr static size_t BLOCKS = N / BITS_PER_BLOCK + (N % BITS_PER_BLOCK != 0);
29  constexpr static size_t MAX_SIZE = N;
30  constexpr static T ZERO = T{0};
31  constexpr static T ONES = ~T{0};
32  /** The last block might not be fully used.
33  If we have two blocks of size 8 each, but N = 10, only 2 bits are relevant in the last block.
34  We have 10 % 8 = 2, then ONES << 2 gives 11111100, and ~(ONES << 2) = 00000011. */
35  constexpr static T ONES_LAST = PADDING_LAST_BLOCK == 0 ? ONES : (T)(~(ONES << BITS_LAST_BLOCK));
36 
37  using block_type = typename Mem::atomic_type<T>;
38 
39  /** Suppose T = char, with 2 blocks. Then the bitset "0000 00100000" is represented as:
40  *
41  * blocks index: 0 1
42  * blocks: 00100000 00000000
43  * indexes: 76543210 ...98
44  *
45  * We have `bitset.test(5) == true`.
46  *
47  * Note that the last block is the one carrying the most significant bits, and also the one that is potentially padded with zeroes.
48  * */
49  block_type blocks[BLOCKS];
50 
51 public:
52  static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>, "Block of a bitset must be defined on an unsigned integer type.");
53 
54  CUDA constexpr bitset(): blocks() {}

56  CUDA constexpr bitset(const char* bit_str): blocks() {
57  size_t n = min(strlen(bit_str), MAX_SIZE);
58  for(int i = n-1, j = 0; i >= 0; --i, ++j) {
59  if(bit_str[i] == '1') {
60  set(j);
61  }
62  }
63  }

67  for(int i = 0; i < BLOCKS; ++i) {

112  int i = 0;
113  for(; i < BLOCKS - 1; ++i) {
114  if(Mem::load(blocks[i]) != ONES) {

184  CUDA constexpr bitset& flip() {
185  int i = 0;
186  for(; i < BLOCKS - 1; ++i) {
187  store(blocks[i], ~Mem::load(blocks[i]));

267  CUDA constexpr int countl_zero() const {
268  int k = BLOCKS - 1;
269  if(blocks[k] > ZERO) {
270  return ::battery::countl_zero(blocks[k]) - PADDING_LAST_BLOCK;
271  }
272  --k;
273  int i = 0;
274  for(; k >= 0 && blocks[k] == ZERO; --k, ++i) {}
275  return i * BITS_PER_BLOCK
276  + BITS_LAST_BLOCK
277  + (k == -1 ? 0 : ::battery::countl_zero(blocks[k]));
278  }

280  CUDA constexpr int countl_one() const {
281  int k = BLOCKS - 1;
282  if(blocks[k] != ONES_LAST) {
283  return battery::countl_one((T)(blocks[k] << PADDING_LAST_BLOCK));
284  }
285  --k;
286  int i = 0;
287  for(; k >= 0 && blocks[k] == ONES; --k, ++i) {}
288  return i * BITS_PER_BLOCK
289  + BITS_LAST_BLOCK
290  + (k == -1 ? 0 : ::battery::countl_one(blocks[k]));
291  }
