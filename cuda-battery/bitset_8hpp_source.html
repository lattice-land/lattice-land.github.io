25 template <size_t N, class Mem, class T = unsigned long long>
26 class Bitset {
27 private:
28  constexpr static size_t BITS_PER_BLOCK = sizeof(T) * CHAR_BIT;
29  constexpr static size_t BITS_LAST_BLOCK = (N % BITS_PER_BLOCK) == 0 ? BITS_PER_BLOCK : (N % BITS_PER_BLOCK);
30  constexpr static size_t PADDING_LAST_BLOCK = BITS_PER_BLOCK - BITS_LAST_BLOCK;
31  constexpr static size_t BLOCKS = N / BITS_PER_BLOCK + (N % BITS_PER_BLOCK != 0);
32  constexpr static size_t MAX_SIZE = N;
33  constexpr static T ZERO = T{0};
34  constexpr static T ONES = ~T{0};
35  /** The last block might not be fully used.
36  If we have two blocks of size 8 each, but N = 10, only 2 bits are relevant in the last block.
37  We have 10 % 8 = 2, then ONES << 2 gives 11111100, and ~(ONES << 2) = 00000011. */
38  constexpr static T ONES_LAST = PADDING_LAST_BLOCK == 0 ? ONES : (T)(~(ONES << BITS_LAST_BLOCK));
39 
40  using block_type = typename Mem::atomic_type<T>;
41 
42  /** Suppose T = char, with 2 blocks. Then the bitset "0000 00100000" is represented as:
43  *
44  * blocks index: 0 1
45  * blocks: 00100000 00000000
46  * indexes: 76543210 ...98
47  *
48  * We have `bitset.test(5) == true`.
49  *
50  * Note that the last block is the one carrying the most significant bits, and also the one that is potentially padded with zeroes.
51  * */
52  block_type blocks[BLOCKS];
53 
54 public:
55  static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>, "Block of a bitset must be defined on an unsigned integer type.");
56 
57  CUDA Bitset(): blocks() {}
58 
59  CUDA Bitset(const char* bit_str): blocks() {
60  size_t n = min(strlen(bit_str), MAX_SIZE);
61  for(int i = n-1, j = 0; i >= 0; --i, ++j) {
62  if(bit_str[i] == '1') {
63  set(j);
64  }
65  }
66  }
67 
68  template<class Mem2>
69  CUDA Bitset(const Bitset<N, Mem2, T>& other) {
70  for(int i = 0; i < BLOCKS; ++i) {
71  store(blocks[i], Mem::load(other[i]));
72  }
73  }
74 
75  CUDA static Bitset zeroes() {
76  return Bitset();
77  }

79  CUDA static Bitset ones() {

84  CUDA void store(block_type& block, T data) {
85  Mem::store(block, data);

88  CUDA block_type& block_of(size_t pos) {
89  return blocks[pos / BITS_PER_BLOCK];
90  }
91 
92  CUDA const block_type& block_of(size_t pos) const {
93  return blocks[pos / BITS_PER_BLOCK];
94  }
95 
96  CUDA T load_block(size_t pos) const {
97  return Mem::load(block_of(pos));
98  }
99 
100  CUDA void store_block(size_t pos, T data) {
101  store(block_of(pos), data);
102  }
103 
104  CUDA T bit_of(size_t pos) const {
105  return static_cast<T>(1) << (pos % BITS_PER_BLOCK);
106  }
107 
108 public:
109  CUDA bool test(size_t pos) const {
110  assert(pos < MAX_SIZE);
111  return load_block(pos) & bit_of(pos);
112  }
113 
114  CUDA bool all() const {
115  int i = 0;
116  for(; i < BLOCKS - 1; ++i) {
117  if(Mem::load(blocks[i]) != ONES) {
118  return false;
119  }
120  }
121  return Mem::load(blocks[i]) == ONES_LAST;
122  }
123 
124  CUDA bool any() const {
125  for(int i = 0; i < BLOCKS; ++i) {
126  if(Mem::load(blocks[i]) > ZERO) {
127  return true;
128  }
129  }
130  return false;
131  }
132 
133  CUDA bool none() const {
134  for(int i = 0; i < BLOCKS; ++i) {
135  if(Mem::load(blocks[i]) != ZERO) {
136  return false;
137  }
138  }
139  return true;
140  }
141 
142  CUDA size_t size() const {
143  return MAX_SIZE;
144  }

146  CUDA size_t count() const {
147  size_t bits_at_one = 0;
148  for(int i = 0; i < BLOCKS; ++i){
149  bits_at_one += popcount(blocks[i]);
150  }
151  return bits_at_one;
152  }

154  CUDA Bitset& set() {
155  int i = 0;
156  for(; i < BLOCKS - 1; ++i) {
157  store(blocks[i], ONES);
158  }
159  store(blocks[i], ONES_LAST);
160  return *this;
161  }
162 
163  CUDA Bitset& set(size_t pos) {
164  assert(pos < MAX_SIZE);
165  store_block(pos, load_block(pos) | bit_of(pos));
166  return *this;
167  }
168 
169  CUDA Bitset& set(size_t pos, bool value) {
170  assert(pos < MAX_SIZE);
171  return value ? set(pos) : reset(pos);
172  }
173 
174  CUDA Bitset& reset() {
175  for(int i = 0; i < BLOCKS; ++i) {
176  store(blocks[i], ZERO);
177  }
178  return *this;
179  }
180 
181  CUDA Bitset& reset(size_t pos) {
182  assert(pos < MAX_SIZE);
183  store_block(pos, load_block(pos) & ~bit_of(pos));
184  return *this;
185  }

187  CUDA Bitset& flip() {
188  int i = 0;
189  for(; i < BLOCKS - 1; ++i) {
190  store(blocks[i], ~Mem::load(blocks[i]));
191  }
192  store(blocks[i], ONES_LAST & ~Mem::load(blocks[i]));
193  return *this;
194  }
195 
196  CUDA Bitset& flip(size_t pos) {
197  assert(pos < MAX_SIZE);
198  store_block(pos, load_block(pos) ^ bit_of(pos));
199  return *this;
200  }

245  CUDA int countl_zero() const {
246  int k = BLOCKS - 1;
247  if(blocks[k] > ZERO) {
248  return ::battery::countl_zero(blocks[k]) - PADDING_LAST_BLOCK;
249  }
250  --k;
251  int i = 0;
252  for(; k >= 0 && blocks[k] == ZERO; --k, ++i) {}
253  return i * BITS_PER_BLOCK
254  + BITS_LAST_BLOCK
255  + (k == -1 ? 0 : ::battery::countl_zero(blocks[k]));
256  }
257 
258  CUDA int countl_one() const {
259  int k = BLOCKS - 1;
260  if(blocks[k] != ONES_LAST) {
261  return battery::countl_one((T)(blocks[k] << PADDING_LAST_BLOCK));
262  }
263  --k;
264  int i = 0;
265  for(; k >= 0 && blocks[k] == ONES; --k, ++i) {}
266  return i * BITS_PER_BLOCK
267  + BITS_LAST_BLOCK
268  + (k == -1 ? 0 : ::battery::countl_one(blocks[k]));
269  }
270 
271  CUDA int countr_zero() const {
272  int k = 0;
273  for(; k < BLOCKS && blocks[k] == ZERO; ++k) {}
274  return k * BITS_PER_BLOCK
275  + (k == BLOCKS ? -PADDING_LAST_BLOCK : ::battery::countr_zero(blocks[k]));
276  }
277 
278  CUDA int countr_one() const {
279  int k = 0;
280  for(; k < BLOCKS && blocks[k] == ONES; ++k) {}
281  return k * BITS_PER_BLOCK
282  + (k == BLOCKS ? 0 : ::battery::countr_one(blocks[k]));
283  }
