6 /** \file allocator.hpp
7 We provide several allocators compatible with the data structures provided by this library.
8 The allocators are aimed to be used to distinguish in which memory (shared, global, managed or the "standard" C++ memory) we should allocate data.
9 This allows us to provide uniform interfaces for both host (C++) and device (CUDA) code.
10 
11 As a general comment, be careful to always deallocate the memory from the side you allocated it, e.g., do not allocate on the host then try to deallocate it on the device.
12 To avoid these kind of mistakes, you should use `battery::shared_ptr` when passing data to a CUDA kernel, see the manual for examples.
13 */

24 /** An allocator using the global memory of CUDA.
25 This can be used from both the host and device side, but the memory can only be accessed when in a device function.
26 Beware that allocation and deallocation must occur on the same side. */
27 class global_allocator {
28 public:
29  CUDA void* allocate(size_t bytes) {
30  if(bytes == 0) {
31  return nullptr;
32  }
33  #ifdef __CUDA_ARCH__
34  return std::malloc(bytes);
35  #else
36  void* data;
37  cudaError_t rc = cudaMalloc(&data, bytes);
38  if (rc != cudaSuccess) {
39  printf("Allocation in global memory failed (error = %d)\n", rc);
40  return nullptr;
41  }
42  return data;
43  #endif
44  }
45 
46  CUDA void deallocate(void* data) {
47  #ifdef __CUDA_ARCH__
48  std::free(data);
49  #else
50  cudaFree(data);
51  #endif
52  }
53 };

55 /** An allocator using the managed memory of CUDA when the memory is allocated on the host.
56  * We delegate the allocation to `global_allocator` when the allocation is done on the device (since managed memory cannot be allocated in device functions). */
57 class managed_allocator {
58 public:
59  CUDA void* allocate(size_t bytes) {
60  #ifdef __CUDA_ARCH__

103 /** An allocator allocating memory from a pool of memory.
104 The memory can for instance be the CUDA shared memory.
105 This allocator is rather incomplete as it never frees the memory allocated.
106 It allocates a control block using the "normal" `operator new`, where the address to the pool, its capacity and current offset are stored.
107 */
108 class pool_allocator {
109 
110  struct control_block {
111  unsigned char* mem;
112  size_t capacity;
113  size_t offset;
114  size_t alignment;
115  size_t counter;
116  size_t num_deallocations;
117 
118  CUDA control_block(unsigned char* mem, size_t capacity, size_t alignment)
119  : mem(mem), capacity(capacity), offset(0), alignment(alignment), num_deallocations(0), counter(1)
120  {}
121 
122  CUDA void* allocate(size_t bytes) {
123  // printf("%p: allocate %lu bytes / %lu offset / %lu capacity / %lu alignment / %p current mem\n", mem, bytes, offset, capacity, alignment, &mem[offset]);
124  if(bytes == 0) {
125  return nullptr;
126  }
127  if(size_t(&mem[offset]) % alignment != 0) { // If we are currently unaligned.
128  offset += alignment - (size_t(&mem[offset]) % alignment);
129  }
130  assert(offset + bytes <= capacity);
131  assert((size_t)&mem[offset] % alignment == 0);
132  void* m = (void*)&mem[offset];
133  offset += bytes;
134  return m;
135  }
136 
137  CUDA void deallocate() {
138  num_deallocations++;
139  }
140  };
141 
142  control_block* block;
143 
144 public:
145  CUDA pool_allocator(const pool_allocator& other):
146  block(other.block)
147  {
148  block->counter++;
149  }
150 
151  CUDA pool_allocator(unsigned char* mem, size_t capacity, size_t alignment = alignof(std::max_align_t))
152  : block(::new control_block(mem, capacity, alignment))
153  {}

162  }
