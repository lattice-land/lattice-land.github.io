6 /** \file allocator.hpp
7 We provide several allocators compatible with the data structures provided by this library.
8 The allocators are aimed to be used to distinguish in which memory (shared, global, managed or the "standard" C++ memory) we should allocate data.
9 This allows us to provide uniform interfaces for both host (C++) and device (CUDA) code.
10 
11 As a general comment, be careful to always deallocate the memory from the side you allocated it, e.g., do not allocate on the host then try to deallocate it on the device.
12 To avoid these kind of mistakes, you should use `battery::shared_ptr` when passing data to a CUDA kernel, see the manual for examples.
13 */

24 /** An allocator using the global memory of CUDA.
25 This can be used from both the host and device side, but the memory can only be accessed when in a device function.
26 Beware that allocation and deallocation must occur on the same side. */
27 class global_allocator {
28 public:
29  CUDA NI void* allocate(size_t bytes) {
30  if(bytes == 0) {
31  return nullptr;
32  }
33  #ifdef __CUDA_ARCH__
34  return std::malloc(bytes);
35  #else
36  void* data;
37  cudaError_t rc = cudaMalloc(&data, bytes);
38  if (rc != cudaSuccess) {
39  printf("Allocation in global memory failed (error = %d)\n", rc);
40  return nullptr;
41  }
42  return data;
43  #endif
44  }
45 
46  CUDA NI void deallocate(void* data) {
47  #ifdef __CUDA_ARCH__
48  std::free(data);
49  #else
50  cudaFree(data);
51  #endif
52  }
53 };

55 /** An allocator using the managed memory of CUDA when the memory is allocated on the host.

110 /** An allocator allocating memory from a pool of memory.
111 The memory can for instance be the CUDA shared memory.
112 This allocator is incomplete as it never frees the memory allocated.
113 It allocates a control block using the "normal" `operator new`, where the address to the pool, its capacity and current offset are stored.
114 */
115 class pool_allocator {
116 
117  struct control_block {
118  unsigned char* mem;
119  size_t capacity;
120  size_t offset;
121  size_t alignment;
122  size_t num_deallocations;
123  size_t num_allocations;
124  size_t unaligned_wasted_bytes;
125  size_t counter;
126 
127  CUDA control_block(unsigned char* mem, size_t capacity, size_t alignment)
128  : mem(mem), capacity(capacity), offset(0), alignment(alignment), num_deallocations(0), num_allocations(0), unaligned_wasted_bytes(0), counter(1)
129  {}
130 
131  CUDA void* allocate(size_t bytes) {
132  // printf("%p: allocate %lu bytes / %lu offset / %lu capacity / %lu alignment / %p current mem\n", mem, bytes, offset, capacity, alignment, &mem[offset]);
133  if(bytes == 0) {
134  return nullptr;
135  }
136  size_t smallest_alignment = (bytes > alignment || alignment > 16) ? alignment : impl::power2[bytes];
137  if(size_t(&mem[offset]) % smallest_alignment != 0) { // If we are currently unaligned.
138  size_t old_offset = offset;
139  offset += smallest_alignment - (size_t(&mem[offset]) % smallest_alignment);
140  unaligned_wasted_bytes += (offset - old_offset);
141  }
142  assert(offset + bytes <= capacity);
143  assert((size_t)&mem[offset] % smallest_alignment == 0);
144  void* m = (void*)&mem[offset];
145  offset += bytes;
146  num_allocations++;
147  return m;
148  }
149 
150  CUDA void deallocate(void* ptr) {
151  if(ptr != nullptr) {
152  num_deallocations++;
153  }
154  }
155  };
156 
157  control_block* block;
158 
159 public:
160  CUDA NI pool_allocator(const pool_allocator& other):
161  block(other.block)

164  block->counter++;
165  }
166  }

177  }
178  }
179  }
