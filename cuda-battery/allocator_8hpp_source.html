109 /** An allocator allocating memory from a pool of memory.
110 The memory can for instance be the CUDA shared memory.
111 This allocator is rather incomplete as it never frees the memory allocated.
112 It allocates a control block using the "normal" `operator new`, where the address to the pool, its capacity and current offset are stored.
113 */
114 class PoolAllocator {
115 
116  struct ControlBlock {
117  unsigned char* mem;
118  size_t capacity;
119  size_t offset;
120  size_t alignment;
121  size_t counter;
122 
123  CUDA ControlBlock(unsigned char* mem, size_t capacity, size_t alignment)
124  : mem(mem), capacity(capacity), offset(0), alignment(alignment), counter(1) {}
125 
126  CUDA void* allocate(size_t bytes) {
127  if(bytes == 0) {
128  return nullptr;
129  }
130  offset += (alignment - (((size_t)&mem[offset]) % alignment)) % alignment;
131  assert(offset + bytes < capacity);
132  assert((size_t)&mem[offset] % alignment == 0);
133  void* m = (void*)&mem[offset];
134  offset += bytes;
135  return m;
136  }
137  };
138 
139  ControlBlock* block;
140 
141 public:
142  CUDA PoolAllocator(const PoolAllocator& other):
143  block(other.block)
144  {
145  block->counter++;
146  }
147 
148  CUDA PoolAllocator(unsigned char* mem, size_t capacity, size_t alignment = alignof(std::max_align_t))
149  : block(::new ControlBlock(mem, capacity, alignment))
150  {}
151 
152  CUDA PoolAllocator() = delete;
153 
154  CUDA ~PoolAllocator() {
155  block->counter--;
156  if(block->counter == 0) {
157  ::delete block;
158  }
159  }
160 
161  CUDA size_t align_at(size_t alignment) {
162  size_t old = block->alignment;
163  block->alignment = alignment;
164  return old;
165  }
166 
167  CUDA void* allocate(size_t bytes) {
168  return block->allocate(bytes);
169  }
170 
171  CUDA void deallocate(void*) {}
172

174  printf("Used %lu / %lu\n", block->offset, block->capacity);
175  }
