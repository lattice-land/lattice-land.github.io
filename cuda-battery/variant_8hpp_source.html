17 template<size_t n, typename F, typename... Ts>
18 struct variant_helper_rec<n, F, Ts...>
19 {
20  CUDA inline static void destroy(size_t id, void* data)
21  {
22  if (n == id) {
23  reinterpret_cast<F*>(data)->~F();
24  } else {
25  variant_helper_rec<n + 1, Ts...>::destroy(id, data);
26  }
27  }
28 
29  CUDA inline static void move(size_t id, void* from, void* to)
30  {
31  if (n == id) {
32  new (to) F(std::move(*reinterpret_cast<F*>(from)));
33  } else {
34  variant_helper_rec<n + 1, Ts...>::move(id, from, to);
35  }
36  }
37 
38  CUDA inline static void copy(size_t id, const void* from, void* to)
39  {
40  if (n == id) {
41  new (to) F(*reinterpret_cast<const F*>(from));
42  } else {
43  variant_helper_rec<n + 1, Ts...>::copy(id, from, to);
44  }
45  }
46 
47  CUDA inline static bool equals(size_t id, const void* a, const void* b)
48  {
49  if (n == id) {
50  return *reinterpret_cast<const F*>(a) == *reinterpret_cast<const F*>(b);
51  } else {
52  return variant_helper_rec<n + 1, Ts...>::equals(id, a, b);
53  }
54  }
55 
56  CUDA inline static void print(size_t id, const void* a)
57  {
58  if (n == id) {
59  ::battery::print(*reinterpret_cast<const F*>(a));
60  } else {
61  variant_helper_rec<n + 1, Ts...>::print(id, a);
62  }
63  }
