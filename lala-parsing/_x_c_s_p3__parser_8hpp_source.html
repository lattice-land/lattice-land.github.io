200  battery::shared_ptr<F, Allocator> build_formula() {
201  typename F::Sequence seq;
202  seq.reserve(variables.size() + constraints.size());
203  for(int i = 0; i < variables.size(); ++i) {
204  seq.push_back(std::move(variables[i]));
205  }
206  for(int i = 0; i < constraints.size(); ++i) {
207  seq.push_back(std::move(constraints[i]));
208  }
209  auto f = F::make_nary(lala::AND, std::move(seq));
210  return battery::make_shared<F, Allocator>(std::move(f));

434 template<class Allocator>
435 XCSP3_turbo_callbacks<Allocator>::F XCSP3_turbo_callbacks<Allocator>::make_formula(Node* node) {
436  using namespace lala;
437  Sig sig;
438  switch(node->type) {
439  case OUNDEF:
440  throw std::runtime_error("found an undefined symbol (`OUNDEF`) which seems to indicate a bug in the XCSP3 model.");
441  case ONEG: sig = NEG; break;
442  case OABS: sig = ABS; break;
443  case OSQR: sig = POW; break;
444  case OADD: sig = ADD; break;
445  case OSUB: sig = SUB; break;
446  case OMUL: sig = MUL; break;
447  case ODIV: sig = DIV; break;
448  case OMOD: sig = MOD; break;
449  case OPOW: sig = POW; break;
450  case ODIST: throw std::runtime_error("unsupported DIST"); // return make_dist(node);
451  case OMIN: sig = MIN; break;
452  case OMAX: sig = MAX; break;
453  case OLT: sig = lala::LT; break;
454  case OLE: sig = LEQ; break;
455  case OGE: sig = GEQ; break;
456  case OGT: sig = lala::GT; break;
457  case ONE: sig = NEQ; break;
458  case OEQ: sig = lala::EQ; break;
459  case OSET: throw std::runtime_error("unsupported set"); // return make_set(node);
460  case OIN: sig = lala::IN; break;
461  case ONOTIN: throw std::runtime_error("unsupported notin"); // return make_notin(node);
462  case ONOT: sig = NOT; break;
463  case OAND: sig = AND; break;
464  case OOR: sig = OR; break;
465  case OXOR: sig = XOR; break;
466  case OIFF: sig = EQUIV; break;
467  case OIMP: sig = IMPLY; break;
468  case OIF: sig = ITE; break;
469  case OCARD: sig = CARD; break;
470  case OUNION: sig = UNION; break;
471  case OINTER: sig = INTERSECTION; break;
472  case ODIFF:
473  case OSDIFF:
474  case OHULL:
475  case ODJOINT:
476  throw std::runtime_error("unsupported set operation");
477  case OSUBSET: sig = SUBSET; break;
478  case OSUBSEQ: sig = SUBSETEQ; break;
479  case OSUPSEQ: sig = SUPSETEQ; break;
480  case OSUPSET: sig = SUPSET; break;
481  case OCONVEX:
482  case OFDIV:
483  case OFMOD:
484  case OSQRT:
485  case ONROOT:
486  case OEXP:
487  case OLN:
488  case OLOG:
489  case OSIN:
490  case OCOS:
491  case OTAN:
492  case OASIN:
493  case OACOS:
494  case OATAN:
495  case OSINH:
496  case OCOSH:
497  case OTANH: throw std::runtime_error("unsupported arithmetic operation");
498  case OVAR: return F::make_lvar(UNTYPED, LVar<Allocator>(static_cast<NodeVariable*>(node)->var.c_str()));
499  case OPAR:
500  case OLONG:
501  case ORATIONAL: throw std::runtime_error("unsupported constant");
502  case ODECIMAL: return F::make_z(static_cast<NodeConstant*>(node)->val);
503  case OSYMBOL: throw std::runtime_error("OSYMBOL should not occur in intension constraint");
504  }
505  typename F::Sequence seq;
506  seq.reserve(node->parameters.size());
507  for(int i = 0; i < node->parameters.size(); ++i) {
508  seq.push_back(make_formula(node->parameters[i]));
509  }
510  if(node->type == OSQR) {
511  seq.push_back(F::make_z(2)); // we represent `x SQR` by `x POW 2`.
512  }
513  return F::make_nary(sig, std::move(seq));
514 }
