225 battery::shared_ptr<F, Allocator> build_formula() {
226 typename F::Sequence seq;
227 seq.reserve(variables.size() + constraints.size());
228 for(int i = 0; i < variables.size(); ++i) {
229 seq.push_back(std::move(variables[i]));
230 }
231 for(int i = 0; i < constraints.size(); ++i) {
232 seq.push_back(std::move(constraints[i]));
233 }
234 auto f = F::make_nary(lala::AND, std::move(seq));
235 return battery::make_shared<F, Allocator>(std::move(f));

414void XCSP3_turbo_callbacks<Allocator>::buildVariableInteger(string id, vector<int> &values) {
415 if(debug) {
416 cout << " var " << id << " : ";
417 cout << " ";
418 displayList(values);
419 }
420 lala::LVar<Allocator> lvar(id.c_str());
421 output.add_var(id.c_str());
422 variables.push_back(F::make_exists(UNTYPED, lvar, lala::Sort<Allocator>::Int));
423
424 lala::logic_set<F> intervals;
425 int start = values[0];
426 int end = values[0];
427
428 for (size_t i = 1; i < values.size(); ++i) {
429 if (values[i] == end + 1) {
430 end = values[i];
431 }
432 else {
433 auto t = battery::make_tuple(F::make_z(start),F::make_z(end));
434 intervals.push_back(t);
435 start = values[i];
436 end = values[i];
437 }
438 }
439 auto t = battery::make_tuple(F::make_z(start), F::make_z(end));
440 intervals.push_back(t);
441 constraints.push_back(F::make_binary(F::make_lvar(UNTYPED, lvar), lala::IN, F::make_set(intervals)));
442}

444template<class Allocator>
445void XCSP3_turbo_callbacks<Allocator>::buildConstraintExtension(string id, XVariable *variable, vector<int> &tuples, bool support, bool hasStar) {
446 if(debug) {
447 cout << "\n extension constraint with one variable: " << id << endl;
448 cout << " " <<(*variable) << " "<< (support ? "support" : "conflict") << " nb tuples: " << tuples.size() << " star: " << hasStar << endl;
449 cout << endl;
450 }
451 typename F::Sequence seq;
452 lala::Sig sig = support ? lala::EQ : lala::NEQ;
453 for(int i = 0; i < tuples.size(); ++i) {
454 if(tuples[i] == INT_MAX && hasStar) {
455 return; // constraint satisfied by the presence of a star.
456 }
457 seq.push_back(
458 F::make_binary(F::make_lvar(UNTYPED, lala::LVar<Allocator>(variable->id.c_str())), sig, F::make_z(tuples[i])));
459 }
460 if(seq.size() == 1) {
461 constraints.push_back(seq[0]);
462 }
463 else if(seq.size() > 1){
464 constraints.push_back(F::make_nary(lala::OR, std::move(seq)));
465 }

468template<class Allocator>
469void XCSP3_turbo_callbacks<Allocator>::buildConstraintExtension(string id, vector<XVariable *> list,
470 vector<vector<int> >& tuples, bool support,
471 bool hasStar) {
472 if (debug) {
473 cout << "\n extension constraint : " << id << endl;
474 cout << " " << (support ? "support" : "conflict") << " arity: " << list.size() << " nb tuples: " << tuples.
475 size() << " star: " << hasStar << endl;
476 cout << " ";
477 displayList(list);
478 }
479 extensionAs[id] = tuples;
480 lala::Sig sig = support ? lala::EQ : lala::NEQ;
481 if (table_decomposition == lala::TableDecomposition::ELEMENTS && support) {
482 size_t numVars = tuples[0].size();
483 auto auxVar = F::make_lvar(UNTYPED, buildAuxVariableInteger(tuples.size() -1));
484 for (int i = 0; i < numVars; ++i) {
485 for (int j = 0; j < tuples.size(); ++j) {
486 // index = j ==> varName = value
487 if (!hasStar || tuples[j][i] != INT_MAX) {
488 constraints.push_back(F::make_binary(
489 F::make_binary(auxVar, lala::EQ, F::make_z(j)),
490 lala::IMPLY,
491 F::make_binary(to_lala_logical_variable(list[i]), lala::EQ, F::make_z(tuples[j][i]))));
492 }
493 }
494 }
495 }
496 else if (table_decomposition == lala::TableDecomposition::TABLE_PREDICATE) {
497 FSeq t_seq;
498 t_seq.push_back(F::make_z(tuples.size()));
499 t_seq.push_back(F::make_z(list.size()));
500 for (int i = 0; i < tuples.size(); ++i) {
501 for (int j = 0; j < tuples[i].size(); ++j) {
502 if (hasStar && tuples[i][j] == INT_MAX) {
503 t_seq.push_back(F::make_lvar(UNTYPED, lala::LVar<Allocator>("*")));
504 }
505 else {
506 t_seq.push_back(F::make_z(tuples[i][j]));
507 }
508 }
509 }
510 for (int i = 0; i < list.size(); ++i) {
511 t_seq.push_back(F::make_lvar(UNTYPED, lala::LVar<Allocator>(list[i]->id.c_str())));
512 }
513 constraints.push_back(F::make_nary("tables", std::move(t_seq)));
514 }else {
515 //disjunctive case
516 lala::Sig internal_sig = support ? lala::AND : lala::OR;
517 lala::Sig external_sig = support ? lala::OR : lala::AND;
518 FSeq external_seq;
519 for (int i = 0; i < tuples.size(); ++i) {
520 FSeq internal_seq;
521 for (int j = 0; j < tuples[i].size(); ++j) {
522 // Stars are not added in the conjunction.
523 if (!hasStar || tuples[i][j] != INT_MAX) {
524 internal_seq.push_back(
525 F::make_binary(F::make_lvar(UNTYPED, lala::LVar<Allocator>(list[j]->id.c_str())), sig,
526 F::make_z(tuples[i][j])));
527 }
528 }
529 if (internal_seq.size() == 1) {
530 external_seq.push_back(internal_seq[0]);

562template<class Allocator>
563XCSP3_turbo_callbacks<Allocator>::F XCSP3_turbo_callbacks<Allocator>::make_formula(Node* node) {
564 using namespace lala;
565 Sig sig;
566 switch(node->type) {
567 case OUNDEF:
568 throw std::runtime_error("found an undefined symbol (`OUNDEF`) which seems to indicate a bug in the XCSP3 model.");
569 case ONEG: sig = NEG; break;
570 case OABS: sig = ABS; break;
571 case OSQR: sig = POW; break;
572 case OADD: sig = ADD; break;
573 case OSUB: sig = SUB; break;
574 case OMUL: sig = MUL; break;
575 case ODIV: sig = DIV; break;
576 case OMOD: sig = MOD; break;
577 case OPOW: sig = POW; break;
578 case ODIST: throw std::runtime_error("unsupported DIST"); // return make_dist(node);
579 case OMIN: sig = MIN; break;
580 case OMAX: sig = MAX; break;
581 case OLT: sig = lala::LT; break;
582 case OLE: sig = LEQ; break;
583 case OGE: sig = GEQ; break;
584 case OGT: sig = lala::GT; break;
585 case ONE: sig = NEQ; break;
586 case OEQ: sig = lala::EQ; break;
587 case OSET: throw std::runtime_error("unsupported set"); // return make_set(node);
588 case OIN: sig = lala::IN; break;
589 case ONOTIN: throw std::runtime_error("unsupported notin"); // return make_notin(node);
590 case ONOT: sig = NOT; break;
591 case OAND: sig = AND; break;
592 case OOR: sig = OR; break;
593 case OXOR: sig = XOR; break;
594 case OIFF: sig = EQUIV; break;
595 case OIMP: sig = IMPLY; break;
596 case OIF: sig = ITE; break;
597 case OCARD: sig = CARD; break;
598 case OUNION: sig = UNION; break;
599 case OINTER: sig = INTERSECTION; break;
600 case ODIFF:
601 case OSDIFF:
602 case OHULL:
603 case ODJOINT:
604 throw std::runtime_error("unsupported set operation");
605 case OSUBSET: sig = SUBSET; break;
606 case OSUBSEQ: sig = SUBSETEQ; break;
607 case OSUPSEQ: sig = SUPSETEQ; break;
608 case OSUPSET: sig = SUPSET; break;
609 case OCONVEX:
610 case OFDIV:
611 case OFMOD:
612 case OSQRT:
613 case ONROOT:
614 case OEXP:
615 case OLN:
616 case OLOG:
617 case OSIN:
618 case OCOS:
619 case OTAN:
620 case OASIN:
621 case OACOS:
622 case OATAN:
623 case OSINH:
624 case OCOSH:
625 case OTANH: throw std::runtime_error("unsupported arithmetic operation");
626 case OVAR: return F::make_lvar(UNTYPED, LVar<Allocator>(static_cast<NodeVariable*>(node)->var.c_str()));
627 case OPAR:
628 case OLONG:
629 case ORATIONAL: throw std::runtime_error("unsupported constant");
630 case ODECIMAL: return F::make_z(static_cast<NodeConstant*>(node)->val);
631 case OSYMBOL: throw std::runtime_error("OSYMBOL should not occur in intension constraint");
632 }
633 FSeq seq;
634 seq.reserve(node->parameters.size());
635 for(int i = 0; i < node->parameters.size(); ++i) {
636 seq.push_back(make_formula(node->parameters[i]));
637 }
638 if(node->type == OSQR) {
639 seq.push_back(F::make_z(2)); // we represent `x SQR` by `x POW 2`.
640 }
641 return F::make_nary(sig, std::move(seq));
642}

1551/**
1552 * The callback function related to a cumulative constraint with variable origins, int lengths and int heights
1553 * See http://xcsp.org/specifications/cumulative
1554 *
1555 * Example:
1556 * <cumulative>
1557 * <origins> s1 s2 s3 s4 </origins>
1558 * <lengths> 1 2 3 4 </lengths>
1559 * <heights> 3 4 5 6 </heights>
1560 * <condition> (le,4) </condition>
1561 * </cumulative>
1562 *
1563 * @param id the id (name) of the constraint
1564 * @param origins the vector of origins
1565 * @param lengths the vector of lenghts (here ints)
1566 * @param heights the vector of heights (here ints)
1567 * @param xc the condition (see XCondition)
1568 */
1569// string id, vector<vector<XV
1570template<class Allocator>
1571void XCSP3_turbo_callbacks<Allocator>::buildConstraintCumulative(string id, vector<XVariable *> &origins, vector<int> &lengths, vector<int> &heights, XCondition &xc) {
1572 battery::vector<F> vars;
1573 for(int j=0;j<origins.size();j++) {
1574 for(int i=0;i<origins.size();i++) {
1575 if(i == j) {
1576 continue;
1577 }
1578 auto left = F::make_binary(to_lala_logical_variable(origins[i]), lala::LEQ, to_lala_logical_variable(origins[j]));
1579 auto right = F::make_binary(to_lala_logical_variable(origins[j]), lala::LT, F::make_binary(to_lala_logical_variable(origins[i]), lala::ADD, F::make_z(lengths[i])));
1580 auto andand = F::make_binary(left, lala::AND, right);
1581 auto bij = F::make_lvar(UNTYPED, buildAuxVariableInteger(1));
1582 vars.push_back(bij);
1583 auto equiv = F::make_binary(bij,lala::EQUIV,andand);
1584 constraints.push_back(equiv);
1585 }
1586 }
1587
1588 for(int j=0;j<origins.size();j++) {
1589 auto rkj = F::make_z(heights[j]);
1590 FSeq sum;
1591 sum.push_back(rkj);
1592 for(int i = 0; i < origins.size(); i++) {
1593 if(i == j) {
1594 continue;
1595 }
1596 auto index = i * (origins.size() - 1) + j;
1597 sum.push_back(F::make_binary(F::make_z(heights[i]), lala::MUL, vars[index]));
1598 }
1599 auto left = F::make_nary(lala::ADD, std::move(sum));
1600 constraints.push_back(F::make_binary(left, to_lala_operator(xc.op), to_lala_formula(xc)));
1601 }
1602}
