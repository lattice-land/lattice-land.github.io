27/** The tables abstract domain is designed to represent predicates in extension by listing all their solutions explicitly.
28 * It is inspired by the table global constraint and generalizes it by lifting each element of the table to a lattice element.
29 * We expect `U` to be equally or less expressive than `A::universe_type`, this is because we compute the meet in `A::universe_type` and not in `U`.
30 */

299 template <class F>
300 CUDA void flatten_and(const F& f, typename F::Sequence& conjuncts) const {
301 if(f.is(F::Seq) && f.sig() == AND) {
302 for(int i = 0; i < f.seq().size(); ++i) {
303 flatten_and(f.seq(i), conjuncts);
304 }
305 }
306 else {
307 conjuncts.push_back(f);
308 }
309 }
310
311 template <class F>
312 CUDA void flatten_or(const F& f, typename F::Sequence& disjuncts) const {
313 if(f.is(F::Seq) && f.sig() == OR) {
314 for(int i = 0; i < f.seq().size(); ++i) {
315 flatten_or(f.seq(i), disjuncts);
316 }
317 }
318 else {
319 typename F::Sequence conjuncts{disjuncts.get_allocator()};
320 flatten_and(f, conjuncts);
321 if(conjuncts.size() > 1) {
322 disjuncts.push_back(F::make_nary(AND, std::move(conjuncts)));
323 }
324 else {
325 disjuncts.push_back(std::move(conjuncts[0]));
326 }
327 }
328 }
329
330 template <class F>
331 CUDA F flatten(const F& f, const typename F::allocator_type& alloc) const {
332 typename F::Sequence disjuncts{alloc};
333 flatten_or(f, disjuncts);
334 if(disjuncts.size() > 1) {
335 return F::make_nary(OR, std::move(disjuncts));
336 }
337 else {
338 return std::move(disjuncts[0]);
339 }
340 }

342 template <IKind kind, bool diagnose = false, class F, class Env, class Alloc>
343 CUDA NI bool interpret_atom(
344 battery::vector<AVar, Alloc>& header,
345 battery::vector<battery::vector<local_universe, Alloc>, Alloc>& tell_table,
346 battery::vector<battery::vector<local_universe, Alloc>, Alloc>& ask_table,
347 const F& f, Env& env, IDiagnostics& diagnostics) const
348 {
349 if(num_vars(f) != 1) {
350 RETURN_INTERPRETATION_ERROR("Only unary formulas are supported in the cell of the table.");
351 }
352 else {
353 auto x_opt = var_in(f, env);
354 if(!x_opt.has_value() || !x_opt.value().avar_of(store_aty).has_value()) {
355 RETURN_INTERPRETATION_ERROR("Undeclared variable.");
356 }
357 AVar x = x_opt.value().avar_of(store_aty).value();
358 int idx = 0;
359 for(; idx < header.size() && header[idx] != x; ++idx) {}
360 // If it's a new variable not present in the previous rows, we add it in each row with bottom value.
361 if(idx == header.size()) {
362 header.push_back(x);
363 for(int i = 0; i < tell_table.size(); ++i) {
364 if constexpr(kind == IKind::TELL) {
365 tell_table[i].push_back(local_universe::bot());
366 }
367 ask_table[i].push_back(local_universe::bot());
368 }
369 }
370 local_universe ask_u{local_universe::bot()};
371 if(ginterpret_in<IKind::ASK, diagnose>(f, env, ask_u, diagnostics)) {
372 ask_table.back()[idx].tell(ask_u);
373 if constexpr(kind == IKind::TELL) {
374 local_universe tell_u{local_universe::bot()};
375 if(ginterpret_in<IKind::TELL, diagnose>(f, env, tell_u, diagnostics)) {
376 tell_table.back()[idx].tell(tell_u);
377 }
378 else {
379 return false;
380 }
381 }
382 }
383 else {
384 return false;
385 }
386 }
387 return true;
388 }

391 template <IKind kind, bool diagnose = false, class F, class Env, class I>
392 CUDA NI bool interpret(const F& f2, Env& env, I& intermediate, IDiagnostics& diagnostics) const {
393 F f = flatten(f2, env.get_allocator());
394 using Alloc = typename I::allocator_type;
395 if(f.is(F::Seq) && f.sig() == OR) {
396 battery::vector<AVar, Alloc> header(intermediate.get_allocator());
397 battery::vector<battery::vector<local_universe, Alloc>, Alloc> tell_table(intermediate.get_allocator());
398 battery::vector<battery::vector<local_universe, Alloc>, Alloc> ask_table(intermediate.get_allocator());
399 for(int i = 0; i < f.seq().size(); ++i) {
400 // Add a row in the table.
401 tell_table.push_back(battery::vector<local_universe, Alloc>(header.size(), local_universe::bot(), intermediate.get_allocator()));
402 ask_table.push_back(battery::vector<local_universe, Alloc>(header.size(), local_universe::bot(), intermediate.get_allocator()));
403 if(f.seq(i).is(F::Seq) && f.seq(i).sig() == AND) {
404 const auto& row = f.seq(i).seq();
405 for(int j = 0; j < row.size(); ++j) {
406 size_t error_ctx = diagnostics.num_suberrors();
407 if(!interpret_atom<kind, diagnose>(header, tell_table, ask_table, row[j], env, diagnostics)) {
408 if(!sub->template interpret<kind, diagnose>(f2, env, intermediate.sub, diagnostics)) {
409 return false;
410 }
411 diagnostics.cut(error_ctx);
412 return true;
413 }
414 }
415 }
416 else {
417 size_t error_ctx = diagnostics.num_suberrors();
418 if(!interpret_atom<kind, diagnose>(header, tell_table, ask_table, f.seq(i), env, diagnostics)) {
419 if(!sub->template interpret<kind, diagnose>(f2, env, intermediate.sub, diagnostics)) {
420 return false;
421 }
422 diagnostics.cut(error_ctx);
423 return true;
424 }
425 }
426 }
427 intermediate.headers.push_back(std::move(header));
428 if constexpr(kind == IKind::TELL) {
429 intermediate.tell_tables.push_back(std::move(tell_table));
430 }
431 intermediate.ask_tables.push_back(std::move(ask_table));
432 return true;
433 }
434 else {
435 return sub->template interpret<kind, diagnose>(f, env, intermediate.sub, diagnostics);
436 }
437 }

478 template <class Alloc, class Mem>
479 CUDA this_type& tell(const tell_type<Alloc>& t, BInc<Mem>& has_changed) {
480 if(t.headers.size() > 0) {
481 has_changed.tell_top();
482 }
483 sub->tell(t.sub, has_changed);
484 for(int i = 0; i < t.headers.size(); ++i) {
485 headers.push_back(battery::vector<AVar, allocator_type>(t.headers[i], get_allocator()));
486 for(int j = 0; j < headers[i].size(); ++j) {
487 column_to_table_idx.push_back(i);
488 }
489 table_idx_to_column.push_back(table_idx_to_column.back() + t.tell_tables[i][0].size());
490 tell_tables.push_back(table_type(t.tell_tables[i], get_allocator()));
491 ask_tables.push_back(table_type(t.ask_tables[i], get_allocator()));
492 eliminated_rows.push_back(bitset_type(tell_tables.back().size(), get_allocator()));
493 total_cells += tell_tables.back().size() * tell_tables.back()[0].size();
494 }
495 return *this;
496 }

516 template <class Alloc>
517 CUDA local::BInc ask(const battery::vector<battery::vector<AVar, Alloc>, Alloc>& header,
518 const battery::vector<battery::vector<battery::vector<universe_type, Alloc>, Alloc>, Alloc>& ask_tables) const
519 {
520 for(int i = 0; i < ask_tables.size(); ++i) {
521 bool table_entailed = false;
522 for(int j = 0; j < ask_tables[i].size() && !table_entailed; ++j) {
523 bool row_entailed = true;
524 for(int k = 0; k < ask_tables[i][j].size(); ++k) {
525 if(!(sub->project(headers[i][k]) >= convert<IKind::ASK>(ask_tables[i][j][k]))) {
526 row_entailed = false;
527 break;
528 }
529 }
530 if(row_entailed) {
531 table_entailed = true;
532 }
533 }
534 if(!table_entailed) {
535 return false;
536 }
537 }
538 return true;
539 }

547 template <class Mem>
548 CUDA void crefine(size_t table_num, size_t col, BInc<Mem>& has_changed) {
549 sub_local_universe u{sub_local_universe::top()};
550 for(int j = 0; j < tell_tables[table_num].size(); ++j) {
551 if(!eliminated_rows[table_num].test(j)) {
552 u.dtell(convert<IKind::TELL>(tell_tables[table_num][j][col]));
553 }
554 }
555 sub->tell(headers[table_num][col], u, has_changed);
556 }

558 template <class Mem>
559 CUDA void lrefine(size_t table_num, size_t row, size_t col, BInc<Mem>& has_changed)
560 {
561 if(!eliminated_rows[table_num].test(row))
562 {
563 if(join(convert<IKind::TELL>(tell_tables[table_num][row][col]), sub->project(headers[table_num][col])).is_top()) {
564 eliminated_rows[table_num].set(row);
565 has_changed.tell_top();
566 }
567 }
568 }

570 CUDA size_t num_refinements() const {
571 return
572 sub->num_refinements() +

574 total_cells; // number of lrefine (one per cell).

577 template <class Mem>
578 CUDA void refine(size_t i, BInc<Mem>& has_changed) {
579 assert(i < num_refinements());
580 if(i < sub->num_refinements()) {
581 sub->refine(i, has_changed);
582 }
583 else {
584 i -= sub->num_refinements();
585 if(i < column_to_table_idx.size()) {
586 crefine(column_to_table_idx[i], i - table_idx_to_column[column_to_table_idx[i]], has_changed);
587 }
588 else {
589 i -= column_to_table_idx.size();
590 size_t table_num = 0;
591 bool unfinished = true;
592 // This loop computes the table number of the cell `i`, we avoid stopping the loop earlier to avoid thread divergence.
593 for(int j = 0; j < tell_tables.size(); ++j) {
594 size_t dim_table = tell_tables[j].size() * tell_tables[j][0].size();
595 unfinished &= (i >= dim_table);
596 i -= (unfinished ? dim_table : 0);
597 table_num += (unfinished ? 1 : 0);
598 }
599 lrefine(table_num, i / tell_tables[table_num][0].size(), i % tell_tables[table_num][0].size(), has_changed);
600 }
601 }
