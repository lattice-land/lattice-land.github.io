122 CUDA NI void move_to_next_unassigned_var() {
123 while(current_strategy < strategies.size()) {
124 const auto& vars = current_vars();
125 while(next_unassigned_var < vars.size()) {
126 const auto& v = a->project(vars[next_unassigned_var]);
127 if(v.lb() < dual<LB>(v.ub())) {
128 return;
129 }
130 next_unassigned_var++;
131 }
132 current_strategy++;
133 next_unassigned_var = 0;
134 }
135 }

137 template <class MapFunction>
138 CUDA NI AVar var_map_fold_left(const battery::vector<AVar, allocator_type>& vars, MapFunction op) {
139 int i = next_unassigned_var;
140 int best_i = i;
141 auto best = op(a->project(vars[i]));
142 for(++i; i < vars.size(); ++i) {
143 const auto& u = a->project(vars[i]);
144 if(u.lb() < dual<LB>(u.ub())) {
145 local::BInc has_changed;
146 best.tell(op(u), has_changed);
147 if(has_changed) {
148 best_i = i;
149 }
150 }
151 }
152 return vars[best_i];
153 }

155 CUDA AVar select_var() {
156 const auto& strat = strategies[current_strategy];
157 const auto& vars = strat.vars;
158 switch(strat.var_order) {
159 case VariableOrder::INPUT_ORDER: return vars[next_unassigned_var];
160 case VariableOrder::FIRST_FAIL: return var_map_fold_left(vars, [](const universe_type& u) { return u.width().ub(); });
161 case VariableOrder::ANTI_FIRST_FAIL: return var_map_fold_left(vars, [](const universe_type& u) { return dual<LB>(u.width().ub()); });
162 case VariableOrder::LARGEST: return var_map_fold_left(vars, [](const universe_type& u) { return dual<LB>(u.ub()); });
163 case VariableOrder::SMALLEST: return var_map_fold_left(vars, [](const universe_type& u) { return dual<UB>(u.lb()); });
164 default: printf("unsupported variable order strategy\n"); assert(false); return AVar{};
165 }
166 }

168 template <class U>
169 CUDA NI branch_type make_branch(AVar x, Sig left_op, Sig right_op, const U& u) {
170 if((u.is_top() && U::preserve_top) || (u.is_bot() && U::preserve_bot)) {
171 if(u.is_bot()) {
172 printf("%% WARNING: Cannot currently branch on unbounded variables.\n");
173 }
174 return branch_type{get_allocator()};
175 }
176 using F = TFormula<allocator_type>;
177 using branch_vector = battery::vector<sub_tell_type, allocator_type>;
178 VarEnv<allocator_type> empty_env{};
179 auto k = u.template deinterpret<F>();
180 auto left = a->interpret_tell_in(F::make_binary(F::make_avar(x), left_op, k, x.aty(), get_allocator()), empty_env);
181 auto right = a->interpret_tell_in(F::make_binary(F::make_avar(x), right_op, k, x.aty(), get_allocator()), empty_env);
182 if(left.has_value() && right.has_value()) {
183 return Branch(branch_vector({std::move(left.value()), std::move(right.value())}, get_allocator()));
184 }
185 // Fallback on a more standard split search strategy.
186 // We don't print anything because it might interfere with the output (without lock).
187 else if(left_op != LEQ || right_op != GT) {
188 return make_branch(x, LEQ, GT, u);
189 }
190 else {
191 printf("%% WARNING: The subdomain does not support the underlying search strategy.\n");
192 left.print_diagnostics();
193 right.print_diagnostics();
194 return branch_type{get_allocator()};
195 }
196 }

234 /** This interpretation function expects `f` to be a predicate of the form `search(VariableOrder, ValueOrder, x_1, x_2, ..., x_n)`. */
235 template <class F, class Env>
236 CUDA NI iresult_tell<F, Env> interpret_tell_in(const F& f, Env& env) {
237 if(!(f.is(F::ESeq)
238 && f.eseq().size() >= 3
239 && f.esig() == "search"
240 && f.eseq()[0].is(F::ESeq) && f.eseq()[0].eseq().size() == 0
241 && f.eseq()[1].is(F::ESeq) && f.eseq()[1].eseq().size() == 0))
242 {
243 return iresult_tell<F, Env>(
244 IError<F>(true, name,
245 "We only interpret predicate of the form `search(input_order, indomain_min, x1, ..., xN)`.", f));
246 }
247 VariableOrder var_order;
248 ValueOrder val_order;
249 const auto& var_order_str = f.eseq()[0].esig();
250 const auto& val_order_str = f.eseq()[1].esig();
251 if(var_order_str == "input_order") { var_order = VariableOrder::INPUT_ORDER; }
252 else if(var_order_str == "first_fail") { var_order = VariableOrder::FIRST_FAIL; }
253 else if(var_order_str == "anti_first_fail") { var_order = VariableOrder::ANTI_FIRST_FAIL; }
254 else if(var_order_str == "smallest") { var_order = VariableOrder::SMALLEST; }
255 else if(var_order_str == "largest") { var_order = VariableOrder::LARGEST; }
256 else {
257 return iresult_tell<F, Env>(
258 IError<F>(true, name, "This variable order strategy is unsupported.", f));
259 }
260 if(val_order_str == "indomain_min") { val_order = ValueOrder::MIN; }
261 else if(val_order_str == "indomain_max") { val_order = ValueOrder::MAX; }
262 else if(val_order_str == "indomain_median") { val_order = ValueOrder::MEDIAN; }
263 else if(val_order_str == "indomain_split") { val_order = ValueOrder::SPLIT; }
264 else if(val_order_str == "indomain_reverse_split") { val_order = ValueOrder::REVERSE_SPLIT; }
265 else {
266 return iresult_tell<F, Env>(
267 IError<F>(true, name, "This value order strategy is unsupported.", f));
268 }
269 battery::vector<AVar, typename Env::allocator_type> vars;
270 for(int i = 2; i < f.eseq().size(); ++i) {
271 if(f.eseq(i).is(F::LV)) {
272 auto res_var = env.interpret(f.eseq(i));
273 if(res_var.has_value()) {
274 vars.push_back(res_var.value());
275 }
276 else {
277 return std::move(res_var.error());
278 }
279 }
280 else if(f.eseq(i).is(F::V)) {
281 vars.push_back(f.eseq(i).v());
282 }
283 else {
284 return iresult_tell<F, Env>(
285 IError<F>(true, name, "A non-variable expression is passed to the predicate `search` after the variable and value order strategies.", f.eseq(i)));
286 }
287 }
288 return iresult_tell<F, Env>(
289 tell_type<typename Env::allocator_type>(var_order, val_order, std::move(vars)));
290 }

305 /** Split the next unassigned variable according to the current strategy.
306 * If all variables of the current strategy are assigned, use the next strategy.
307 * If no strategy remains, returns an empty set of branches.
308
309 If the next unassigned variable cannot be split, for instance because the value ordering strategy maps to `bot` or `top`, an empty set of branches is returned.
310 This also means that you cannot suppose `split(a) = {}` to mean `a` is at `top`. */
311 CUDA NI branch_type split() {
312 if(a->is_top()) {
313 return branch_type{get_allocator()};
314 }
315 move_to_next_unassigned_var();
316 if(current_strategy < strategies.size()) {
317 AVar x = select_var();
318 switch(strategies[current_strategy].val_order) {
319 case ValueOrder::MIN: return make_branch(x, EQ, GT, a->project(x).lb());
320 case ValueOrder::MAX: return make_branch(x, EQ, LT, a->project(x).ub());
321 case ValueOrder::MEDIAN: return make_branch(x, EQ, NEQ, a->project(x).median().lb());
322 case ValueOrder::SPLIT: return make_branch(x, LEQ, GT, a->project(x).median().lb());
323 case ValueOrder::REVERSE_SPLIT: return make_branch(x, GT, LEQ, a->project(x).median().lb());
324 default: printf("unsupported value order strategy\n"); assert(false); return branch_type{get_allocator()};
325 }
326 }
327 else {
328 return branch_type{get_allocator()};
329 }
330 }
