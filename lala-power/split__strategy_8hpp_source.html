40/** A split strategy consists of a variable order and value order on a subset of the variables. */
41template <class Allocator>
42struct StrategyType {

143 CUDA NI void move_to_next_unassigned_var() {
144 while(current_strategy < strategies.size()) {
145 const auto& vars = current_vars();
146 int n = vars.empty() ? a->vars() : vars.size();
147 while(next_unassigned_var < n) {
148 universe_type v = (*a)[vars.empty() ? next_unassigned_var : vars[next_unassigned_var].vid()];
149 if(v.lb().value() != v.ub().value()) {
150 return;
151 }
152 next_unassigned_var++;
153 }
154 current_strategy++;
155 next_unassigned_var = 0;
156 }

159 template <class MapFunction>
160 CUDA NI AVar var_map_fold_left(const battery::vector<AVar, allocator_type>& vars, MapFunction op) {
161 int i = next_unassigned_var;
162 int best_i = i;
163 auto best = op((*a)[vars.empty() ? i : vars[i].vid()]);
164 int n = vars.empty() ? a->vars() : vars.size();
165 for(++i; i < n; ++i) {
166 const auto& u = (*a)[vars.empty() ? i : vars[i].vid()];
167 if(u.lb().value() != u.ub().value()) {
168 if(best.meet(op(u))) {
169 best_i = i;
170 }
171 }
172 }
173 return vars.empty() ? AVar{a->aty(), best_i} : vars[best_i];
174 }

176 CUDA AVar select_var() {
177 const auto& strat = strategies[current_strategy];
178 const auto& vars = strat.vars;
179 switch(strat.var_order) {
180 case VariableOrder::INPUT_ORDER: return vars.empty() ? AVar{a->aty(), next_unassigned_var} : vars[next_unassigned_var];
181 case VariableOrder::FIRST_FAIL: return var_map_fold_left(vars, [](const universe_type& u) { return u.width().ub(); });
182 case VariableOrder::ANTI_FIRST_FAIL: return var_map_fold_left(vars, [](const universe_type& u) { return dual_bound<LB>(u.width().ub()); });
183 case VariableOrder::LARGEST: return var_map_fold_left(vars, [](const universe_type& u) { return dual_bound<LB>(u.ub()); });
184 case VariableOrder::SMALLEST: return var_map_fold_left(vars, [](const universe_type& u) { return dual_bound<UB>(u.lb()); });
185 default: printf("BUG: unsupported variable order strategy\n"); assert(false); return AVar{};
186 }
187 }

189 template <class U>
190 CUDA NI branch_type make_branch(AVar x, Sig left_op, Sig right_op, const U& u) {
191 if((u.is_top() && U::preserve_top) || (u.is_bot() && U::preserve_bot)) {
192 if(u.is_top()) {
193 printf("%% WARNING: Cannot currently branch on unbounded variables.\n");
194 }
195 return branch_type(get_allocator());
196 }
197 using F = TFormula<allocator_type>;
198 using branch_vector = battery::vector<sub_tell_type, allocator_type>;
199 VarEnv<allocator_type> empty_env{};
200 auto k = u.template deinterpret<F>();
201 IDiagnostics diagnostics;
202 sub_tell_type left(get_allocator());
203 sub_tell_type right(get_allocator());
204 bool res = a->interpret_tell(F::make_binary(F::make_avar(x), left_op, k, x.aty(), get_allocator()), empty_env, left, diagnostics);
205 res &= a->interpret_tell(F::make_binary(F::make_avar(x), right_op, k, x.aty(), get_allocator()), empty_env, right, diagnostics);
206 if(res) {
207 return Branch(branch_vector({std::move(left), std::move(right)}, get_allocator()));
208 }
209 // Fallback on a more standard split search strategy.
210 // We don't print anything because it might interfere with the output (without lock).
211 else if(left_op != LEQ || right_op != GT) {
212 return make_branch(x, LEQ, GT, u);
213 }
214 else {
215 printf("%% WARNING: The subdomain does not support the underlying search strategy.\n");
216 // Diagnostics mode.
217 a->template interpret_tell<true>(F::make_binary(F::make_avar(x), left_op, k, x.aty(), get_allocator()), empty_env, left, diagnostics);
218 a->template interpret_tell<true>(F::make_binary(F::make_avar(x), right_op, k, x.aty(), get_allocator()), empty_env, right, diagnostics);
219 diagnostics.print();
220 return branch_type(get_allocator());
221 }
222 }

266 /** This interpretation function expects `f` to be a predicate of the form `search(VariableOrder, ValueOrder, x_1, x_2, ..., x_n)`. */
267 template <bool diagnose = false, class F, class Env, class Alloc2>
268 CUDA NI bool interpret_tell(const F& f, Env& env, tell_type<Alloc2>& tell, IDiagnostics& diagnostics) const {
269 if(!(f.is(F::ESeq)
270 && f.eseq().size() >= 2
271 && f.esig() == "search"
272 && f.eseq()[0].is(F::ESeq) && f.eseq()[0].eseq().size() == 0
273 && f.eseq()[1].is(F::ESeq) && f.eseq()[1].eseq().size() == 0))
274 {
275 RETURN_INTERPRETATION_ERROR("SplitStrategy can only interpret predicates of the form `search(input_order, indomain_min, x1, ..., xN)`.");
276 }
277 StrategyType<Alloc2> strat{tell.get_allocator()};
278 const auto& var_order_str = f.eseq()[0].esig();
279 const auto& val_order_str = f.eseq()[1].esig();
280 if(var_order_str == "input_order") { strat.var_order = VariableOrder::INPUT_ORDER; }
281 else if(var_order_str == "first_fail") { strat.var_order = VariableOrder::FIRST_FAIL; }
282 else if(var_order_str == "anti_first_fail") { strat.var_order = VariableOrder::ANTI_FIRST_FAIL; }
283 else if(var_order_str == "smallest") { strat.var_order = VariableOrder::SMALLEST; }
284 else if(var_order_str == "largest") { strat.var_order = VariableOrder::LARGEST; }
285 else {
286 RETURN_INTERPRETATION_ERROR("This variable order strategy is unsupported.");
287 }
288 if(val_order_str == "indomain_min") { strat.val_order = ValueOrder::MIN; }
289 else if(val_order_str == "indomain_max") { strat.val_order = ValueOrder::MAX; }
290 else if(val_order_str == "indomain_median") {
291 printf("WARNING: indomain_median is not supported since we use interval domain. It is replaced by SPLIT");
292 strat.val_order = ValueOrder::SPLIT;
293 }
294 else if(val_order_str == "indomain_split") { strat.val_order = ValueOrder::SPLIT; }
295 else if(val_order_str == "indomain_reverse_split") { strat.val_order = ValueOrder::REVERSE_SPLIT; }
296 else {
297 RETURN_INTERPRETATION_ERROR("This value order strategy is unsupported.");
298 }
299 for(int i = 2; i < f.eseq().size(); ++i) {
300 if(f.eseq(i).is(F::LV)) {
301 strat.vars.push_back(AVar{});
302 if(!env.interpret(f.eseq(i), strat.vars.back(), diagnostics)) {
303 return false;
304 }
305 }
306 else if(f.eseq(i).is(F::V)) {
307 strat.vars.push_back(f.eseq(i).v());
308 }
309 else if(num_vars(f.eseq(i)) > 0) {
310 RETURN_INTERPRETATION_ERROR("The predicate `search` only supports variables or constants, but an expression containing a variable was passed to it.");
311 }
312 // Ignore constant expressions.
313 else {}
314 }
315 tell.push_back(std::move(strat));
316 return true;
317 }

339 /** Split the next unassigned variable according to the current strategy.
340 * If all variables of the current strategy are assigned, use the next strategy.
341 * If no strategy remains, returns an empty set of branches.
342
343 If the next unassigned variable cannot be split, for instance because the value ordering strategy maps to `bot` or `top`, an empty set of branches is returned.
344 This also means that you cannot suppose `split(a) = {}` to mean `a` is at `bot`. */
345 CUDA NI branch_type split() {
346 if(a->is_bot()) {
347 return branch_type(get_allocator());
348 }
349 move_to_next_unassigned_var();
350 if(current_strategy < strategies.size()) {
351 AVar x = select_var();
352 // printf("split on %d (", x.vid()); a->project(x).print(); printf(")\n");
353 switch(strategies[current_strategy].val_order) {
354 case ValueOrder::MIN: return make_branch(x, EQ, GT, a->project(x).lb());
355 case ValueOrder::MAX: return make_branch(x, EQ, LT, a->project(x).ub());
356 // case ValueOrder::MEDIAN: return make_branch(x, EQ, NEQ, a->project(x).median().lb());
357 case ValueOrder::SPLIT: return make_branch(x, LEQ, GT, a->project(x).median().lb());
358 case ValueOrder::REVERSE_SPLIT: return make_branch(x, GT, LEQ, a->project(x).median().lb());
359 default: printf("BUG: unsupported value order strategy\n"); assert(false); return branch_type(get_allocator());
360 }
361 }
362 else {
363 // printf("%% All variables are already assigned, we could not split anymore. It means the underlying abstract domain has not detected the satisfiability or unsatisfiability of the problem although all variables were assigned.\n");
364 return branch_type(get_allocator());
365 }
366 }
