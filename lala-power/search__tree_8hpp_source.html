231 template <class Alloc, class Mem>
232 CUDA this_type& tell(const tell_type<Alloc>& t, BInc<Mem>& has_changed) {
233 if(!is_top()) {
234 if(!is_singleton()) {
235 // We will add `t` to root when we backtrack (see `pop`) and have a chance to modify the root node.
236 root_tell.sub_tells.push_back(t.sub_tell);
237 root_tell.split_tells.push_back(t.split_tell);
238 }
239 // Nevertheless, the rest of the subtree to be explored is still updated with `t`.
240 tell_current(t, has_changed);
241 }
242 return *this;
243 }

245 /** The refinement of `a` and `split` is not done here, and if needed, should be done before calling this method.
246 * This refinement operator performs one iteration of \f$ \mathit{pop} \circ \mathit{push} \circ \mathit{split} \f$.
247 * In short, it initializes `a` to the next node of the search tree.
248 * If we observe `a` from the outside of this domain, `a` can backtrack, and therefore does not always evolve extensively and monotonically.
249 * Nevertheless, the refinement operator of the search tree abstract domain is extensive and monotonic (if split is) over the search tree. */
250 template <class Mem>
251 CUDA void refine(BInc<Mem>& has_changed) {
252 pop(push(split->split()), has_changed);
253 }

260 /** Extract an under-approximation if the last node popped \f$ a \f$ is an under-approximation.
261 * If `B` is a search tree, the under-approximation consists in a search tree \f$ \{a\} \f$ with a single node, in that case, `ua` must be different from `top`. */
262 template <class B>
263 CUDA void extract(B& ua) const {
264 if constexpr(impl::is_search_tree_like<B>::value) {
265 assert(bool(ua.a));
266 a->extract(*ua.a);
267 ua.stack.clear();
268 ua.root_tell.sub_tells.clear();
269 ua.root_tell.split_tells.clear();
270 }
271 else {
272 a->extract(ua);
273 }
274 }

308 /** \return `true` if the current node is pruned, and `false` if a new branch was pushed. */
309 CUDA bool push(branch_type&& branch) {
310 if(branch.size() > 0) {
311 if(is_singleton()) {
312 root = battery::make_tuple(
313 a->snapshot(get_allocator()),
314 split->snapshot(get_allocator()));
315 }
316 stack.push_back(std::move(branch));
317 return false;
318 }
319 return true;
320 }

322 /** If the current node was pruned, we need to backtrack, otherwise we just consider the next node along the branch. */
323 template <class Mem>
324 CUDA void pop(bool pruned, BInc<Mem>& has_changed) {
325 if(!pruned) {
326 commit_left(has_changed);
327 }
328 else {
329 backtrack(has_changed);
330 commit_right(has_changed);
331 }
332 }
333
334 /** Given the current node, commit to the node on the left.
335 * If we are on the root node, we save a snapshot of root before committing to the left node. */
336 template <class Mem>
337 CUDA void commit_left(BInc<Mem>& has_changed) {
338 assert(bool(a));
339 a->tell(stack.back().next(), has_changed);
340 }
341
342 /** We explore the next node of the search tree available (after we backtracked, so it cannot be a left node). */
343 template <class Mem>
344 CUDA void commit_right(BInc<Mem>& has_changed) {
345 if(!stack.empty()) {
346 assert(bool(a));
347 stack.back().next();
348 replay(has_changed);
349 }
350 }
351
352 /** Goes from the current node to root. */
353 template <class Mem>
354 CUDA void backtrack(BInc<Mem>& has_changed) {
355 while(!stack.empty() && !stack.back().has_next()) {
356 stack.pop_back();
357 }
358 if(!stack.empty()) {
359 a->restore(battery::get<0>(root));
360 split->restore(battery::get<1>(root));
361 tell_root(has_changed);
362 }
363 else if(a) {
364 a = nullptr;
365 has_changed.tell_top();
366 }
367 }
368
369 /** We do not always have access to the root node, so formulas that are added to the search tree are kept in `root_tell`.
370 * During backtracking, root is available through `a`, and we add to root the formulas stored until now, so they become automatically available to the remaining nodes in the search tree. */
371 template <class Mem>
372 CUDA void tell_root(BInc<Mem>& has_changed) {
373 if(root_tell.sub_tells.size() > 0 || root_tell.split_tells.size() > 0) {
374 for(int i = 0; i < root_tell.sub_tells.size(); ++i) {
375 a->tell(root_tell.sub_tells[i], has_changed);
376 }
377 for(int i = 0; i < root_tell.split_tells.size(); ++i) {
378 split->tell(root_tell.split_tells[i], has_changed);
379 }
380 root_tell.sub_tells.clear();
381 root_tell.split_tells.clear();
382 // A new snapshot is necessary since we modified `a` and `split`.
383 root = battery::make_tuple(
384 a->snapshot(get_allocator()),
385 split->snapshot(get_allocator()));
386 }
387 }
388
389 /** Goes from `root` to the new node to be explored. */
390 template <class Mem>
391 CUDA void replay(BInc<Mem>& has_changed) {
392 for(int i = 0; i < stack.size(); ++i) {
393 a->tell(stack[i].current(), has_changed);
394 }
395 }
