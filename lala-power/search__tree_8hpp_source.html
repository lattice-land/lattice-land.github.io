178  if(f.is(F::Seq) && f.sig() == AND) {

231  template <class Alloc, class Mem>
232  CUDA this_type& tell(const tell_type<Alloc>& t, BInc<Mem>& has_changed) {
233  if(!is_top()) {
234  if(!is_singleton()) {
235  // We will add `t` to root when we backtrack (see `pop`) and have a chance to modify the root node.
236  for(int i = 0; i < t.sub_tells.size(); ++i) {
237  root_tell.sub_tells.push_back(t.sub_tells[i]);
238  }
239  for(int i = 0; i < t.split_tells.size(); ++i) {
240  root_tell.split_tells.push_back(t.split_tells[i]);
241  }
242  }
243  // Nevertheless, the rest of the subtree to be explored is still updated with `t`.
244  tell_current(t, has_changed);
245  }
246  return *this;
247  }

249  /** The refinement of `a` and `split` is not done here, and if needed, should be done before calling this method.
250  * This refinement operator performs one iteration of \f$ \mathit{pop} \circ \mathit{push} \circ \mathit{split} \f$.
251  * In short, it initializes `a` to the next node of the search tree.
252  * If we observe `a` from the outside of this domain, `a` can backtrack, and therefore does not always evolve extensively and monotonically.
253  * Nevertheless, the refinement operator of the search tree abstract domain is extensive and monotonic (if split is) over the search tree. */
254  template <class Mem>
255  CUDA void refine(BInc<Mem>& has_changed) {
256  pop(push(split->split()), has_changed);
257  }

326  /** If the current node was pruned, we need to backtrack, otherwise we just consider the next node along the branch. */
327  template <class Mem>
328  CUDA void pop(bool pruned, BInc<Mem>& has_changed) {
329  if(!pruned) {
330  commit_left(has_changed);
331  }
332  else {
333  backtrack(has_changed);
334  commit_right(has_changed);
335  }
336  }

338  /** Given the current node, commit to the node on the left.
339  * If we are on the root node, we save a snapshot of root before committing to the left node. */
340  template <class Mem>
341  CUDA void commit_left(BInc<Mem>& has_changed) {
342  assert(bool(a));
343  a->tell(stack.back().next(), has_changed);
344  }

346  /** We explore the next node of the search tree available (after we backtracked, so it cannot be a left node). */
347  template <class Mem>
348  CUDA void commit_right(BInc<Mem>& has_changed) {
349  if(!stack.empty()) {
350  assert(bool(a));
351  stack.back().next();
352  replay(has_changed);
353  }
354  }

356  /** Goes from the current node to root. */
357  template <class Mem>
358  CUDA void backtrack(BInc<Mem>& has_changed) {
359  while(!stack.empty() && !stack.back().has_next()) {
360  stack.pop_back();
361  }
362  if(!stack.empty()) {
363  a->restore(battery::get<0>(root));
364  split->restore(battery::get<1>(root));
365  tell_root(has_changed);
366  }
367  else if(a) {
368  a = nullptr;
369  has_changed.tell_top();
370  }
371  }
372 
373  /** We do not always have access to the root node, so formulas that are added to the search tree are kept in `root_tell`.
374  * During backtracking, root is available through `a`, and we add to root the formulas stored until now, so they become automatically available to the remaining nodes in the search tree. */
375  template <class Mem>
376  CUDA void tell_root(BInc<Mem>& has_changed) {
377  if(root_tell.sub_tells.size() > 0 || root_tell.split_tells.size() > 0) {
378  for(int i = 0; i < root_tell.sub_tells.size(); ++i) {
379  a->tell(root_tell.sub_tells[i], has_changed);
380  }
381  for(int i = 0; i < root_tell.split_tells.size(); ++i) {
382  split->tell(root_tell.split_tells[i], has_changed);
383  }
384  root_tell.sub_tells.clear();
385  root_tell.split_tells.clear();
386  // A new snapshot is necessary since we modified `a` and `split`.
387  root = battery::make_tuple(
388  a->snapshot(get_allocator()),
389  split->snapshot(get_allocator()));
390  }
391  }
392 
393  /** Goes from `root` to the new node to be explored. */
394  template <class Mem>
395  CUDA void replay(BInc<Mem>& has_changed) {
396  for(int i = 0; i < stack.size(); ++i) {
397  a->tell(stack[i].current(), has_changed);
398  }
